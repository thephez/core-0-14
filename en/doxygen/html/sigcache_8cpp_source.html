96  // nMaxCacheSize is unsigned. If -maxsigcachesize is set to zero,
97  // setup_bytes creates the minimum possible cache (2 elements).
98  size_t nMaxCacheSize = std::min(std::max((int64_t)0, GetArg("-maxsigcachesize", DEFAULT_MAX_SIG_CACHE_SIZE)), MAX_MAX_SIG_CACHE_SIZE) * ((size_t) 1 << 20);
99  size_t nElems = signatureCache.setup_bytes(nMaxCacheSize);
100  LogPrintf("Using %zu MiB out of %zu requested for signature cache, able to store %zu elements\n",
