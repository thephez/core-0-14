159  inline void SerializationOp(Stream& s, Operation ser_action) {
160  std::string strVersion;
161  if(ser_action.ForRead()) {
162  READWRITE(strVersion);
163  if (strVersion != SERIALIZATION_VERSION_STRING) {
164  return;
165  }
166  } else {
167  strVersion = SERIALIZATION_VERSION_STRING;
168  READWRITE(strVersion);
169  }
170  // we don't serialize pubkey ids because pubkeys should be
171  // hardcoded or be setted with cmdline or options, should
172  // not reuse pubkeys from previous dashd run
173  READWRITE(mapSporksByHash);
174  READWRITE(mapSporksActive);
175  // we don't serialize private key to prevent its leakage
176  }
