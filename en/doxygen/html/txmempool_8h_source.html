237 class CompareTxMemPoolEntryByDescendantScore
238 {
239 public:
240  bool operator()(const CTxMemPoolEntry& a, const CTxMemPoolEntry& b) const
241  {
242  bool fUseADescendants = UseDescendantScore(a);
243  bool fUseBDescendants = UseDescendantScore(b);
244 
245  double aModFee = fUseADescendants ? a.GetModFeesWithDescendants() : a.GetModifiedFee();
246  double aSize = fUseADescendants ? a.GetSizeWithDescendants() : a.GetTxSize();
247 
248  double bModFee = fUseBDescendants ? b.GetModFeesWithDescendants() : b.GetModifiedFee();
249  double bSize = fUseBDescendants ? b.GetSizeWithDescendants() : b.GetTxSize();
250 
251  // Avoid division by rewriting (a/b > c/d) as (a*d > c*b).
252  double f1 = aModFee * bSize;
253  double f2 = aSize * bModFee;
254 
255  if (f1 == f2) {
256  return a.GetTime() >= b.GetTime();
257  }
258  return f1 < f2;
259  }
260 
261  // Calculate which score to use for an entry (avoiding division).
262  bool UseDescendantScore(const CTxMemPoolEntry &a) const
263  {
264  double f1 = (double)a.GetModifiedFee() * a.GetSizeWithDescendants();
265  double f2 = (double)a.GetModFeesWithDescendants() * a.GetTxSize();
266  return f2 > f1;
267  }
268 };

280  double f2 = (double)b.GetModifiedFee() * a.GetTxSize();
