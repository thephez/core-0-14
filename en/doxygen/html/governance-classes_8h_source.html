94  fValid(false),
95  script(),
96  nAmount(0)
97  {
98  try {
99  CTxDestination dest = addrIn.Get();
100  script = GetScriptForDestination(dest);
101  nAmount = nAmountIn;
102  fValid = true;
103  } catch (std::exception& e) {
104  LogPrintf("CGovernancePayment Payment not valid: addrIn = %s, nAmountIn = %d, what = %s\n",
105  addrIn.ToString(), nAmountIn, e.what());
106  } catch (...) {
107  LogPrintf("CGovernancePayment Payment not valid: addrIn = %s, nAmountIn = %d\n",
108  addrIn.ToString(), nAmountIn);
109  }
110  }
111 
112  bool IsValid() { return fValid; }
113 };
