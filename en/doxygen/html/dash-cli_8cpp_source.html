163 static void http_request_done(struct evhttp_request *req, void *ctx)
164 {
165  HTTPReply *reply = static_cast<HTTPReply*>(ctx);
166 
167  if (req == NULL) {
168  /* If req is NULL, it means an error occurred while connecting: the
169  * error code will have been passed to http_error_cb.
170  */
171  reply->status = 0;
172  return;
173  }
174 
175  reply->status = evhttp_request_get_response_code(req);
176 
177  struct evbuffer *buf = evhttp_request_get_input_buffer(req);
178  if (buf)
179  {
180  size_t size = evbuffer_get_length(buf);
181  const char *data = (const char*)evbuffer_pullup(buf, size);
182  if (data)
183  reply->body = std::string(data, size);
184  evbuffer_drain(buf, size);
185  }
186 }

191  HTTPReply *reply = static_cast<HTTPReply*>(ctx);
192  reply->error = err;

196 UniValue CallRPC(const std::string& strMethod, const UniValue& params)
197 {
198  std::string host = GetArg("-rpcconnect", DEFAULT_RPCCONNECT);
199  int port = GetArg("-rpcport", BaseParams().RPCPort());
200 
201  // Obtain event base
202  raii_event_base base = obtain_event_base();
203 
204  // Synchronously look up hostname
205  raii_evhttp_connection evcon = obtain_evhttp_connection_base(base.get(), host, port);
206  evhttp_connection_set_timeout(evcon.get(), GetArg("-rpcclienttimeout", DEFAULT_HTTP_CLIENT_TIMEOUT));
207 
208  HTTPReply response;
209  raii_evhttp_request req = obtain_evhttp_request(http_request_done, (void*)&response);
210  if (req == NULL)
211  throw std::runtime_error("create http request failed");
212 #if LIBEVENT_VERSION_NUMBER >= 0x02010300
213  evhttp_request_set_error_cb(req.get(), http_error_cb);
214 #endif
215 
216  // Get credentials
217  std::string strRPCUserColonPass;
218  if (GetArg("-rpcpassword", "") == "") {
219  // Try fall back to cookie-based authentication if no password is provided
220  if (!GetAuthCookie(&strRPCUserColonPass)) {
221  throw std::runtime_error(strprintf(
222  _("Could not locate RPC credentials. No authentication cookie could be found, and no rpcpassword is set in the configuration file (%s)"),
223  GetConfigFile(GetArg("-conf", BITCOIN_CONF_FILENAME)).string().c_str()));
224 
225  }
226  } else {
227  strRPCUserColonPass = GetArg("-rpcuser", "") + ":" + GetArg("-rpcpassword", "");
228  }
229 
230  struct evkeyvalq* output_headers = evhttp_request_get_output_headers(req.get());
231  assert(output_headers);
232  evhttp_add_header(output_headers, "Host", host.c_str());
233  evhttp_add_header(output_headers, "Connection", "close");
234  evhttp_add_header(output_headers, "Authorization", (std::string("Basic ") + EncodeBase64(strRPCUserColonPass)).c_str());
235 
236  // Attach request data
237  std::string strRequest = JSONRPCRequestObj(strMethod, params, 1).write() + "\n";
238  struct evbuffer* output_buffer = evhttp_request_get_output_buffer(req.get());
239  assert(output_buffer);
240  evbuffer_add(output_buffer, strRequest.data(), strRequest.size());
241 
242  int r = evhttp_make_request(evcon.get(), req.get(), EVHTTP_REQ_POST, "/");
243  req.release(); // ownership moved to evcon in above call
244  if (r != 0) {
245  throw CConnectionFailed("send http request failed");
246  }
247 
248  event_base_dispatch(base.get());
249 
250  if (response.status == 0)
251  throw CConnectionFailed(strprintf("couldn't connect to server: %s (code %d)\n(make sure server is running and you are connecting to the correct RPC port)", http_errorstring(response.error), response.error));
252  else if (response.status == HTTP_UNAUTHORIZED)
253  throw std::runtime_error("incorrect rpcuser or rpcpassword (authorization failed)");
254  else if (response.status >= 400 && response.status != HTTP_BAD_REQUEST && response.status != HTTP_NOT_FOUND && response.status != HTTP_INTERNAL_SERVER_ERROR)
255  throw std::runtime_error(strprintf("server returned HTTP error %d", response.status));
256  else if (response.body.empty())
257  throw std::runtime_error("no response from server");
258 
259  // Parse reply
260  UniValue valReply(UniValue::VSTR);
261  if (!valReply.read(response.body))
262  throw std::runtime_error("couldn't parse reply from server");
263  const UniValue& reply = valReply.get_obj();
264  if (reply.empty())
265  throw std::runtime_error("expected reply to have result, error and id properties");
266 
267  return reply;
268 }

270 int CommandLineRPC(int argc, char *argv[])
271 {
272  std::string strPrint;
273  int nRet = 0;
274  try {
275  // Skip switches
276  while (argc > 1 && IsSwitchChar(argv[1][0])) {
277  argc--;
278  argv++;
279  }
280  std::vector<std::string> args = std::vector<std::string>(&argv[1], &argv[argc]);
281  if (GetBoolArg("-stdin", false)) {
282  // Read one arg per line from stdin and append
283  std::string line;
284  while (std::getline(std::cin,line))
285  args.push_back(line);
286  }
287  if (args.size() < 1)
288  throw std::runtime_error("too few parameters (need at least command)");
289  std::string strMethod = args[0];
290  args.erase(args.begin()); // Remove trailing method name from arguments vector
291 
292  UniValue params;
293  if(GetBoolArg("-named", DEFAULT_NAMED)) {
294  params = RPCConvertNamedValues(strMethod, args);
295  } else {
296  params = RPCConvertValues(strMethod, args);
297  }
298 
299  // Execute and handle connection failures with -rpcwait
300  const bool fWait = GetBoolArg("-rpcwait", false);
301  do {
302  try {
303  const UniValue reply = CallRPC(strMethod, params);
304 
305  // Parse reply
306  const UniValue& result = find_value(reply, "result");
307  const UniValue& error = find_value(reply, "error");
308 
309  if (!error.isNull()) {
310  // Error
311  int code = error["code"].get_int();
312  if (fWait && code == RPC_IN_WARMUP)
313  throw CConnectionFailed("server in warmup");
314  strPrint = "error: " + error.write();
315  nRet = abs(code);
316  if (error.isObject())
317  {
318  UniValue errCode = find_value(error, "code");
319  UniValue errMsg = find_value(error, "message");
320  strPrint = errCode.isNull() ? "" : "error code: "+errCode.getValStr()+"\n";
321 
322  if (errMsg.isStr())
323  strPrint += "error message:\n"+errMsg.get_str();
324  }
325  } else {
326  // Result
327  if (result.isNull())
328  strPrint = "";
329  else if (result.isStr())
330  strPrint = result.get_str();
331  else
332  strPrint = result.write(2);
333  }
334  // Connection succeeded, no need to retry.
335  break;
336  }
337  catch (const CConnectionFailed&) {
338  if (fWait)
339  MilliSleep(1000);
340  else
341  throw;
342  }
343  } while (fWait);
344  }
345  catch (const boost::thread_interrupted&) {
346  throw;
347  }
348  catch (const std::exception& e) {
349  strPrint = std::string("error: ") + e.what();
350  nRet = EXIT_FAILURE;
351  }
352  catch (...) {
353  PrintExceptionContinue(NULL, "CommandLineRPC()");
354  throw;
355  }
356 
357  if (strPrint != "") {
358  fprintf((nRet == 0 ? stdout : stderr), "%s\n", strPrint.c_str());
359  }
360  return nRet;
361 }
