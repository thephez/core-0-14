56  template<typename Stream>
57  void Serialize(Stream &s) const {
58  assert(!IsSpent());
59  uint32_t code = nHeight * 2 + fCoinBase;
60  ::Serialize(s, VARINT(code));

64  template<typename Stream>
65  void Unserialize(Stream &s) {
66  uint32_t code = 0;
67  ::Unserialize(s, VARINT(code));
68  nHeight = code >> 1;
69  fCoinBase = code & 1;
70  ::Unserialize(s, REF(CTxOutCompressor(out)));
71  }

101 struct CCoinsCacheEntry
102 {
103  Coin coin; // The actual cached data.
104  unsigned char flags;
105 
106  enum Flags {
107  DIRTY = (1 << 0), // This cache entry is potentially different from the version in the parent view.
108  FRESH = (1 << 1), // The parent view does not have this entry (or it is pruned).
109  /* Note that FRESH is a performance optimization with which we can
110  * erase coins that are fully spent if we know we do not need to
111  * flush the changes to the parent cache. It is always safe to
112  * not mark FRESH if that condition is not guaranteed.
113  */
114  };
115 
116  CCoinsCacheEntry() : flags(0) {}
117  explicit CCoinsCacheEntry(Coin&& coin_) : coin(std::move(coin_)), flags(0) {}
118 };
