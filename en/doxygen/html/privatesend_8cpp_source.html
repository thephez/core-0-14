324 // check to make sure the collateral provided by the client is valid
325 bool CPrivateSend::IsCollateralValid(const CTransaction& txCollateral)
326 {
327  if (txCollateral.vout.empty()) return false;
328  if (txCollateral.nLockTime != 0) return false;
329 
330  CAmount nValueIn = 0;
331  CAmount nValueOut = 0;
332 
333  for (const auto& txout : txCollateral.vout) {
334  nValueOut += txout.nValue;
335 
336  if (!txout.scriptPubKey.IsPayToPublicKeyHash() && !txout.scriptPubKey.IsUnspendable()) {
337  LogPrintf("CPrivateSend::IsCollateralValid -- Invalid Script, txCollateral=%s", txCollateral.ToString());
338  return false;
339  }
340  }
341 
342  for (const auto& txin : txCollateral.vin) {
343  Coin coin;
344  if (!GetUTXOCoin(txin.prevout, coin)) {
345  LogPrint("privatesend", "CPrivateSend::IsCollateralValid -- Unknown inputs in collateral transaction, txCollateral=%s", txCollateral.ToString());
346  return false;
347  }
348  nValueIn += coin.out.nValue;
349  }
350 
351  //collateral transactions are required to pay out a small fee to the miners
352  if (nValueIn - nValueOut < GetCollateralAmount()) {
353  LogPrint("privatesend", "CPrivateSend::IsCollateralValid -- did not include enough fees in transaction: fees: %d, txCollateral=%s", nValueOut - nValueIn, txCollateral.ToString());
354  return false;
355  }
356 
357  LogPrint("privatesend", "CPrivateSend::IsCollateralValid -- %s", txCollateral.ToString());
358 
359  {
360  LOCK(cs_main);
361  CValidationState validationState;
362  if (!AcceptToMemoryPool(mempool, validationState, MakeTransactionRef(txCollateral), false, NULL, false, maxTxFee, true)) {
363  LogPrint("privatesend", "CPrivateSend::IsCollateralValid -- didn't pass AcceptToMemoryPool()\n");
364  return false;
365  }
366  }
367 
368  return true;
369 }

387 std::string CPrivateSend::GetDenominationsToString(int nDenom)
388 {
389  std::string strDenom = "";
390  int nMaxDenoms = vecStandardDenominations.size();
391 
392  if (nDenom >= (1 << nMaxDenoms)) {
393  return "out-of-bounds";
394  }
395 
396  for (int i = 0; i < nMaxDenoms; ++i) {
397  if (nDenom & (1 << i)) {
398  strDenom += (strDenom.empty() ? "" : "+") + FormatMoney(vecStandardDenominations[i]);
399  }
400  }
401 
402  if (strDenom.empty()) {
403  return "non-denom";
404  }
405 
406  return strDenom;
407 }

409 /* Return a bitshifted integer representing the denominations in this list
410  Function returns as follows (for 4 denominations):
411  ( bit on if present )
412  10 - bit 0
413  1 - bit 1
414  .1 - bit 2
415  .01 - bit 3
416  non-denom - 0, all bits off
417 */
418 int CPrivateSend::GetDenominations(const std::vector<CTxOut>& vecTxOut, bool fSingleRandomDenom)
419 {
420  std::vector<std::pair<CAmount, int> > vecDenomUsed;
421 
422  // make a list of denominations, with zero uses
423  for (const auto& nDenomValue : vecStandardDenominations)
424  vecDenomUsed.push_back(std::make_pair(nDenomValue, 0));
425 
426  // look for denominations and update uses to 1
427  for (const auto& txout : vecTxOut) {
428  bool found = false;
429  for (auto& s : vecDenomUsed) {
430  if (txout.nValue == s.first) {
431  s.second = 1;
432  found = true;
433  }
434  }
435  if (!found) return 0;
436  }
437 
438  int nDenom = 0;
439  int c = 0;
440  // if the denomination is used, shift the bit on
441  for (const auto& s : vecDenomUsed) {
442  int bit = (fSingleRandomDenom ? GetRandInt(2) : 1) & s.second;
443  nDenom |= bit << c++;
444  if (fSingleRandomDenom && bit) break; // use just one random denomination
445  }
446 
447  return nDenom;
448 }
