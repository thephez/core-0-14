30 void ge_equals_ge(const secp256k1_ge *a, const secp256k1_ge *b) {
31  CHECK(a->infinity == b->infinity);
32  if (a->infinity) {
33  return;
34  }
35  CHECK(secp256k1_fe_equal_var(&a->x, &b->x));
36  CHECK(secp256k1_fe_equal_var(&a->y, &b->y));
37 }

39 void ge_equals_gej(const secp256k1_ge *a, const secp256k1_gej *b) {
40  secp256k1_fe z2s;
41  secp256k1_fe u1, u2, s1, s2;
42  CHECK(a->infinity == b->infinity);
43  if (a->infinity) {
44  return;
45  }
46  /* Check a.x * b.z^2 == b.x && a.y * b.z^3 == b.y, to avoid inverses. */
47  secp256k1_fe_sqr(&z2s, &b->z);
48  secp256k1_fe_mul(&u1, &a->x, &z2s);
49  u2 = b->x; secp256k1_fe_normalize_weak(&u2);
50  secp256k1_fe_mul(&s1, &a->y, &z2s); secp256k1_fe_mul(&s1, &s1, &b->z);
51  s2 = b->y; secp256k1_fe_normalize_weak(&s2);
52  CHECK(secp256k1_fe_equal_var(&u1, &u2));
53  CHECK(secp256k1_fe_equal_var(&s1, &s2));
54 }

67 int secp256k1_nonce_function_smallint(unsigned char *nonce32, const unsigned char *msg32,
68  const unsigned char *key32, const unsigned char *algo16,
69  void *data, unsigned int attempt) {
70  secp256k1_scalar s;
71  int *idata = data;
72  (void)msg32;
73  (void)key32;
74  (void)algo16;
75  /* Some nonces cannot be used because they'd cause s and/or r to be zero.
76  * The signing function has retry logic here that just re-calls the nonce
77  * function with an increased `attempt`. So if attempt > 0 this means we
78  * need to change the nonce to avoid an infinite loop. */
79  if (attempt > 0) {
80  (*idata)++;
81  }
82  secp256k1_scalar_set_int(&s, *idata);
83  secp256k1_scalar_get_b32(nonce32, &s);
84  return 1;
85 }

98 void test_exhaustive_addition(const secp256k1_ge *group, const secp256k1_gej *groupj, int order) {
99  int i, j;
100 
101  /* Sanity-check (and check infinity functions) */
102  CHECK(secp256k1_ge_is_infinity(&group[0]));
103  CHECK(secp256k1_gej_is_infinity(&groupj[0]));
104  for (i = 1; i < order; i++) {
105  CHECK(!secp256k1_ge_is_infinity(&group[i]));
106  CHECK(!secp256k1_gej_is_infinity(&groupj[i]));
107  }
108 
109  /* Check all addition formulae */
110  for (j = 0; j < order; j++) {
111  secp256k1_fe fe_inv;
112  secp256k1_fe_inv(&fe_inv, &groupj[j].z);
113  for (i = 0; i < order; i++) {
114  secp256k1_ge zless_gej;
115  secp256k1_gej tmp;
116  /* add_var */
117  secp256k1_gej_add_var(&tmp, &groupj[i], &groupj[j], NULL);
118  ge_equals_gej(&group[(i + j) % order], &tmp);
119  /* add_ge */
120  if (j > 0) {
121  secp256k1_gej_add_ge(&tmp, &groupj[i], &group[j]);
122  ge_equals_gej(&group[(i + j) % order], &tmp);
123  }
124  /* add_ge_var */
125  secp256k1_gej_add_ge_var(&tmp, &groupj[i], &group[j], NULL);
126  ge_equals_gej(&group[(i + j) % order], &tmp);
127  /* add_zinv_var */
128  zless_gej.infinity = groupj[j].infinity;
129  zless_gej.x = groupj[j].x;
130  zless_gej.y = groupj[j].y;
131  secp256k1_gej_add_zinv_var(&tmp, &groupj[i], &zless_gej, &fe_inv);
132  ge_equals_gej(&group[(i + j) % order], &tmp);
133  }
134  }
135 
136  /* Check doubling */
137  for (i = 0; i < order; i++) {
138  secp256k1_gej tmp;
139  if (i > 0) {
140  secp256k1_gej_double_nonzero(&tmp, &groupj[i], NULL);
141  ge_equals_gej(&group[(2 * i) % order], &tmp);
142  }
143  secp256k1_gej_double_var(&tmp, &groupj[i], NULL);
144  ge_equals_gej(&group[(2 * i) % order], &tmp);
145  }
146 
147  /* Check negation */
148  for (i = 1; i < order; i++) {
149  secp256k1_ge tmp;
150  secp256k1_gej tmpj;
151  secp256k1_ge_neg(&tmp, &group[i]);
152  ge_equals_ge(&group[order - i], &tmp);
153  secp256k1_gej_neg(&tmpj, &groupj[i]);
154  ge_equals_gej(&group[order - i], &tmpj);
155  }
156 }

190 void test_exhaustive_verify(const secp256k1_context *ctx, const secp256k1_ge *group, int order) {
191  int s, r, msg, key;
192  for (s = 1; s < order; s++) {
193  for (r = 1; r < order; r++) {
194  for (msg = 1; msg < order; msg++) {
195  for (key = 1; key < order; key++) {
196  secp256k1_ge nonconst_ge;
197  secp256k1_ecdsa_signature sig;
198  secp256k1_pubkey pk;
199  secp256k1_scalar sk_s, msg_s, r_s, s_s;
200  secp256k1_scalar s_times_k_s, msg_plus_r_times_sk_s;
201  int k, should_verify;
202  unsigned char msg32[32];
203 
204  secp256k1_scalar_set_int(&s_s, s);
205  secp256k1_scalar_set_int(&r_s, r);
206  secp256k1_scalar_set_int(&msg_s, msg);
207  secp256k1_scalar_set_int(&sk_s, key);
208 
209  /* Verify by hand */
210  /* Run through every k value that gives us this r and check that *one* works.
211  * Note there could be none, there could be multiple, ECDSA is weird. */
212  should_verify = 0;
213  for (k = 0; k < order; k++) {
214  secp256k1_scalar check_x_s;
215  r_from_k(&check_x_s, group, k);
216  if (r_s == check_x_s) {
217  secp256k1_scalar_set_int(&s_times_k_s, k);
218  secp256k1_scalar_mul(&s_times_k_s, &s_times_k_s, &s_s);
219  secp256k1_scalar_mul(&msg_plus_r_times_sk_s, &r_s, &sk_s);
220  secp256k1_scalar_add(&msg_plus_r_times_sk_s, &msg_plus_r_times_sk_s, &msg_s);
221  should_verify |= secp256k1_scalar_eq(&s_times_k_s, &msg_plus_r_times_sk_s);
222  }
223  }
224  /* nb we have a "high s" rule */
225  should_verify &= !secp256k1_scalar_is_high(&s_s);
226 
227  /* Verify by calling verify */
228  secp256k1_ecdsa_signature_save(&sig, &r_s, &s_s);
229  memcpy(&nonconst_ge, &group[sk_s], sizeof(nonconst_ge));
230  secp256k1_pubkey_save(&pk, &nonconst_ge);
231  secp256k1_scalar_get_b32(msg32, &msg_s);
232  CHECK(should_verify ==
233  secp256k1_ecdsa_verify(ctx, &sig, msg32, &pk));
234  }
235  }
236  }
237  }

240 void test_exhaustive_sign(const secp256k1_context *ctx, const secp256k1_ge *group, int order) {
241  int i, j, k;
242 
243  /* Loop */
244  for (i = 1; i < order; i++) { /* message */
245  for (j = 1; j < order; j++) { /* key */
246  for (k = 1; k < order; k++) { /* nonce */
247  secp256k1_ecdsa_signature sig;
248  secp256k1_scalar sk, msg, r, s, expected_r;
249  unsigned char sk32[32], msg32[32];
250  secp256k1_scalar_set_int(&msg, i);
251  secp256k1_scalar_set_int(&sk, j);
252  secp256k1_scalar_get_b32(sk32, &sk);
253  secp256k1_scalar_get_b32(msg32, &msg);
254 
255  secp256k1_ecdsa_sign(ctx, &sig, msg32, sk32, secp256k1_nonce_function_smallint, &k);
256 
257  secp256k1_ecdsa_signature_load(ctx, &r, &s, &sig);
258  /* Note that we compute expected_r *after* signing -- this is important
259  * because our nonce-computing function function might change k during
260  * signing. */
261  r_from_k(&expected_r, group, k);
262  CHECK(r == expected_r);
263  CHECK((k * s) % order == (i + r * j) % order ||
264  (k * (EXHAUSTIVE_TEST_ORDER - s)) % order == (i + r * j) % order);
265  }
266  }
267  }

269  /* We would like to verify zero-knowledge here by counting how often every
270  * possible (s, r) tuple appears, but because the group order is larger
271  * than the field order, when coercing the x-values to scalar values, some
272  * appear more often than others, so we are actually not zero-knowledge.
273  * (This effect also appears in the real code, but the difference is on the
274  * order of 1/2^128th the field order, so the deviation is not useful to a
275  * computationally bounded attacker.)

279 int main(void) {
280  int i;
281  secp256k1_gej groupj[EXHAUSTIVE_TEST_ORDER];
282  secp256k1_ge group[EXHAUSTIVE_TEST_ORDER];
283 
284  /* Build context */
285  secp256k1_context *ctx = secp256k1_context_create(SECP256K1_CONTEXT_SIGN | SECP256K1_CONTEXT_VERIFY);
286 
287  /* TODO set z = 1, then do num_tests runs with random z values */
288 
289  /* Generate the entire group */
290  secp256k1_gej_set_infinity(&groupj[0]);
291  secp256k1_ge_set_gej(&group[0], &groupj[0]);
292  for (i = 1; i < EXHAUSTIVE_TEST_ORDER; i++) {
293  /* Set a different random z-value for each Jacobian point */
294  secp256k1_fe z;
295  random_fe(&z);
296 
297  secp256k1_gej_add_ge(&groupj[i], &groupj[i - 1], &secp256k1_ge_const_g);
298  secp256k1_ge_set_gej(&group[i], &groupj[i]);
299  secp256k1_gej_rescale(&groupj[i], &z);
300 
301  /* Verify against ecmult_gen */
302  {
303  secp256k1_scalar scalar_i;
304  secp256k1_gej generatedj;
305  secp256k1_ge generated;
306 
307  secp256k1_scalar_set_int(&scalar_i, i);
308  secp256k1_ecmult_gen(&ctx->ecmult_gen_ctx, &generatedj, &scalar_i);
309  secp256k1_ge_set_gej(&generated, &generatedj);
310 
311  CHECK(group[i].infinity == 0);
312  CHECK(generated.infinity == 0);
313  CHECK(secp256k1_fe_equal_var(&generated.x, &group[i].x));
314  CHECK(secp256k1_fe_equal_var(&generated.y, &group[i].y));
315  }
316  }
317 
318  /* Run the tests */
319 #ifdef USE_ENDOMORPHISM
320  test_exhaustive_endomorphism(group, EXHAUSTIVE_TEST_ORDER);
321 #endif
322  test_exhaustive_addition(group, groupj, EXHAUSTIVE_TEST_ORDER);
323  test_exhaustive_ecmult(ctx, group, groupj, EXHAUSTIVE_TEST_ORDER);
324  test_exhaustive_sign(ctx, group, EXHAUSTIVE_TEST_ORDER);
325  test_exhaustive_verify(ctx, group, EXHAUSTIVE_TEST_ORDER);
326 
327  return 0;
328 }
