14 bool CheckCbTx(const CTransaction& tx, const CBlockIndex* pindexPrev, CValidationState& state)
15 {
16  if (tx.nType != TRANSACTION_COINBASE) {
17  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-type");
18  }
19 
20  if (!tx.IsCoinBase()) {
21  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-invalid");
22  }
23 
24  CCbTx cbTx;
25  if (!GetTxPayload(tx, cbTx)) {
26  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-payload");
27  }
28 
29  if (cbTx.nVersion == 0 || cbTx.nVersion > CCbTx::CURRENT_VERSION) {
30  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-version");
31  }
32 
33  if (pindexPrev && pindexPrev->nHeight + 1 != cbTx.nHeight) {
34  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-height");
35  }
36 
37  return true;
38 }

40 // This can only be done after the block has been fully processed, as otherwise we won't have the finished MN list
41 bool CheckCbTxMerkleRootMNList(const CBlock& block, const CBlockIndex* pindex, CValidationState& state)
42 {
43  if (block.vtx[0]->nType != TRANSACTION_COINBASE) {
44  return true;
45  }
46 
47  CCbTx cbTx;
48  if (!GetTxPayload(*block.vtx[0], cbTx)) {
49  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-payload");
50  }
51 
52  if (pindex) {
53  uint256 calculatedMerkleRoot;
54  if (!CalcCbTxMerkleRootMNList(block, pindex->pprev, calculatedMerkleRoot, state)) {
55  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-mnmerkleroot");
56  }
57  if (calculatedMerkleRoot != cbTx.merkleRootMNList) {
58  return state.DoS(100, false, REJECT_INVALID, "bad-cbtx-mnmerkleroot");
59  }
60  }
61 
62  return true;
63 }

65 bool CalcCbTxMerkleRootMNList(const CBlock& block, const CBlockIndex* pindexPrev, uint256& merkleRootRet, CValidationState& state)
66 {
67  LOCK(deterministicMNManager->cs);
68 
69  CDeterministicMNList tmpMNList;
70  if (!deterministicMNManager->BuildNewListFromBlock(block, pindexPrev, state, tmpMNList, false)) {
71  return false;
72  }
73 
74  CSimplifiedMNList sml(tmpMNList);
75 
76  bool mutated = false;
77  merkleRootRet = sml.CalcMerkleRoot(&mutated);
78  return !mutated;
79 }
