60 void CActiveDeterministicMasternodeManager::Init()
61 {
62  LOCK(cs_main);
63 
64  if (!fMasternodeMode) return;
65 
66  if (!deterministicMNManager->IsDeterministicMNsSporkActive()) return;
67 
68  // Check that our local network configuration is correct
69  if (!fListen) {
70  // listen option is probably overwritten by smth else, no good
71  state = MASTERNODE_ERROR;
72  strError = "Masternode must accept connections from outside. Make sure listen configuration option is not overwritten by some another parameter.";
73  LogPrintf("CActiveDeterministicMasternodeManager::Init -- ERROR: %s\n", strError);
74  return;
75  }
76 
77  if (!GetLocalAddress(activeMasternodeInfo.service)) {
78  state = MASTERNODE_ERROR;
79  return;
80  }
81 
82  CDeterministicMNList mnList = deterministicMNManager->GetListAtChainTip();
83 
84  CDeterministicMNCPtr dmn = mnList.GetMNByOperatorKey(*activeMasternodeInfo.blsPubKeyOperator);
85  if (!dmn) {
86  // MN not appeared on the chain yet
87  return;
88  }
89 
90  if (!mnList.IsMNValid(dmn->proTxHash)) {
91  if (mnList.IsMNPoSeBanned(dmn->proTxHash)) {
92  state = MASTERNODE_POSE_BANNED;
93  } else {
94  state = MASTERNODE_REMOVED;
95  }
96  return;
97  }
98 
99  mnListEntry = dmn;
100 
101  LogPrintf("CActiveDeterministicMasternodeManager::Init -- proTxHash=%s, proTx=%s\n", mnListEntry->proTxHash.ToString(), mnListEntry->ToString());
102 
103  if (activeMasternodeInfo.service != mnListEntry->pdmnState->addr) {
104  state = MASTERNODE_ERROR;
105  strError = "Local address does not match the address from ProTx";
106  LogPrintf("CActiveDeterministicMasternodeManager::Init -- ERROR: %s", strError);
107  return;
108  }
109 
110  if (Params().NetworkIDString() != CBaseChainParams::REGTEST) {
111  // Check socket connectivity
112  LogPrintf("CActiveDeterministicMasternodeManager::Init -- Checking inbound connection to '%s'\n", activeMasternodeInfo.service.ToString());
113  SOCKET hSocket;
114  bool fConnected = ConnectSocket(activeMasternodeInfo.service, hSocket, nConnectTimeout) && IsSelectableSocket(hSocket);
115  CloseSocket(hSocket);
116 
117  if (!fConnected) {
118  state = MASTERNODE_ERROR;
119  strError = "Could not connect to " + activeMasternodeInfo.service.ToString();
120  LogPrintf("CActiveDeterministicMasternodeManager::Init -- ERROR: %s\n", strError);
121  return;
122  }
123  }
124 
125  activeMasternodeInfo.proTxHash = mnListEntry->proTxHash;
126  activeMasternodeInfo.outpoint = mnListEntry->collateralOutpoint;
127  state = MASTERNODE_READY;
128 }

130 void CActiveDeterministicMasternodeManager::UpdatedBlockTip(const CBlockIndex* pindexNew, const CBlockIndex* pindexFork, bool fInitialDownload)
131 {
132  LOCK(cs_main);
133 
134  if (!fMasternodeMode) return;
135 
136  if (!deterministicMNManager->IsDeterministicMNsSporkActive(pindexNew->nHeight)) return;
137 
138  if (state == MASTERNODE_READY) {
139  auto mnList = deterministicMNManager->GetListForBlock(pindexNew->GetBlockHash());
140  if (!mnList.IsMNValid(mnListEntry->proTxHash)) {
141  // MN disappeared from MN list
142  state = MASTERNODE_REMOVED;
143  activeMasternodeInfo.proTxHash = uint256();
144  activeMasternodeInfo.outpoint.SetNull();
145  // MN might have reappeared in same block with a new ProTx
146  Init();
147  } else if (mnList.GetMN(mnListEntry->proTxHash)->pdmnState->pubKeyOperator != mnListEntry->pdmnState->pubKeyOperator) {
148  // MN operator key changed or revoked
149  state = MASTERNODE_OPERATOR_KEY_CHANGED;
150  activeMasternodeInfo.proTxHash = uint256();
151  activeMasternodeInfo.outpoint.SetNull();
152  // MN might have reappeared in same block with a new ProTx
153  Init();
154  }
155  } else {
156  // MN might have (re)appeared with a new ProTx or we've found some peers and figured out our local address
157  Init();
158  }
159 }

161 bool CActiveDeterministicMasternodeManager::GetLocalAddress(CService& addrRet)
162 {
163  // First try to find whatever local address is specified by externalip option
164  bool fFoundLocal = GetLocal(addrRet) && CMasternode::IsValidNetAddr(addrRet);
165  if (!fFoundLocal && Params().NetworkIDString() == CBaseChainParams::REGTEST) {
166  if (Lookup("127.0.0.1", addrRet, GetListenPort(), false)) {
167  fFoundLocal = true;
168  }
169  }
170  if (!fFoundLocal) {
171  bool empty = true;
172  // If we have some peers, let's try to find our local address from one of them
173  g_connman->ForEachNodeContinueIf(CConnman::AllNodes, [&fFoundLocal, &empty](CNode* pnode) {
174  empty = false;
175  if (pnode->addr.IsIPv4())
176  fFoundLocal = GetLocal(activeMasternodeInfo.service, &pnode->addr) && CMasternode::IsValidNetAddr(activeMasternodeInfo.service);
177  return !fFoundLocal;
178  });
179  // nothing and no live connections, can't do anything for now
180  if (empty) {
181  strError = "Can't detect valid external address. Please consider using the externalip configuration option if problem persists. Make sure to use IPv4 address only.";
182  LogPrintf("CActiveDeterministicMasternodeManager::GetLocalAddress -- ERROR: %s\n", strError);
183  return false;
184  }
185  }
186  return true;
187 }

273 bool CActiveLegacyMasternodeManager::SendMasternodePing(CConnman& connman)
274 {
275  if (deterministicMNManager->IsDeterministicMNsSporkActive()) return false;
276 
277  if (!fPingerEnabled) {
278  LogPrint("masternode", "CActiveLegacyMasternodeManager::SendMasternodePing -- %s: masternode ping service is disabled, skipping...\n", GetStateString());
279  return false;
280  }
281 
282  if (!mnodeman.Has(activeMasternodeInfo.outpoint)) {
283  strNotCapableReason = "Masternode not in masternode list";
284  nState = ACTIVE_MASTERNODE_NOT_CAPABLE;
285  LogPrintf("CActiveLegacyMasternodeManager::SendMasternodePing -- %s: %s\n", GetStateString(), strNotCapableReason);
286  return false;
287  }
288 
289  CMasternodePing mnp(activeMasternodeInfo.outpoint);
290  mnp.nSentinelVersion = nSentinelVersion;
291  mnp.fSentinelIsCurrent =
292  (abs(GetAdjustedTime() - nSentinelPingTime) < MASTERNODE_SENTINEL_PING_MAX_SECONDS);
293  if (!mnp.Sign(activeMasternodeInfo.legacyKeyOperator, activeMasternodeInfo.legacyKeyIDOperator)) {
294  LogPrintf("CActiveLegacyMasternodeManager::SendMasternodePing -- ERROR: Couldn't sign Masternode Ping\n");
295  return false;
296  }
297 
298  // Update lastPing for our masternode in Masternode list
299  if (mnodeman.IsMasternodePingedWithin(activeMasternodeInfo.outpoint, MASTERNODE_MIN_MNP_SECONDS, mnp.sigTime)) {
300  LogPrintf("CActiveLegacyMasternodeManager::SendMasternodePing -- Too early to send Masternode Ping\n");
301  return false;
302  }
303 
304  mnodeman.SetMasternodeLastPing(activeMasternodeInfo.outpoint, mnp);
305 
306  LogPrintf("CActiveLegacyMasternodeManager::SendMasternodePing -- Relaying ping, collateral=%s\n", activeMasternodeInfo.outpoint.ToStringShort());
307  mnp.Relay(connman);
308 
309  return true;
310 }

410 void CActiveLegacyMasternodeManager::ManageStateRemote()
411 {
412  if (deterministicMNManager->IsDeterministicMNsSporkActive()) return;
413 
414  LogPrint("masternode", "CActiveLegacyMasternodeManager::ManageStateRemote -- Start status = %s, type = %s, pinger enabled = %d, keyIDOperator = %s\n",
415  GetStatus(), GetTypeString(), fPingerEnabled, activeMasternodeInfo.legacyKeyIDOperator.ToString());
416 
417  mnodeman.CheckMasternode(activeMasternodeInfo.legacyKeyIDOperator, true);
418  masternode_info_t infoMn;
419  if (mnodeman.GetMasternodeInfo(activeMasternodeInfo.legacyKeyIDOperator, infoMn)) {
420  if (infoMn.nProtocolVersion != PROTOCOL_VERSION) {
421  nState = ACTIVE_MASTERNODE_NOT_CAPABLE;
422  strNotCapableReason = "Invalid protocol version";
423  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- %s: %s\n", GetStateString(), strNotCapableReason);
424  return;
425  }
426  if (activeMasternodeInfo.service != infoMn.addr) {
427  nState = ACTIVE_MASTERNODE_NOT_CAPABLE;
428  strNotCapableReason = "Broadcasted IP doesn't match our external address. Make sure you issued a new broadcast if IP of this masternode changed recently.";
429  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- %s: %s\n", GetStateString(), strNotCapableReason);
430  return;
431  }
432  if (!CMasternode::IsValidStateForAutoStart(infoMn.nActiveState)) {
433  nState = ACTIVE_MASTERNODE_NOT_CAPABLE;
434  strNotCapableReason = strprintf("Masternode in %s state", CMasternode::StateToString(infoMn.nActiveState));
435  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- %s: %s\n", GetStateString(), strNotCapableReason);
436  return;
437  }
438  auto dmn = deterministicMNManager->GetListAtChainTip().GetMNByCollateral(infoMn.outpoint);
439  if (dmn) {
440  if (dmn->pdmnState->addr != infoMn.addr) {
441  nState = ACTIVE_MASTERNODE_NOT_CAPABLE;
442  strNotCapableReason = strprintf("Masternode collateral is a ProTx and ProTx address does not match local address");
443  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- %s: %s\n", GetStateString(), strNotCapableReason);
444  return;
445  }
446  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- Collateral is a ProTx\n");
447  }
448  if (nState != ACTIVE_MASTERNODE_STARTED) {
449  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- STARTED!\n");
450  activeMasternodeInfo.outpoint = infoMn.outpoint;
451  activeMasternodeInfo.service = infoMn.addr;
452  fPingerEnabled = true;
453  nState = ACTIVE_MASTERNODE_STARTED;
454  }
455  } else {
456  nState = ACTIVE_MASTERNODE_NOT_CAPABLE;
457  strNotCapableReason = "Masternode not in masternode list";
458  LogPrintf("CActiveLegacyMasternodeManager::ManageStateRemote -- %s: %s\n", GetStateString(), strNotCapableReason);
459  }
