367  friend bool visit_relaxed(this_t, PosL&& posl, PosR&& posr,
368  Iter&& first, size_t idx)
369  {
370  auto nl = posl.node();
371  auto nr = posr.node();
372  if (nl == nr)
373  return true;
374  auto cl = posl.count();
375  auto cr = posr.count();
376  assert(cr > 0);
377  auto sbr = size_t{};
378  auto i = count_t{};
379  auto j = count_t{};
380  for (; i < cl; ++i) {
381  auto sbl = posl.size_before(i);
382  for (; j + 1 < cr && (sbr = posr.size_before(j)) < sbl; ++j);
383  auto res = sbl == sbr

1660  std::move(counts + i + 1, counts + n, counts + i);
