75 // This matches the calculation in CompareTxMemPoolEntryByAncestorFee,
76 // except operating on CTxMemPoolModifiedEntry.
77 // TODO: refactor to avoid duplication of this logic.
78 struct CompareModifiedEntry {
79  bool operator()(const CTxMemPoolModifiedEntry &a, const CTxMemPoolModifiedEntry &b) const
80  {
81  double f1 = (double)a.nModFeesWithAncestors * b.nSizeWithAncestors;
82  double f2 = (double)b.nModFeesWithAncestors * a.nSizeWithAncestors;
83  if (f1 == f2) {
84  return CTxMemPool::CompareIteratorByHash()(a.iter, b.iter);
85  }
86  return f1 > f2;
87  }
88 };

90 // A comparator that sorts transactions based on number of ancestors.
91 // This is sufficient to sort an ancestor package in an order that is valid
92 // to appear in a block.
93 struct CompareTxIterByAncestorCount {
94  bool operator()(const CTxMemPool::txiter &a, const CTxMemPool::txiter &b) const
95  {
96  if (a->GetCountWithAncestors() != b->GetCountWithAncestors())
97  return a->GetCountWithAncestors() < b->GetCountWithAncestors();
98  return CTxMemPool::CompareIteratorByHash()(a, b);
99  }
100 };
