178 template <typename Stream, typename K, typename T, typename Hash, typename Equal>
179 void SerializeImmerMap(Stream& os, const immer::map<K, T, Hash, Equal>& m)
180 {
181  WriteCompactSize(os, m.size());
182  for (typename immer::map<K, T, Hash, Equal>::const_iterator mi = m.begin(); mi != m.end(); ++mi)
183  Serialize(os, (*mi));
184 }
185 
186 template <typename Stream, typename K, typename T, typename Hash, typename Equal>
187 void UnserializeImmerMap(Stream& is, immer::map<K, T, Hash, Equal>& m)
188 {
189  m = immer::map<K, T, Hash, Equal>();
190  unsigned int nSize = ReadCompactSize(is);
191  for (unsigned int i = 0; i < nSize; i++) {
192  std::pair<K, T> item;
193  Unserialize(is, item);
194  m = m.set(item.first, item.second);
195  }
196 }

356  template <typename T>
357  bool HasUniqueProperty(const T& v) const
358  {
359  return mnUniquePropertyMap.count(::SerializeHash(v)) != 0;
360  }
361  template <typename T>
362  CDeterministicMNCPtr GetUniquePropertyMN(const T& v) const
363  {
364  auto p = mnUniquePropertyMap.find(::SerializeHash(v));
365  if (!p) {
366  return nullptr;
367  }
368  return GetMN(p->first);
369  }
370 
371 private:
372  template <typename T>
373  void AddUniqueProperty(const CDeterministicMNCPtr& dmn, const T& v)
374  {
375  static const T nullValue;
376  assert(v != nullValue);
377 
378  auto hash = ::SerializeHash(v);
379  auto oldEntry = mnUniquePropertyMap.find(hash);
380  assert(!oldEntry || oldEntry->first == dmn->proTxHash);
381  std::pair<uint256, uint32_t> newEntry(dmn->proTxHash, 1);
382  if (oldEntry) {
383  newEntry.second = oldEntry->second + 1;
384  }
385  mnUniquePropertyMap = mnUniquePropertyMap.set(hash, newEntry);
386  }
387  template <typename T>
388  void DeleteUniqueProperty(const CDeterministicMNCPtr& dmn, const T& oldValue)
389  {
390  static const T nullValue;
391  assert(oldValue != nullValue);
392 
393  auto oldHash = ::SerializeHash(oldValue);
394  auto p = mnUniquePropertyMap.find(oldHash);
395  assert(p && p->first == dmn->proTxHash);
396  if (p->second == 1) {
397  mnUniquePropertyMap = mnUniquePropertyMap.erase(oldHash);
398  } else {
399  mnUniquePropertyMap = mnUniquePropertyMap.set(oldHash, std::make_pair(dmn->proTxHash, p->second - 1));
400  }
401  }
402  template <typename T>
403  void UpdateUniqueProperty(const CDeterministicMNCPtr& dmn, const T& oldValue, const T& newValue)
404  {
405  if (oldValue == newValue) {
406  return;
407  }
408  static const T nullValue;
409 
410  if (oldValue != nullValue) {
411  DeleteUniqueProperty(dmn, oldValue);
412  }
413 
414  if (newValue != nullValue) {
415  AddUniqueProperty(dmn, newValue);
416  }
417  }
