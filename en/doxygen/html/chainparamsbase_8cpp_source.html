92 CBaseChainParams& BaseParams(const std::string& chain)
93 {
94  if (chain == CBaseChainParams::MAIN)
95  return mainParams;
96  else if (chain == CBaseChainParams::TESTNET)
97  return testNetParams;
98  else if (chain == CBaseChainParams::DEVNET) {
99  assert(devNetParams);
100  return *devNetParams;
101  } else if (chain == CBaseChainParams::REGTEST)
102  return regTestParams;
103  else
104  throw std::runtime_error(strprintf("%s: Unknown chain %s.", __func__, chain));
105 }

121 std::string ChainNameFromCommandLine()
122 {
123  bool fRegTest = GetBoolArg("-regtest", false);
124  bool fDevNet = IsArgSet("-devnet");
125  bool fTestNet = GetBoolArg("-testnet", false);
126 
127  int nameParamsCount = (fRegTest ? 1 : 0) + (fDevNet ? 1 : 0) + (fTestNet ? 1 : 0);
128  if (nameParamsCount > 1)
129  throw std::runtime_error("Only one of -regtest, -testnet or -devnet can be used.");
130 
131  if (fDevNet)
132  return CBaseChainParams::DEVNET;
133  if (fRegTest)
134  return CBaseChainParams::REGTEST;
135  if (fTestNet)
136  return CBaseChainParams::TESTNET;
137  return CBaseChainParams::MAIN;
138 }
