225 std::vector<CDeterministicMNCPtr> CDeterministicMNList::CalculateQuorum(size_t maxSize, const uint256& modifier) const
226 {
227  auto scores = CalculateScores(modifier);
228 
229  // sort is descending order
230  std::sort(scores.rbegin(), scores.rend(), [](const std::pair<arith_uint256, CDeterministicMNCPtr>& a, std::pair<arith_uint256, CDeterministicMNCPtr>& b) {
231  if (a.first == b.first) {
232  // this should actually never happen, but we should stay compatible with how the non deterministic MNs did the sorting
233  return a.second->collateralOutpoint < b.second->collateralOutpoint;
234  }
235  return a.first < b.first;
236  });
237 
238  // take top maxSize entries and return it
239  std::vector<CDeterministicMNCPtr> result;
240  result.resize(std::min(maxSize, scores.size()));
241  for (size_t i = 0; i < result.size(); i++) {
242  result[i] = std::move(scores[i].second);
243  }
244  return result;
245 }

247 std::vector<std::pair<arith_uint256, CDeterministicMNCPtr>> CDeterministicMNList::CalculateScores(const uint256& modifier) const
248 {
249  std::vector<std::pair<arith_uint256, CDeterministicMNCPtr>> scores;
250  scores.reserve(GetAllMNsCount());
251  ForEachMN(true, [&](const CDeterministicMNCPtr& dmn) {
252  if (dmn->pdmnState->confirmedHash.IsNull()) {
253  // we only take confirmed MNs into account to avoid hash grinding on the ProRegTxHash to sneak MNs into a
254  // future quorums
255  return;
256  }
257  // calculate sha256(sha256(proTxHash, confirmedHash), modifier) per MN
258  // Please note that this is not a double-sha256 but a single-sha256
259  // The first part is already precalculated (confirmedHashWithProRegTxHash)
260  // TODO When https://github.com/bitcoin/bitcoin/pull/13191 gets backported, implement something that is similar but for single-sha256
261  uint256 h;
262  CSHA256 sha256;
263  sha256.Write(dmn->pdmnState->confirmedHashWithProRegTxHash.begin(), dmn->pdmnState->confirmedHashWithProRegTxHash.size());
264  sha256.Write(modifier.begin(), modifier.size());
265  sha256.Finalize(h.begin());
266 
267  scores.emplace_back(UintToArith256(h), dmn);
268  });
269 
270  return scores;
271 }

287 void CDeterministicMNList::PoSePunish(const uint256& proTxHash, int penalty, bool debugLogs)
288 {
289  assert(penalty > 0);
290 
291  auto dmn = GetMN(proTxHash);
292  assert(dmn);
293 
294  int maxPenalty = CalcMaxPoSePenalty();
295 
296  auto newState = std::make_shared<CDeterministicMNState>(*dmn->pdmnState);
297  newState->nPoSePenalty += penalty;
298  newState->nPoSePenalty = std::min(maxPenalty, newState->nPoSePenalty);
299 
300  if (debugLogs) {
301  LogPrintf("CDeterministicMNList::%s -- punished MN %s, penalty %d->%d (max=%d)\n",
302  __func__, proTxHash.ToString(), dmn->pdmnState->nPoSePenalty, newState->nPoSePenalty, maxPenalty);
303  }
304 
305  if (newState->nPoSePenalty >= maxPenalty && newState->nPoSeBanHeight == -1) {
306  newState->nPoSeBanHeight = nHeight;
307  if (debugLogs) {
308  LogPrintf("CDeterministicMNList::%s -- banned MN %s at height %d\n",
309  __func__, proTxHash.ToString(), nHeight);
310  }
311  }
312  UpdateMN(proTxHash, newState);
313 }

326 CDeterministicMNListDiff CDeterministicMNList::BuildDiff(const CDeterministicMNList& to) const
327 {
328  CDeterministicMNListDiff diffRet;
329  diffRet.prevBlockHash = blockHash;
330  diffRet.blockHash = to.blockHash;
331  diffRet.nHeight = to.nHeight;
332 
333  to.ForEachMN(false, [&](const CDeterministicMNCPtr& toPtr) {
334  auto fromPtr = GetMN(toPtr->proTxHash);
335  if (fromPtr == nullptr) {
336  diffRet.addedMNs.emplace(toPtr->proTxHash, toPtr);
337  } else if (*toPtr->pdmnState != *fromPtr->pdmnState) {
338  diffRet.updatedMNs.emplace(toPtr->proTxHash, toPtr->pdmnState);
339  }
340  });
341  ForEachMN(false, [&](const CDeterministicMNCPtr& fromPtr) {
342  auto toPtr = to.GetMN(fromPtr->proTxHash);
343  if (toPtr == nullptr) {
344  diffRet.removedMns.insert(fromPtr->proTxHash);
345  }
346  });
347 
348  return diffRet;
349 }

382 
383  CDeterministicMNList result = *this;

385  result.nHeight = diff.nHeight;

448 bool CDeterministicMNManager::ProcessBlock(const CBlock& block, const CBlockIndex* pindex, CValidationState& _state)
449 {
450  AssertLockHeld(cs_main);
451 
452  bool fDIP0003Active = VersionBitsState(pindex->pprev, Params().GetConsensus(), Consensus::DEPLOYMENT_DIP0003, versionbitscache) == THRESHOLD_ACTIVE;
453  if (!fDIP0003Active) {
454  return true;
455  }
456 
457  LOCK(cs);
458 
459  int nHeight = pindex->nHeight;
460 
461  CDeterministicMNList newList;
462  if (!BuildNewListFromBlock(block, pindex->pprev, _state, newList, true)) {
463  return false;
464  }
465 
466  if (newList.GetHeight() == -1) {
467  newList.SetHeight(nHeight);
468  }
469 
470  newList.SetBlockHash(block.GetHash());
471 
472  CDeterministicMNList oldList = GetListForBlock(pindex->pprev->GetBlockHash());
473  CDeterministicMNListDiff diff = oldList.BuildDiff(newList);
474 
475  evoDb.Write(std::make_pair(DB_LIST_DIFF, diff.blockHash), diff);
476  if ((nHeight % SNAPSHOT_LIST_PERIOD) == 0 || oldList.GetHeight() == -1) {
477  evoDb.Write(std::make_pair(DB_LIST_SNAPSHOT, diff.blockHash), newList);
478  LogPrintf("CDeterministicMNManager::%s -- Wrote snapshot. nHeight=%d, mapCurMNs.allMNsCount=%d\n",
479  __func__, nHeight, newList.GetAllMNsCount());
480  }
481 
482  if (nHeight == GetSpork15Value()) {
483  LogPrintf("CDeterministicMNManager::%s -- spork15 is active now. nHeight=%d\n", __func__, nHeight);
484  }
485 
486  CleanupCache(nHeight);
487 
488  return true;

517 bool CDeterministicMNManager::BuildNewListFromBlock(const CBlock& block, const CBlockIndex* pindexPrev, CValidationState& _state, CDeterministicMNList& mnListRet, bool debugLogs)
518 {
519  AssertLockHeld(cs);
520 
521  int nHeight = pindexPrev->nHeight + 1;
522 
523  CDeterministicMNList oldList = GetListForBlock(pindexPrev->GetBlockHash());
524  CDeterministicMNList newList = oldList;
525  newList.SetBlockHash(uint256()); // we can't know the final block hash, so better not return a (invalid) block hash
526  newList.SetHeight(nHeight);
527 
528  auto payee = oldList.GetMNPayee();
529 
530  // we iterate the oldList here and update the newList
531  // this is only valid as long these have not diverged at this point, which is the case as long as we don't add
532  // code above this loop that modifies newList
533  oldList.ForEachMN(false, [&](const CDeterministicMNCPtr& dmn) {
534  if (!dmn->pdmnState->confirmedHash.IsNull()) {
535  // already confirmed
536  return;
537  }
538  // this works on the previous block, so confirmation will happen one block after nMasternodeMinimumConfirmations
539  // has been reached, but the block hash will then point to the block at nMasternodeMinimumConfirmations
540  int nConfirmations = pindexPrev->nHeight - dmn->pdmnState->nRegisteredHeight;
541  if (nConfirmations >= Params().GetConsensus().nMasternodeMinimumConfirmations) {
542  CDeterministicMNState newState = *dmn->pdmnState;
543  newState.UpdateConfirmedHash(dmn->proTxHash, pindexPrev->GetBlockHash());
544  newList.UpdateMN(dmn->proTxHash, std::make_shared<CDeterministicMNState>(newState));
545  }
546  });
547 
548  DecreasePoSePenalties(newList);
549 
550  // we skip the coinbase
551  for (int i = 1; i < (int)block.vtx.size(); i++) {
552  const CTransaction& tx = *block.vtx[i];
553 
554  if (tx.nVersion != 3) {
555  // only interested in special TXs
556  continue;
557  }
558 
559  if (tx.nType == TRANSACTION_PROVIDER_REGISTER) {
560  CProRegTx proTx;
561  if (!GetTxPayload(tx, proTx)) {
562  assert(false); // this should have been handled already
563  }
564 
565  auto dmn = std::make_shared<CDeterministicMN>();
566  dmn->proTxHash = tx.GetHash();
567 
568  // collateralOutpoint is either pointing to an external collateral or to the ProRegTx itself
569  if (proTx.collateralOutpoint.hash.IsNull()) {
570  dmn->collateralOutpoint = COutPoint(tx.GetHash(), proTx.collateralOutpoint.n);
571  } else {
572  dmn->collateralOutpoint = proTx.collateralOutpoint;
573  }
574 
575  Coin coin;
576  if (!proTx.collateralOutpoint.hash.IsNull() && (!GetUTXOCoin(dmn->collateralOutpoint, coin) || coin.out.nValue != 1000 * COIN)) {
577  // should actually never get to this point as CheckProRegTx should have handled this case.
578  // We do this additional check nevertheless to be 100% sure
579  return _state.DoS(100, false, REJECT_INVALID, "bad-protx-collateral");
580  }
581 
582  auto replacedDmn = newList.GetMNByCollateral(dmn->collateralOutpoint);
583  if (replacedDmn != nullptr) {
584  // This might only happen with a ProRegTx that refers an external collateral
585  // In that case the new ProRegTx will replace the old one. This means the old one is removed
586  // and the new one is added like a completely fresh one, which is also at the bottom of the payment list
587  newList.RemoveMN(replacedDmn->proTxHash);
588  if (debugLogs) {
589  LogPrintf("CDeterministicMNManager::%s -- MN %s removed from list because collateral was used for a new ProRegTx. collateralOutpoint=%s, nHeight=%d, mapCurMNs.allMNsCount=%d\n",
590  __func__, replacedDmn->proTxHash.ToString(), dmn->collateralOutpoint.ToStringShort(), nHeight, newList.GetAllMNsCount());
591  }
592  }
593 
594  if (newList.HasUniqueProperty(proTx.addr)) {
595  return _state.DoS(100, false, REJECT_CONFLICT, "bad-protx-dup-addr");
596  }
597  if (newList.HasUniqueProperty(proTx.keyIDOwner) || newList.HasUniqueProperty(proTx.pubKeyOperator)) {
598  return _state.DoS(100, false, REJECT_CONFLICT, "bad-protx-dup-key");
599  }
600 
601  dmn->nOperatorReward = proTx.nOperatorReward;
602  dmn->pdmnState = std::make_shared<CDeterministicMNState>(proTx);
603 
604  CDeterministicMNState dmnState = *dmn->pdmnState;
605  dmnState.nRegisteredHeight = nHeight;
606 
607  if (proTx.addr == CService()) {
608  // start in banned pdmnState as we need to wait for a ProUpServTx
609  dmnState.nPoSeBanHeight = nHeight;
610  }
611 
612  dmn->pdmnState = std::make_shared<CDeterministicMNState>(dmnState);
613 
614  newList.AddMN(dmn);
615 
616  if (debugLogs) {
617  LogPrintf("CDeterministicMNManager::%s -- MN %s added at height %d: %s\n",
618  __func__, tx.GetHash().ToString(), nHeight, proTx.ToString());
619  }
620  } else if (tx.nType == TRANSACTION_PROVIDER_UPDATE_SERVICE) {
621  CProUpServTx proTx;
622  if (!GetTxPayload(tx, proTx)) {
623  assert(false); // this should have been handled already
624  }
625 
626  if (newList.HasUniqueProperty(proTx.addr) && newList.GetUniquePropertyMN(proTx.addr)->proTxHash != proTx.proTxHash) {
627  return _state.DoS(100, false, REJECT_CONFLICT, "bad-protx-dup-addr");
628  }
629 
630  CDeterministicMNCPtr dmn = newList.GetMN(proTx.proTxHash);
631  if (!dmn) {
632  return _state.DoS(100, false, REJECT_INVALID, "bad-protx-hash");
633  }
634  auto newState = std::make_shared<CDeterministicMNState>(*dmn->pdmnState);
635  newState->addr = proTx.addr;
636  newState->scriptOperatorPayout = proTx.scriptOperatorPayout;
637 
638  if (newState->nPoSeBanHeight != -1) {
639  // only revive when all keys are set
640  if (newState->pubKeyOperator.IsValid() && !newState->keyIDVoting.IsNull() && !newState->keyIDOwner.IsNull()) {
641  newState->nPoSePenalty = 0;
642  newState->nPoSeBanHeight = -1;
643  newState->nPoSeRevivedHeight = nHeight;
644 
645  if (debugLogs) {
646  LogPrintf("CDeterministicMNManager::%s -- MN %s revived at height %d\n",
647  __func__, proTx.proTxHash.ToString(), nHeight);
648  }
649  }
650  }
651 
652  newList.UpdateMN(proTx.proTxHash, newState);
653  if (debugLogs) {
654  LogPrintf("CDeterministicMNManager::%s -- MN %s updated at height %d: %s\n",
655  __func__, proTx.proTxHash.ToString(), nHeight, proTx.ToString());
656  }
657  } else if (tx.nType == TRANSACTION_PROVIDER_UPDATE_REGISTRAR) {
658  CProUpRegTx proTx;
659  if (!GetTxPayload(tx, proTx)) {
660  assert(false); // this should have been handled already
661  }
662 
663  CDeterministicMNCPtr dmn = newList.GetMN(proTx.proTxHash);
664  if (!dmn) {
665  return _state.DoS(100, false, REJECT_INVALID, "bad-protx-hash");
666  }
667  auto newState = std::make_shared<CDeterministicMNState>(*dmn->pdmnState);
668  if (newState->pubKeyOperator != proTx.pubKeyOperator) {
669  // reset all operator related fields and put MN into PoSe-banned state in case the operator key changes
670  newState->ResetOperatorFields();
671  newState->BanIfNotBanned(nHeight);
672  }
673  newState->pubKeyOperator = proTx.pubKeyOperator;
674  newState->keyIDVoting = proTx.keyIDVoting;
675  newState->scriptPayout = proTx.scriptPayout;
676 
677  newList.UpdateMN(proTx.proTxHash, newState);
678 
679  if (debugLogs) {
680  LogPrintf("CDeterministicMNManager::%s -- MN %s updated at height %d: %s\n",
681  __func__, proTx.proTxHash.ToString(), nHeight, proTx.ToString());
682  }
683  } else if (tx.nType == TRANSACTION_PROVIDER_UPDATE_REVOKE) {
684  CProUpRevTx proTx;
685  if (!GetTxPayload(tx, proTx)) {
686  assert(false); // this should have been handled already
687  }
688 
689  CDeterministicMNCPtr dmn = newList.GetMN(proTx.proTxHash);
690  if (!dmn) {
691  return _state.DoS(100, false, REJECT_INVALID, "bad-protx-hash");
692  }
693  auto newState = std::make_shared<CDeterministicMNState>(*dmn->pdmnState);
694  newState->ResetOperatorFields();
695  newState->BanIfNotBanned(nHeight);
696  newState->nRevocationReason = proTx.nReason;
697 
698  newList.UpdateMN(proTx.proTxHash, newState);
699 
700  if (debugLogs) {
701  LogPrintf("CDeterministicMNManager::%s -- MN %s revoked operator key at height %d: %s\n",
702  __func__, proTx.proTxHash.ToString(), nHeight, proTx.ToString());
703  }
704  } else if (tx.nType == TRANSACTION_QUORUM_COMMITMENT) {
705  llmq::CFinalCommitmentTxPayload qc;
706  if (!GetTxPayload(tx, qc)) {
707  assert(false); // this should have been handled already
708  }
709  if (!qc.commitment.IsNull()) {
710  HandleQuorumCommitment(qc.commitment, newList, debugLogs);
711  }
712  }
713  }
714 
715  // we skip the coinbase
716  for (int i = 1; i < (int)block.vtx.size(); i++) {
717  const CTransaction& tx = *block.vtx[i];
718 
719  // check if any existing MN collateral is spent by this transaction
720  for (const auto& in : tx.vin) {
721  auto dmn = newList.GetMNByCollateral(in.prevout);
722  if (dmn && dmn->collateralOutpoint == in.prevout) {
723  newList.RemoveMN(dmn->proTxHash);
724 
725  if (debugLogs) {
726  LogPrintf("CDeterministicMNManager::%s -- MN %s removed from list because collateral was spent. collateralOutpoint=%s, nHeight=%d, mapCurMNs.allMNsCount=%d\n",
727  __func__, dmn->proTxHash.ToString(), dmn->collateralOutpoint.ToStringShort(), nHeight, newList.GetAllMNsCount());
728  }
729  }
730  }
731  }
732 
733  // The payee for the current block was determined by the previous block's list but it might have disappeared in the
734  // current block. We still pay that MN one last time however.
735  if (payee && newList.HasMN(payee->proTxHash)) {
736  auto newState = std::make_shared<CDeterministicMNState>(*newList.GetMN(payee->proTxHash)->pdmnState);
737  newState->nLastPaidHeight = nHeight;
738  newList.UpdateMN(payee->proTxHash, newState);
739  }
740 
741  mnListRet = std::move(newList);
742 
743  return true;
744 }

746 void CDeterministicMNManager::HandleQuorumCommitment(llmq::CFinalCommitment& qc, CDeterministicMNList& mnList, bool debugLogs)
747 {
748  // The commitment has already been validated at this point so it's safe to use members of it
749 
750  auto members = llmq::CLLMQUtils::GetAllQuorumMembers((Consensus::LLMQType)qc.llmqType, qc.quorumHash);
751 
752  for (size_t i = 0; i < members.size(); i++) {
753  if (!mnList.HasMN(members[i]->proTxHash)) {
754  continue;
755  }
756  if (!qc.validMembers[i]) {
757  // punish MN for failed DKG participation
758  // The idea is to immediately ban a MN when it fails 2 DKG sessions with only a few blocks in-between
759  // If there were enough blocks between failures, the MN has a chance to recover as he reduces his penalty by 1 for every block
760  // If it however fails 3 times in the timespan of a single payment cycle, it should definitely get banned
761  mnList.PoSePunish(members[i]->proTxHash, mnList.CalcPenalty(66), debugLogs);
762  }
763  }
764 }

766 void CDeterministicMNManager::DecreasePoSePenalties(CDeterministicMNList& mnList)
767 {
768  std::vector<uint256> toDecrease;
769  toDecrease.reserve(mnList.GetValidMNsCount() / 10);
770  // only iterate and decrease for valid ones (not PoSe banned yet)
771  // if a MN ever reaches the maximum, it stays in PoSe banned state until revived
772  mnList.ForEachMN(true, [&](const CDeterministicMNCPtr& dmn) {
773  if (dmn->pdmnState->nPoSePenalty > 0 && dmn->pdmnState->nPoSeBanHeight == -1) {
774  toDecrease.emplace_back(dmn->proTxHash);
775  }
776  });
777 
778  for (const auto& proTxHash : toDecrease) {
779  mnList.PoSeDecrease(proTxHash);
780  }

783 CDeterministicMNList CDeterministicMNManager::GetListForBlock(const uint256& blockHash)
784 {
785  LOCK(cs);
786 
787  auto it = mnListsCache.find(blockHash);
788  if (it != mnListsCache.end()) {
789  return it->second;
790  }
791 
792  uint256 blockHashTmp = blockHash;
793  CDeterministicMNList snapshot;
794  std::list<CDeterministicMNListDiff> listDiff;
795 
796  while (true) {
797  // try using cache before reading from disk
798  it = mnListsCache.find(blockHashTmp);
799  if (it != mnListsCache.end()) {
800  snapshot = it->second;
801  break;
802  }
803 
804  if (evoDb.Read(std::make_pair(DB_LIST_SNAPSHOT, blockHashTmp), snapshot)) {
805  break;
806  }
807 
808  CDeterministicMNListDiff diff;
809  if (!evoDb.Read(std::make_pair(DB_LIST_DIFF, blockHashTmp), diff)) {
810  snapshot = CDeterministicMNList(blockHashTmp, -1);
811  break;
812  }
813 
814  listDiff.emplace_front(diff);
815  blockHashTmp = diff.prevBlockHash;
816  }
817 
818  for (const auto& diff : listDiff) {
819  if (diff.HasChanges()) {
820  snapshot = snapshot.ApplyDiff(diff);
821  } else {
822  snapshot.SetBlockHash(diff.blockHash);
823  snapshot.SetHeight(diff.nHeight);
824  }
825  }
826 
827  mnListsCache.emplace(blockHash, snapshot);
828  return snapshot;
829 }
