26 void CQuorumBlockProcessor::ProcessMessage(CNode* pfrom, const std::string& strCommand, CDataStream& vRecv, CConnman& connman)
27 {
28  if (strCommand == NetMsgType::QFCOMMITMENT) {
29  CFinalCommitment qc;
30  vRecv >> qc;
31 
32  if (qc.IsNull()) {
33  LOCK(cs_main);
34  LogPrintf("CQuorumBlockProcessor::%s -- null commitment from peer=%d\n", __func__, pfrom->id);
35  Misbehaving(pfrom->id, 100);
36  return;
37  }
38 
39  if (!Params().GetConsensus().llmqs.count((Consensus::LLMQType)qc.llmqType)) {
40  LOCK(cs_main);
41  LogPrintf("CQuorumBlockProcessor::%s -- invalid commitment type %d from peer=%d\n", __func__,
42  qc.llmqType, pfrom->id);
43  Misbehaving(pfrom->id, 100);
44  return;
45  }
46  auto type = (Consensus::LLMQType)qc.llmqType;
47  const auto& params = Params().GetConsensus().llmqs.at(type);
48 
49  // Verify that quorumHash is part of the active chain and that it's the first block in the DKG interval
50  {
51  LOCK(cs_main);
52  if (!mapBlockIndex.count(qc.quorumHash)) {
53  LogPrintf("CQuorumBlockProcessor::%s -- unknown block %s in commitment, peer=%d\n", __func__,
54  qc.quorumHash.ToString(), pfrom->id);
55  // can't really punish the node here, as we might simply be the one that is on the wrong chain or not
56  // fully synced
57  return;
58  }
59  auto pquorumIndex = mapBlockIndex[qc.quorumHash];
60  if (chainActive.Tip()->GetAncestor(pquorumIndex->nHeight) != pquorumIndex) {
61  LogPrintf("CQuorumBlockProcessor::%s -- block %s not in active chain, peer=%d\n", __func__,
62  qc.quorumHash.ToString(), pfrom->id);
63  // same, can't punish
64  return;
65  }
66  int quorumHeight = pquorumIndex->nHeight - (pquorumIndex->nHeight % params.dkgInterval);
67  if (quorumHeight != pquorumIndex->nHeight) {
68  LogPrintf("CQuorumBlockProcessor::%s -- block %s is not the first block in the DKG interval, peer=%d\n", __func__,
69  qc.quorumHash.ToString(), pfrom->id);
70  Misbehaving(pfrom->id, 100);
71  return;
72  }
73  }
74 
75  {
76  // Check if we already got a better one locally
77  // We do this before verifying the commitment to avoid DoS
78  LOCK(minableCommitmentsCs);
79  auto k = std::make_pair(type, qc.quorumHash);
80  auto it = minableCommitmentsByQuorum.find(k);
81  if (it != minableCommitmentsByQuorum.end()) {
82  auto jt = minableCommitments.find(it->second);
83  if (jt != minableCommitments.end()) {
84  if (jt->second.CountSigners() <= qc.CountSigners()) {
85  return;
86  }
87  }
88  }
89  }
90 
91  auto members = CLLMQUtils::GetAllQuorumMembers(type, qc.quorumHash);
92 
93  if (!qc.Verify(members, true)) {
94  LOCK(cs_main);
95  LogPrintf("CQuorumBlockProcessor::%s -- commitment for quorum %s:%d is not valid, peer=%d\n", __func__,
96  qc.quorumHash.ToString(), qc.llmqType, pfrom->id);
97  Misbehaving(pfrom->id, 100);
98  return;
99  }
100 
101  LogPrintf("CQuorumBlockProcessor::%s -- received commitment for quorum %s:%d, validMembers=%d, signers=%d, peer=%d\n", __func__,
102  qc.quorumHash.ToString(), qc.llmqType, qc.CountValidMembers(), qc.CountSigners(), pfrom->id);
103 
104  AddMinableCommitment(qc);
105  }
106 }

108 bool CQuorumBlockProcessor::ProcessBlock(const CBlock& block, const CBlockIndex* pindex, CValidationState& state)
109 {
110  AssertLockHeld(cs_main);
111 
112  bool fDIP0003Active = VersionBitsState(pindex->pprev, Params().GetConsensus(), Consensus::DEPLOYMENT_DIP0003, versionbitscache) == THRESHOLD_ACTIVE;
113  if (!fDIP0003Active) {
114  return true;
115  }
116 
117  std::map<Consensus::LLMQType, CFinalCommitment> qcs;
118  if (!GetCommitmentsFromBlock(block, pindex->pprev, qcs, state)) {
119  return false;
120  }
121 
122  // The following checks make sure that there is always a (possibly null) commitment while in the mining phase
123  // until the first non-null commitment has been mined. After the non-null commitment, no other commitments are
124  // allowed, including null commitments.
125  for (const auto& p : Params().GetConsensus().llmqs) {
126  auto type = p.first;
127 
128  // does the currently processed block contain a (possibly null) commitment for the current session?
129  bool hasCommitmentInNewBlock = qcs.count(type) != 0;
130  bool isCommitmentRequired = IsCommitmentRequired(type, pindex->nHeight);
131 
132  if (hasCommitmentInNewBlock && !isCommitmentRequired) {
133  // If we're either not in the mining phase or a non-null commitment was mined already, reject the block
134  return state.DoS(100, false, REJECT_INVALID, "bad-qc-not-allowed");
135  }
136 
137  if (!hasCommitmentInNewBlock && isCommitmentRequired) {
138  // If no non-null commitment was mined for the mining phase yet and the new block does not include
139  // a (possibly null) commitment, the block should be rejected.
140  return state.DoS(100, false, REJECT_INVALID, "bad-qc-missing");
141  }
142  }
143 
144  for (auto& p : qcs) {
145  auto& qc = p.second;
146  if (!ProcessCommitment(pindex, qc, state)) {
147  return false;
148  }
149  }
150  return true;
151 }

153 bool CQuorumBlockProcessor::ProcessCommitment(const CBlockIndex* pindex, const CFinalCommitment& qc, CValidationState& state)
154 {
155  auto& params = Params().GetConsensus().llmqs.at((Consensus::LLMQType)qc.llmqType);
156 
157  uint256 quorumHash = GetQuorumBlockHash((Consensus::LLMQType)qc.llmqType, pindex->nHeight);
158  if (quorumHash.IsNull()) {
159  return state.DoS(100, false, REJECT_INVALID, "bad-qc-block");
160  }
161  if (quorumHash != qc.quorumHash) {
162  return state.DoS(100, false, REJECT_INVALID, "bad-qc-block");
163  }
164 
165  if (qc.IsNull()) {
166  if (!qc.VerifyNull()) {
167  return state.DoS(100, false, REJECT_INVALID, "bad-qc-invalid-null");
168  }
169  return true;
170  }
171 
172  if (HasMinedCommitment(params.type, quorumHash)) {
173  // should not happen as it's already handled in ProcessBlock
174  return state.DoS(100, false, REJECT_INVALID, "bad-qc-dup");
175  }
176 
177  if (!IsMiningPhase(params.type, pindex->nHeight)) {
178  // should not happen as it's already handled in ProcessBlock
179  return state.DoS(100, false, REJECT_INVALID, "bad-qc-height");
180  }
181 
182  auto members = CLLMQUtils::GetAllQuorumMembers(params.type, quorumHash);
183 
184  if (!qc.Verify(members, true)) {
185  return state.DoS(100, false, REJECT_INVALID, "bad-qc-invalid");
186  }
187 
188  // Store commitment in DB
189  evoDb.Write(std::make_pair(DB_MINED_COMMITMENT, std::make_pair((uint8_t)params.type, quorumHash)), qc);
190 
191  LogPrintf("CQuorumBlockProcessor::%s -- processed commitment from block. type=%d, quorumHash=%s, signers=%s, validMembers=%d, quorumPublicKey=%s\n", __func__,
192  qc.llmqType, quorumHash.ToString(), qc.CountSigners(), qc.CountValidMembers(), qc.quorumPublicKey.ToString());
193 
194  return true;
195 }

197 bool CQuorumBlockProcessor::UndoBlock(const CBlock& block, const CBlockIndex* pindex)
198 {
199  AssertLockHeld(cs_main);
200 
201  std::map<Consensus::LLMQType, CFinalCommitment> qcs;
202  CValidationState dummy;
203  if (!GetCommitmentsFromBlock(block, pindex->pprev, qcs, dummy)) {
204  return false;
205  }
206 
207  for (auto& p : qcs) {
208  auto& qc = p.second;
209  if (qc.IsNull()) {
210  continue;
211  }
212 
213  evoDb.Erase(std::make_pair(DB_MINED_COMMITMENT, std::make_pair(qc.llmqType, qc.quorumHash)));
214 
215  if (!qc.IsNull()) {
216  // if a reorg happened, we should allow to mine this commitment later
217  AddMinableCommitment(qc);
218  }
219  }
220 
221  return true;
222 }

224 bool CQuorumBlockProcessor::GetCommitmentsFromBlock(const CBlock& block, const CBlockIndex* pindexPrev, std::map<Consensus::LLMQType, CFinalCommitment>& ret, CValidationState& state)
225 {
226  AssertLockHeld(cs_main);
227 
228  auto& consensus = Params().GetConsensus();
229  bool fDIP0003Active = VersionBitsState(pindexPrev, consensus, Consensus::DEPLOYMENT_DIP0003, versionbitscache) == THRESHOLD_ACTIVE;
230 
231  ret.clear();
232 
233  for (const auto& tx : block.vtx) {
234  if (tx->nType == TRANSACTION_QUORUM_COMMITMENT) {
235  CFinalCommitmentTxPayload qc;
236  if (!GetTxPayload(*tx, qc)) {
237  // should not happen as it was verified before processing the block
238  return state.DoS(100, false, REJECT_INVALID, "bad-tx-payload");
239  }
240 
241  // only allow one commitment per type and per block
242  if (ret.count((Consensus::LLMQType)qc.commitment.llmqType)) {
243  return state.DoS(100, false, REJECT_INVALID, "bad-qc-dup");
244  }
245 
246  ret.emplace((Consensus::LLMQType)qc.commitment.llmqType, std::move(qc.commitment));
247  }
248  }
249 
250  if (!fDIP0003Active && !ret.empty()) {
251  return state.DoS(100, false, REJECT_INVALID, "bad-qc-premature");
252  }
253 
254  return true;
255 }

267 bool CQuorumBlockProcessor::IsCommitmentRequired(Consensus::LLMQType llmqType, int nHeight)
268 {
269  uint256 quorumHash = GetQuorumBlockHash(llmqType, nHeight);
270 
271  // perform extra check for quorumHash.IsNull as the quorum hash is unknown for the first block of a session
272  // this is because the currently processed block's hash will be the quorumHash of this session
273  bool isMiningPhase = !quorumHash.IsNull() && IsMiningPhase(llmqType, nHeight);
274 
275  // did we already mine a non-null commitment for this session?
276  bool hasMinedCommitment = !quorumHash.IsNull() && HasMinedCommitment(llmqType, quorumHash);
277 
278  return isMiningPhase && !hasMinedCommitment;
279 }

282 uint256 CQuorumBlockProcessor::GetQuorumBlockHash(Consensus::LLMQType llmqType, int nHeight)
283 {
284  AssertLockHeld(cs_main);

287 
288  int quorumStartHeight = nHeight - (nHeight % params.dkgInterval);
289  uint256 quorumBlockHash;
290  if (!GetBlockHash(quorumBlockHash, quorumStartHeight)) {

293  return quorumBlockHash;

314 void CQuorumBlockProcessor::AddMinableCommitment(const CFinalCommitment& fqc)
315 {
316  bool relay = false;
317  uint256 commitmentHash = ::SerializeHash(fqc);
318 
319  {
320  LOCK(minableCommitmentsCs);
321 
322  auto k = std::make_pair((Consensus::LLMQType) fqc.llmqType, fqc.quorumHash);
323  auto ins = minableCommitmentsByQuorum.emplace(k, commitmentHash);
324  if (ins.second) {
325  minableCommitments.emplace(commitmentHash, fqc);
326  relay = true;
327  } else {
328  auto& oldFqc = minableCommitments.at(ins.first->second);
329  if (fqc.CountSigners() > oldFqc.CountSigners()) {
330  // new commitment has more signers, so override the known one
331  ins.first->second = commitmentHash;
332  minableCommitments.erase(ins.first->second);
333  minableCommitments.emplace(commitmentHash, fqc);
334  relay = true;
335  }
336  }
337  }
338 
339  // We only relay the new commitment if it's new or better then the old one
340  if (relay) {
341  CInv inv(MSG_QUORUM_FINAL_COMMITMENT, commitmentHash);
342  g_connman->RelayInv(inv, DMN_PROTO_VERSION);
343  }
344 }

357 // Will return false if no commitment should be mined
358 // Will return true and a null commitment if no minable commitment is known and none was mined yet
359 bool CQuorumBlockProcessor::GetMinableCommitment(Consensus::LLMQType llmqType, int nHeight, CFinalCommitment& ret)
360 {
361  AssertLockHeld(cs_main);
362 
363  if (!IsCommitmentRequired(llmqType, nHeight)) {
364  // no commitment required
365  return false;
366  }
367 
368  uint256 quorumHash = GetQuorumBlockHash(llmqType, nHeight);
369  if (quorumHash.IsNull()) {
370  return false;
371  }
372 
373  LOCK(minableCommitmentsCs);
374 
375  auto k = std::make_pair(llmqType, quorumHash);
376  auto it = minableCommitmentsByQuorum.find(k);
377  if (it == minableCommitmentsByQuorum.end()) {
378  // null commitment required
379  ret = CFinalCommitment(Params().GetConsensus().llmqs.at(llmqType), quorumHash);
380  return true;
381  }
382 
383  ret = minableCommitments.at(it->second);
384 
385  return true;
386 }
