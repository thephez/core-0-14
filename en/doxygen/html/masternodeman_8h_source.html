103  template <typename Stream, typename Operation>
104  inline void SerializationOp(Stream& s, Operation ser_action) {
105  LOCK(cs);
106  std::string strVersion;
107  if(ser_action.ForRead()) {
108  READWRITE(strVersion);
109  }
110  else {
111  strVersion = SERIALIZATION_VERSION_STRING;
112  READWRITE(strVersion);
113  }
114 
115  READWRITE(mapMasternodes);
116  READWRITE(mAskedUsForMasternodeList);
117  READWRITE(mWeAskedForMasternodeList);
118  READWRITE(mWeAskedForMasternodeListEntry);
119  READWRITE(mMnbRecoveryRequests);
120  READWRITE(mMnbRecoveryGoodReplies);
121  READWRITE(nLastSentinelPingTime);
122  READWRITE(nDsqCount);
123 
124  READWRITE(mapSeenMasternodeBroadcast);
125  READWRITE(mapSeenMasternodePing);
126  if(ser_action.ForRead() && (strVersion != SERIALIZATION_VERSION_STRING)) {
127  Clear();
128  }
129  }
