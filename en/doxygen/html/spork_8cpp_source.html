35 bool CSporkManager::SporkValueIsActive(int nSporkID, int64_t &nActiveValueRet) const
36 {
37  LOCK(cs);
38 
39  if (!mapSporksActive.count(nSporkID)) return false;
40 
41  // calc how many values we have and how many signers vote for every value
42  std::map<int64_t, int> mapValueCounts;
43  for (const auto& pair: mapSporksActive.at(nSporkID)) {
44  mapValueCounts[pair.second.nValue]++;
45  if (mapValueCounts.at(pair.second.nValue) >= nMinSporkKeys) {
46  // nMinSporkKeys is always more than the half of the max spork keys number,
47  // so there is only one such value and we can stop here
48  nActiveValueRet = pair.second.nValue;
49  return true;
50  }
51  }
52 
53  return false;
54 }

65 void CSporkManager::CheckAndRemove()
66 {
67  LOCK(cs);
68  bool fSporkAddressIsSet = !setSporkPubKeyIDs.empty();
69  assert(fSporkAddressIsSet);
70 
71  auto itActive = mapSporksActive.begin();
72  while (itActive != mapSporksActive.end()) {
73  auto itSignerPair = itActive->second.begin();
74  while (itSignerPair != itActive->second.end()) {
75  if (setSporkPubKeyIDs.find(itSignerPair->first) == setSporkPubKeyIDs.end()) {
76  mapSporksByHash.erase(itSignerPair->second.GetHash());
77  continue;
78  }
79  if (!itSignerPair->second.CheckSignature(itSignerPair->first, false)) {
80  if (!itSignerPair->second.CheckSignature(itSignerPair->first, true)) {
81  mapSporksByHash.erase(itSignerPair->second.GetHash());
82  itActive->second.erase(itSignerPair++);
83  continue;
84  }
85  }
86  ++itSignerPair;
87  }
88  if (itActive->second.empty()) {
89  mapSporksActive.erase(itActive++);
90  continue;
91  }
92  ++itActive;
93  }
94 
95  auto itByHash = mapSporksByHash.begin();
96  while (itByHash != mapSporksByHash.end()) {
97  bool found = false;
98  for (const auto& signer: setSporkPubKeyIDs) {
99  if (itByHash->second.CheckSignature(signer, false) ||
100  itByHash->second.CheckSignature(signer, true)) {
101  found = true;
102  break;
103  }
104  }
105  if (!found) {
106  mapSporksByHash.erase(itByHash++);
107  continue;
108  }
109  ++itByHash;
110  }
111 }

113 void CSporkManager::ProcessSpork(CNode* pfrom, const std::string& strCommand, CDataStream& vRecv, CConnman& connman)
114 {
115  if(fLiteMode) return; // disable all Dash specific functionality
116 
117  if (strCommand == NetMsgType::SPORK) {
118 
119  CSporkMessage spork;
120  vRecv >> spork;
121 
122  uint256 hash = spork.GetHash();
123 
124  std::string strLogMsg;
125  {
126  LOCK(cs_main);
127  connman.RemoveAskFor(hash);
128  if(!chainActive.Tip()) return;
129  strLogMsg = strprintf("SPORK -- hash: %s id: %d value: %10d bestHeight: %d peer=%d", hash.ToString(), spork.nSporkID, spork.nValue, chainActive.Height(), pfrom->id);
130  }
131 
132  if (spork.nTimeSigned > GetAdjustedTime() + 2 * 60 * 60) {
133  LOCK(cs_main);
134  LogPrintf("CSporkManager::ProcessSpork -- ERROR: too far into the future\n");
135  Misbehaving(pfrom->GetId(), 100);
136  return;
137  }
138 
139  CKeyID keyIDSigner;
140  bool fSpork6IsActive = IsSporkActive(SPORK_6_NEW_SIGS);
141  if (!spork.GetSignerKeyID(keyIDSigner, fSpork6IsActive)
142  || !setSporkPubKeyIDs.count(keyIDSigner)) {
143  // Note: unlike for other messages we have to check for new format even with SPORK_6_NEW_SIGS
144  // inactive because SPORK_6_NEW_SIGS default is OFF and it is not the first spork to sync
145  // (and even if it would, spork order can't be guaranteed anyway).
146  if (!spork.GetSignerKeyID(keyIDSigner, !fSpork6IsActive)
147  || !setSporkPubKeyIDs.count(keyIDSigner)) {
148  LOCK(cs_main);
149  LogPrintf("CSporkManager::ProcessSpork -- ERROR: invalid signature\n");
150  Misbehaving(pfrom->GetId(), 100);
151  return;
152  }
153  }
154 
155  {
156  LOCK(cs); // make sure to not lock this together with cs_main
157  if (mapSporksActive.count(spork.nSporkID)) {
158  if (mapSporksActive[spork.nSporkID].count(keyIDSigner)) {
159  if (mapSporksActive[spork.nSporkID][keyIDSigner].nTimeSigned >= spork.nTimeSigned) {
160  LogPrint("spork", "%s seen\n", strLogMsg);
161  return;
162  } else {
163  LogPrintf("%s updated\n", strLogMsg);
164  }
165  } else {
166  LogPrintf("%s new signer\n", strLogMsg);
167  }
168  } else {
169  LogPrintf("%s new\n", strLogMsg);
170  }
171  }
172 
173 
174  {
175  LOCK(cs); // make sure to not lock this together with cs_main
176  mapSporksByHash[hash] = spork;
177  mapSporksActive[spork.nSporkID][keyIDSigner] = spork;
178  }
179  spork.Relay(connman);
180 
181  //does a task if needed
182  int64_t nActiveValue = 0;
183  if (SporkValueIsActive(spork.nSporkID, nActiveValue)) {
184  ExecuteSpork(spork.nSporkID, nActiveValue);
185  }
186 
187  } else if (strCommand == NetMsgType::GETSPORKS) {
188  LOCK(cs); // make sure to not lock this together with cs_main
189  for (const auto& pair : mapSporksActive) {
190  for (const auto& signerSporkPair: pair.second) {
191  connman.PushMessage(pfrom, CNetMsgMaker(pfrom->GetSendVersion()).Make(NetMsgType::SPORK, signerSporkPair.second));
192  }
193  }
194  }
195 
196 }

198 void CSporkManager::ExecuteSpork(int nSporkID, int nValue)
199 {
200  //correct fork via spork technology
201  if(nSporkID == SPORK_12_RECONSIDER_BLOCKS && nValue > 0) {
202  // allow to reprocess 24h of blocks max, which should be enough to resolve any issues
203  int64_t nMaxBlocks = 576;
204  // this potentially can be a heavy operation, so only allow this to be executed once per 10 minutes
205  int64_t nTimeout = 10 * 60;
206 
207  static int64_t nTimeExecuted = 0; // i.e. it was never executed before
208 
209  if(GetTime() - nTimeExecuted < nTimeout) {
210  LogPrint("spork", "CSporkManager::ExecuteSpork -- ERROR: Trying to reconsider blocks, too soon - %d/%d\n", GetTime() - nTimeExecuted, nTimeout);
211  return;
212  }
213 
214  if(nValue > nMaxBlocks) {
215  LogPrintf("CSporkManager::ExecuteSpork -- ERROR: Trying to reconsider too many blocks %d/%d\n", nValue, nMaxBlocks);
216  return;
217  }
218 
219 
220  LogPrintf("CSporkManager::ExecuteSpork -- Reconsider Last %d Blocks\n", nValue);
221 
222  ReprocessBlocks(nValue);
223  nTimeExecuted = GetTime();
224  }

227 bool CSporkManager::UpdateSpork(int nSporkID, int64_t nValue, CConnman& connman)
228 {
229  CSporkMessage spork = CSporkMessage(nSporkID, nValue, GetAdjustedTime());
230 
231  bool fSpork6IsActive = IsSporkActive(SPORK_6_NEW_SIGS);
232  if(spork.Sign(sporkPrivKey, fSpork6IsActive)) {
233  CKeyID keyIDSigner;
234  if (!spork.GetSignerKeyID(keyIDSigner, fSpork6IsActive) || !setSporkPubKeyIDs.count(keyIDSigner)) {
235  LogPrintf("CSporkManager::UpdateSpork: failed to find keyid for private key\n");
236  return false;
237  }
238  spork.Relay(connman);
239  LOCK(cs);
240  mapSporksByHash[spork.GetHash()] = spork;
241  mapSporksActive[nSporkID][keyIDSigner] = spork;
242  return true;
243  }
244 
245  return false;
