273 UniValue GetNextMasternodeForPayment(int heightShift)
274 {
275  int nCount;
276  int nHeight;
277  masternode_info_t mnInfo;
278  CBlockIndex* pindex = NULL;
279  {
280  LOCK(cs_main);
281  pindex = chainActive.Tip();
282  }
283 
284  nHeight = pindex->nHeight + heightShift;
285  mnodeman.UpdateLastPaid(pindex);
286 
287  CScript payeeScript;
288  if (deterministicMNManager->IsDeterministicMNsSporkActive()) {
289  auto payee = deterministicMNManager->GetListAtChainTip().GetMNPayee();
290  if (!payee || !mnodeman.GetMasternodeInfo(payee->proTxHash, mnInfo))
291  return "unknown";
292  payeeScript = payee->pdmnState->scriptPayout;
293  } else {
294  if (!mnodeman.GetNextMasternodeInQueueForPayment(nHeight, true, nCount, mnInfo))
295  return "unknown";
296  payeeScript = GetScriptForDestination(mnInfo.keyIDCollateralAddress);
297  }
298 
299  CTxDestination payeeDest;
300  CBitcoinAddress payeeAddr;
301  if (ExtractDestination(payeeScript, payeeDest)) {
302  payeeAddr = CBitcoinAddress(payeeDest);
303  }
304 
305  UniValue obj(UniValue::VOBJ);
306 
307  obj.push_back(Pair("height", nHeight));
308  obj.push_back(Pair("IP:port", mnInfo.addr.ToString()));
309  obj.push_back(Pair("protocol", mnInfo.nProtocolVersion));
310  obj.push_back(Pair("outpoint", mnInfo.outpoint.ToStringShort()));
311  obj.push_back(Pair("payee", payeeAddr.IsValid() ? payeeAddr.ToString() : "UNKNOWN"));
312  obj.push_back(Pair("lastseen", mnInfo.nTimeLastPing));
313  obj.push_back(Pair("activeseconds", mnInfo.nTimeLastPing - mnInfo.sigTime));
314  return obj;
315 }
