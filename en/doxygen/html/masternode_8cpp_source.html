63 bool CMasternode::UpdateFromNewBroadcast(CMasternodeBroadcast& mnb, CConnman& connman)
64 {
65  if(mnb.sigTime <= sigTime && !mnb.fRecovery) return false;
66 
67  pubKeyMasternode = mnb.pubKeyMasternode;
68  keyIDOwner = mnb.pubKeyMasternode.GetID();
69  legacyKeyIDOperator = mnb.pubKeyMasternode.GetID();
70  keyIDVoting = mnb.pubKeyMasternode.GetID();
71  sigTime = mnb.sigTime;
72  vchSig = mnb.vchSig;
73  nProtocolVersion = mnb.nProtocolVersion;
74  addr = mnb.addr;
75  nPoSeBanScore = 0;
76  nPoSeBanHeight = 0;
77  nTimeLastChecked = 0;
78  int nDos = 0;
79  if(!mnb.lastPing || (mnb.lastPing && mnb.lastPing.CheckAndUpdate(this, true, nDos, connman))) {
80  lastPing = mnb.lastPing;
81  mnodeman.mapSeenMasternodePing.insert(std::make_pair(lastPing.GetHash(), lastPing));
82  }
83  // if it matches our Masternode privkey...
84  if(fMasternodeMode && legacyKeyIDOperator == activeMasternodeInfo.legacyKeyIDOperator) {
85  nPoSeBanScore = -MASTERNODE_POSE_BAN_MAX_SCORE;
86  if(nProtocolVersion == PROTOCOL_VERSION) {
87  // ... and PROTOCOL_VERSION, then we've been remotely activated ...
88  legacyActiveMasternodeManager.ManageState(connman);
89  } else {
90  // ... otherwise we need to reactivate our node, do not add it to the list and do not relay
91  // but also do not ban the node we get this message from
92  LogPrintf("CMasternode::UpdateFromNewBroadcast -- wrong PROTOCOL_VERSION, re-activate your MN: message nProtocolVersion=%d PROTOCOL_VERSION=%d\n", nProtocolVersion, PROTOCOL_VERSION);
93  return false;
94  }
95  }
96  return true;
97 }

141 void CMasternode::Check(bool fForce)
142 {
143  AssertLockHeld(cs_main);
144  LOCK(cs);
145 
146  if(ShutdownRequested()) return;
147 
148  if(!fForce && (GetTime() - nTimeLastChecked < MASTERNODE_CHECK_SECONDS)) return;
149  nTimeLastChecked = GetTime();
150 
151  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state\n", outpoint.ToStringShort(), GetStateString());
152 
153  //once spent, stop doing the checks
154  if(IsOutpointSpent()) return;
155 
156  int nHeight = 0;
157  if(!fUnitTest) {
158  Coin coin;
159  if(!GetUTXOCoin(outpoint, coin)) {
160  nActiveState = MASTERNODE_OUTPOINT_SPENT;
161  LogPrint("masternode", "CMasternode::Check -- Failed to find Masternode UTXO, masternode=%s\n", outpoint.ToStringShort());
162  return;
163  }
164 
165  nHeight = chainActive.Height();
166  }
167 
168  if(IsPoSeBanned()) {
169  if(nHeight < nPoSeBanHeight) return; // too early?
170  // Otherwise give it a chance to proceed further to do all the usual checks and to change its state.
171  // Masternode still will be on the edge and can be banned back easily if it keeps ignoring mnverify
172  // or connect attempts. Will require few mnverify messages to strengthen its position in mn list.
173  LogPrintf("CMasternode::Check -- Masternode %s is unbanned and back in list now\n", outpoint.ToStringShort());
174  DecreasePoSeBanScore();
175  } else if(nPoSeBanScore >= MASTERNODE_POSE_BAN_MAX_SCORE) {
176  nActiveState = MASTERNODE_POSE_BAN;
177  // ban for the whole payment cycle
178  nPoSeBanHeight = nHeight + mnodeman.size();
179  LogPrintf("CMasternode::Check -- Masternode %s is banned till block %d now\n", outpoint.ToStringShort(), nPoSeBanHeight);
180  return;
181  }
182 
183  int nActiveStatePrev = nActiveState;
184  bool fOurMasternode = fMasternodeMode && activeMasternodeInfo.legacyKeyIDOperator == legacyKeyIDOperator;
185 
186  // masternode doesn't meet payment protocol requirements ...
187  bool fRequireUpdate = nProtocolVersion < mnpayments.GetMinMasternodePaymentsProto() ||
188  // or it's our own node and we just updated it to the new protocol but we are still waiting for activation ...
189  (fOurMasternode && nProtocolVersion < PROTOCOL_VERSION);
190 
191  if(fRequireUpdate) {
192  nActiveState = MASTERNODE_UPDATE_REQUIRED;
193  if(nActiveStatePrev != nActiveState) {
194  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
195  }
196  return;
197  }
198 
199  // keep old masternodes on start, give them a chance to receive updates...
200  bool fWaitForPing = !masternodeSync.IsMasternodeListSynced() && !IsPingedWithin(MASTERNODE_MIN_MNP_SECONDS);
201 
202  if(fWaitForPing && !fOurMasternode) {
203  // ...but if it was already expired before the initial check - return right away
204  if(IsExpired() || IsSentinelPingExpired() || IsNewStartRequired()) {
205  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state, waiting for ping\n", outpoint.ToStringShort(), GetStateString());
206  return;
207  }
208  }
209 
210  // don't expire if we are still in "waiting for ping" mode unless it's our own masternode
211  if(!fWaitForPing || fOurMasternode) {
212 
213  if(!IsPingedWithin(MASTERNODE_NEW_START_REQUIRED_SECONDS)) {
214  nActiveState = MASTERNODE_NEW_START_REQUIRED;
215  if(nActiveStatePrev != nActiveState) {
216  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
217  }
218  return;
219  }
220 
221  if(!IsPingedWithin(MASTERNODE_EXPIRATION_SECONDS)) {
222  nActiveState = MASTERNODE_EXPIRED;
223  if(nActiveStatePrev != nActiveState) {
224  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
225  }
226  return;
227  }
228 
229  // part 1: expire based on dashd ping
230  bool fSentinelPingActive = masternodeSync.IsSynced() && mnodeman.IsSentinelPingActive();
231  bool fSentinelPingExpired = fSentinelPingActive && !IsPingedWithin(MASTERNODE_SENTINEL_PING_MAX_SECONDS);
232  LogPrint("masternode", "CMasternode::Check -- outpoint=%s, GetAdjustedTime()=%d, fSentinelPingExpired=%d\n",
233  outpoint.ToStringShort(), GetAdjustedTime(), fSentinelPingExpired);
234 
235  if(fSentinelPingExpired) {
236  nActiveState = MASTERNODE_SENTINEL_PING_EXPIRED;
237  if(nActiveStatePrev != nActiveState) {
238  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
239  }
240  return;
241  }
242  }
243 
244  // We require MNs to be in PRE_ENABLED until they either start to expire or receive a ping and go into ENABLED state
245  // Works on mainnet/testnet only and not the case on regtest/devnet.
246  if (Params().NetworkIDString() != CBaseChainParams::REGTEST && Params().NetworkIDString() != CBaseChainParams::DEVNET) {
247  if (lastPing.sigTime - sigTime < MASTERNODE_MIN_MNP_SECONDS) {
248  nActiveState = MASTERNODE_PRE_ENABLED;
249  if (nActiveStatePrev != nActiveState) {
250  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
251  }
252  return;
253  }
254  }
255 
256  if(!fWaitForPing || fOurMasternode) {
257  // part 2: expire based on sentinel info
258  bool fSentinelPingActive = masternodeSync.IsSynced() && mnodeman.IsSentinelPingActive();
259  bool fSentinelPingExpired = fSentinelPingActive && !lastPing.fSentinelIsCurrent;
260 
261  LogPrint("masternode", "CMasternode::Check -- outpoint=%s, GetAdjustedTime()=%d, fSentinelPingExpired=%d\n",
262  outpoint.ToStringShort(), GetAdjustedTime(), fSentinelPingExpired);
263 
264  if(fSentinelPingExpired) {
265  nActiveState = MASTERNODE_SENTINEL_PING_EXPIRED;
266  if(nActiveStatePrev != nActiveState) {
267  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
268  }
269  return;
270  }
271  }
272 
273  nActiveState = MASTERNODE_ENABLED; // OK
274  if(nActiveStatePrev != nActiveState) {
275  LogPrint("masternode", "CMasternode::Check -- Masternode %s is in %s state now\n", outpoint.ToStringShort(), GetStateString());
276  }
277 }

326 void CMasternode::UpdateLastPaid(const CBlockIndex *pindex, int nMaxBlocksToScanBack)
327 {
328  AssertLockHeld(cs_main);
329 
330  if(!pindex) return;
331 
332  if (deterministicMNManager->IsDeterministicMNsSporkActive(pindex->nHeight)) {
333  auto dmn = deterministicMNManager->GetListForBlock(pindex->GetBlockHash()).GetMNByCollateral(outpoint);
334  if (!dmn || dmn->pdmnState->nLastPaidHeight == -1) {
335  LogPrint("masternode", "CMasternode::UpdateLastPaidBlock -- searching for block with payment to %s -- not found\n", outpoint.ToStringShort());
336  } else {
337  nBlockLastPaid = (int)dmn->pdmnState->nLastPaidHeight;
338  nTimeLastPaid = chainActive[nBlockLastPaid]->nTime;
339  LogPrint("masternode", "CMasternode::UpdateLastPaidBlock -- searching for block with payment to %s -- found new %d\n", outpoint.ToStringShort(), nBlockLastPaid);
340  }
341  return;
342  }
343 
344  const CBlockIndex *BlockReading = pindex;
345 
346  CScript mnpayee = GetScriptForDestination(keyIDCollateralAddress);
347  // LogPrint("mnpayments", "CMasternode::UpdateLastPaidBlock -- searching for block with payment to %s\n", outpoint.ToStringShort());
348 
349  LOCK(cs_mapMasternodeBlocks);
350 
351  for (int i = 0; BlockReading && BlockReading->nHeight > nBlockLastPaid && i < nMaxBlocksToScanBack; i++) {
352  if(mnpayments.mapMasternodeBlocks.count(BlockReading->nHeight) &&
353  mnpayments.mapMasternodeBlocks[BlockReading->nHeight].HasPayeeWithVotes(mnpayee, 2))
354  {
355  CBlock block;
356  if(!ReadBlockFromDisk(block, BlockReading, Params().GetConsensus()))
357  continue; // shouldn't really happen
358 
359  CAmount nMasternodePayment = GetMasternodePayment(BlockReading->nHeight, block.vtx[0]->GetValueOut());
360 
361  for (const auto& txout : block.vtx[0]->vout)
362  if(mnpayee == txout.scriptPubKey && nMasternodePayment == txout.nValue) {
363  nBlockLastPaid = BlockReading->nHeight;
364  nTimeLastPaid = BlockReading->nTime;
365  LogPrint("mnpayments", "CMasternode::UpdateLastPaidBlock -- searching for block with payment to %s -- found new %d\n", outpoint.ToStringShort(), nBlockLastPaid);
366  return;
367  }
368  }
369 
370  if (BlockReading->pprev == nullptr) { assert(BlockReading); break; }
371  BlockReading = BlockReading->pprev;
372  }
373 
374  // Last payment for this masternode wasn't found in latest mnpayments blocks
375  // or it was found in mnpayments blocks but wasn't found in the blockchain.
376  // LogPrint("mnpayments", "CMasternode::UpdateLastPaidBlock -- searching for block with payment to %s -- keeping old %d\n", outpoint.ToStringShort(), nBlockLastPaid);
377 }

509 bool CMasternodeBroadcast::Update(CMasternode* pmn, int& nDos, CConnman& connman)
510 {
511  nDos = 0;
512 
513  AssertLockHeld(cs_main);
514 
515  if(pmn->sigTime == sigTime && !fRecovery) {
516  // mapSeenMasternodeBroadcast in CMasternodeMan::CheckMnbAndUpdateMasternodeList should filter legit duplicates
517  // but this still can happen if we just started, which is ok, just do nothing here.
518  return false;
519  }
520 
521  // this broadcast is older than the one that we already have - it's bad and should never happen
522  // unless someone is doing something fishy

604  // at which collateral became nMasternodeMinimumConfirmations blocks deep.
605  // NOTE: this is not accurate because block timestamp is NOT guaranteed to be 100% correct one.
