102 bool CGovernanceObject::ProcessVote(CNode* pfrom,
103  const CGovernanceVote& vote,
104  CGovernanceException& exception,
105  CConnman& connman)
106 {
107  LOCK(cs);
108 
109  // do not process already known valid votes twice
110  if (fileVotes.HasVote(vote.GetHash())) {
111  // nothing to do here, not an error
112  std::ostringstream ostr;
113  ostr << "CGovernanceObject::ProcessVote -- Already known valid vote";
114  LogPrint("gobject", "%s\n", ostr.str());
115  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_NONE);
116  return false;
117  }
118 
119  if (!mnodeman.Has(vote.GetMasternodeOutpoint())) {
120  std::ostringstream ostr;
121  ostr << "CGovernanceObject::ProcessVote -- Masternode " << vote.GetMasternodeOutpoint().ToStringShort() << " not found";
122  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_WARNING);
123  if (cmmapOrphanVotes.Insert(vote.GetMasternodeOutpoint(), vote_time_pair_t(vote, GetAdjustedTime() + GOVERNANCE_ORPHAN_EXPIRATION_TIME))) {
124  if (pfrom) {
125  mnodeman.AskForMN(pfrom, vote.GetMasternodeOutpoint(), connman);
126  }
127  LogPrintf("%s\n", ostr.str());
128  } else {
129  LogPrint("gobject", "%s\n", ostr.str());
130  }
131  return false;
132  }
133 
134  vote_m_it it = mapCurrentMNVotes.emplace(vote_m_t::value_type(vote.GetMasternodeOutpoint(), vote_rec_t())).first;
135  vote_rec_t& voteRecordRef = it->second;
136  vote_signal_enum_t eSignal = vote.GetSignal();
137  if (eSignal == VOTE_SIGNAL_NONE) {
138  std::ostringstream ostr;
139  ostr << "CGovernanceObject::ProcessVote -- Vote signal: none";
140  LogPrint("gobject", "%s\n", ostr.str());
141  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_WARNING);
142  return false;
143  }
144  if (eSignal > MAX_SUPPORTED_VOTE_SIGNAL) {
145  std::ostringstream ostr;
146  ostr << "CGovernanceObject::ProcessVote -- Unsupported vote signal: " << CGovernanceVoting::ConvertSignalToString(vote.GetSignal());
147  LogPrintf("%s\n", ostr.str());
148  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_PERMANENT_ERROR, 20);
149  return false;
150  }
151  vote_instance_m_it it2 = voteRecordRef.mapInstances.emplace(vote_instance_m_t::value_type(int(eSignal), vote_instance_t())).first;
152  vote_instance_t& voteInstanceRef = it2->second;
153 
154  // Reject obsolete votes
155  if (vote.GetTimestamp() < voteInstanceRef.nCreationTime) {
156  std::ostringstream ostr;
157  ostr << "CGovernanceObject::ProcessVote -- Obsolete vote";
158  LogPrint("gobject", "%s\n", ostr.str());
159  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_NONE);
160  return false;
161  }
162 
163  int64_t nNow = GetAdjustedTime();
164  int64_t nVoteTimeUpdate = voteInstanceRef.nTime;
165  if (governance.AreRateChecksEnabled()) {
166  int64_t nTimeDelta = nNow - voteInstanceRef.nTime;
167  if (nTimeDelta < GOVERNANCE_UPDATE_MIN) {
168  std::ostringstream ostr;
169  ostr << "CGovernanceObject::ProcessVote -- Masternode voting too often"
170  << ", MN outpoint = " << vote.GetMasternodeOutpoint().ToStringShort()
171  << ", governance object hash = " << GetHash().ToString()
172  << ", time delta = " << nTimeDelta;
173  LogPrint("gobject", "%s\n", ostr.str());
174  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_TEMPORARY_ERROR);
175  nVoteTimeUpdate = nNow;
176  return false;
177  }
178  }
179 
180  bool onlyVotingKeyAllowed = nObjectType == GOVERNANCE_OBJECT_PROPOSAL && vote.GetSignal() == VOTE_SIGNAL_FUNDING;
181 
182  // Finally check that the vote is actually valid (done last because of cost of signature verification)
183  if (!vote.IsValid(onlyVotingKeyAllowed)) {
184  std::ostringstream ostr;
185  ostr << "CGovernanceObject::ProcessVote -- Invalid vote"
186  << ", MN outpoint = " << vote.GetMasternodeOutpoint().ToStringShort()
187  << ", governance object hash = " << GetHash().ToString()
188  << ", vote hash = " << vote.GetHash().ToString();
189  LogPrintf("%s\n", ostr.str());
190  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_PERMANENT_ERROR, 20);
191  governance.AddInvalidVote(vote);
192  return false;
193  }
194 
195  if (!mnodeman.AddGovernanceVote(vote.GetMasternodeOutpoint(), vote.GetParentHash())) {
196  std::ostringstream ostr;
197  ostr << "CGovernanceObject::ProcessVote -- Unable to add governance vote"
198  << ", MN outpoint = " << vote.GetMasternodeOutpoint().ToStringShort()
199  << ", governance object hash = " << GetHash().ToString();
200  LogPrint("gobject", "%s\n", ostr.str());
201  exception = CGovernanceException(ostr.str(), GOVERNANCE_EXCEPTION_PERMANENT_ERROR);
202  return false;
203  }
204 
205  voteInstanceRef = vote_instance_t(vote.GetOutcome(), nVoteTimeUpdate, vote.GetTimestamp());
206  fileVotes.AddVote(vote);
207  fDirtyCache = true;
208  return true;
209 }

226 std::set<uint256> CGovernanceObject::RemoveInvalidProposalVotes(const COutPoint& mnOutpoint)
227 {
228  LOCK(cs);
229 
230  if (nObjectType != GOVERNANCE_OBJECT_PROPOSAL) {
231  return {};
232  }
233 
234  auto it = mapCurrentMNVotes.find(mnOutpoint);
235  if (it == mapCurrentMNVotes.end()) {
236  // don't even try as we don't have any votes from this MN
237  return {};
238  }
239 
240  auto removedVotes = fileVotes.RemoveInvalidProposalVotes(mnOutpoint);
241  if (removedVotes.empty()) {
242  return {};
243  }
244 
245  auto nParentHash = GetHash();
246  for (auto jt = it->second.mapInstances.begin(); jt != it->second.mapInstances.end(); ) {
247  CGovernanceVote tmpVote(mnOutpoint, nParentHash, (vote_signal_enum_t)jt->first, jt->second.eOutcome);
248  tmpVote.SetTime(jt->second.nCreationTime);
249  if (removedVotes.count(tmpVote.GetHash())) {
250  jt = it->second.mapInstances.erase(jt);
251  } else {
252  ++jt;
253  }
254  }
255  if (it->second.mapInstances.empty()) {
256  mapCurrentMNVotes.erase(it);
257  }
258 
259  if (!removedVotes.empty()) {
260  std::string removedStr;
261  for (auto& h : removedVotes) {
262  removedStr += strprintf(" %s\n", h.ToString());
263  }
264  LogPrintf("CGovernanceObject::%s -- Removed %d invalid votes for %s from MN %s:\n%s\n", __func__, removedVotes.size(), nParentHash.ToString(), mnOutpoint.ToString(), removedStr);
265  fDirtyCache = true;
266  }
267 
268  return removedVotes;
269 }

315 {
316  std::string strError;
317 
318  if (sporkManager.IsSporkActive(SPORK_6_NEW_SIGS)) {
319  uint256 hash = GetSignatureHash();
320 
321  if (!CHashSigner::SignHash(hash, key, vchSig)) {
322  LogPrintf("CGovernanceObject::Sign -- SignHash() failed\n");
323  return false;
324  }
325 
326  if (!CHashSigner::VerifyHash(hash, keyID, vchSig, strError)) {
327  LogPrintf("CGovernanceObject::Sign -- VerifyHash() failed, error: %s\n", strError);
328  return false;
329  }
330  } else {
331  std::string strMessage = GetSignatureMessage();
332  if (!CMessageSigner::SignMessage(strMessage, vchSig, key)) {
333  LogPrintf("CGovernanceObject::Sign -- SignMessage() failed\n");
334  return false;
335  }
336 
337  if (!CMessageSigner::VerifyMessage(keyID, vchSig, strMessage, strError)) {
338  LogPrintf("CGovernanceObject::Sign -- VerifyMessage() failed, error: %s\n", strError);
339  return false;
340  }
341  }
342 
343  LogPrint("gobject", "CGovernanceObject::Sign -- pubkey id = %s, masternode = %s\n",
344  keyID.ToString(), masternodeOutpoint.ToStringShort());
345 
346  return true;

519 bool CGovernanceObject::IsValidLocally(std::string& strError, bool& fMissingMasternode, bool& fMissingConfirmations, bool fCheckCollateral) const
520 {
521  fMissingMasternode = false;
522  fMissingConfirmations = false;
523 
524  if (fUnparsable) {
525  strError = "Object data unparseable";
526  return false;
527  }
528 
529  switch (nObjectType) {
530  case GOVERNANCE_OBJECT_WATCHDOG: {
531  // watchdogs are deprecated
532  return false;
533  }
534  case GOVERNANCE_OBJECT_PROPOSAL: {
535  CProposalValidator validator(GetDataAsHexString());
536  // Note: It's ok to have expired proposals
537  // they are going to be cleared by CGovernanceManager::UpdateCachesAndClean()
538  // TODO: should they be tagged as "expired" to skip vote downloading?
539  if (!validator.Validate(false)) {
540  strError = strprintf("Invalid proposal data, error messages: %s", validator.GetErrorMessages());
541  return false;
542  }
543  if (fCheckCollateral && !IsCollateralValid(strError, fMissingConfirmations)) {
544  strError = "Invalid proposal collateral";
545  return false;
546  }
547  return true;
548  }
549  case GOVERNANCE_OBJECT_TRIGGER: {
550  if (!fCheckCollateral) {
551  // nothing else we can check here (yet?)
552  return true;
553  }
554 
555  std::string strOutpoint = masternodeOutpoint.ToStringShort();
556  masternode_info_t infoMn;
557  if (!mnodeman.GetMasternodeInfo(masternodeOutpoint, infoMn)) {
558  CMasternode::CollateralStatus err = CMasternode::CheckCollateral(masternodeOutpoint, CKeyID());
559  if (err == CMasternode::COLLATERAL_UTXO_NOT_FOUND) {
560  strError = "Failed to find Masternode UTXO, missing masternode=" + strOutpoint + "\n";
561  } else if (err == CMasternode::COLLATERAL_INVALID_AMOUNT) {
562  strError = "Masternode UTXO should have 1000 DASH, missing masternode=" + strOutpoint + "\n";
563  } else if (err == CMasternode::COLLATERAL_INVALID_PUBKEY) {
564  fMissingMasternode = true;
565  strError = "Masternode not found: " + strOutpoint;
566  } else if (err == CMasternode::COLLATERAL_OK) {
567  // this should never happen with CPubKey() as a param
568  strError = "CheckCollateral critical failure! Masternode: " + strOutpoint;
569  }
570 
571  return false;
572  }
573 
574  // Check that we have a valid MN signature
575  if (deterministicMNManager->IsDeterministicMNsSporkActive()) {
576  if (!CheckSignature(infoMn.blsPubKeyOperator)) {
577  strError = "Invalid masternode signature for: " + strOutpoint + ", pubkey id = " + infoMn.blsPubKeyOperator.ToString();
578  return false;
579  }
580  } else {
581  if (!CheckSignature(infoMn.legacyKeyIDOperator)) {
582  strError = "Invalid masternode signature for: " + strOutpoint + ", pubkey id = " + infoMn.legacyKeyIDOperator.ToString();
583  return false;
584  }
585  }
586 
587  return true;
588  }
589  default: {
590  strError = strprintf("Invalid object type %d", nObjectType);
591  return false;
592  }
593  }

611 bool CGovernanceObject::IsCollateralValid(std::string& strError, bool& fMissingConfirmations) const
612 {
613  strError = "";
614  fMissingConfirmations = false;
615  CAmount nMinFee = GetMinCollateralFee();
616  uint256 nExpectedHash = GetHash();
617 
618  CTransactionRef txCollateral;
619  uint256 nBlockHash;
620 
621  // RETRIEVE TRANSACTION IN QUESTION
622 
623  if (!GetTransaction(nCollateralHash, txCollateral, Params().GetConsensus(), nBlockHash, true)) {
624  strError = strprintf("Can't find collateral tx %s", nCollateralHash.ToString());
625  LogPrintf("CGovernanceObject::IsCollateralValid -- %s\n", strError);
626  return false;
627  }
628 
629  if (nBlockHash == uint256()) {
630  strError = strprintf("Collateral tx %s is not mined yet", txCollateral->ToString());
631  LogPrintf("CGovernanceObject::IsCollateralValid -- %s\n", strError);
632  return false;
633  }
634 
635  if (txCollateral->vout.size() < 1) {
636  strError = strprintf("tx vout size less than 1 | %d", txCollateral->vout.size());
637  LogPrintf("CGovernanceObject::IsCollateralValid -- %s\n", strError);
638  return false;
639  }
640 
641  // LOOK FOR SPECIALIZED GOVERNANCE SCRIPT (PROOF OF BURN)
642 
643  CScript findScript;
644  findScript << OP_RETURN << ToByteVector(nExpectedHash);
645 
646  DBG(std::cout << "IsCollateralValid: txCollateral->vout.size() = " << txCollateral->vout.size() << std::endl;);
647 
648  DBG(std::cout << "IsCollateralValid: findScript = " << ScriptToAsmStr(findScript, false) << std::endl;);
649 
650  DBG(std::cout << "IsCollateralValid: nMinFee = " << nMinFee << std::endl;);
651 
652 
653  bool foundOpReturn = false;
