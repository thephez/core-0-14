382 CScript _createmultisig_redeemScript(const UniValue& params)
383 {
384  int nRequired = params[0].get_int();
385  const UniValue& keys = params[1].get_array();
386 
387  // Gather public keys
388  if (nRequired < 1)
389  throw std::runtime_error("a multisignature address must require at least one key to redeem");
390  if ((int)keys.size() < nRequired)
391  throw std::runtime_error(
392  strprintf("not enough keys supplied "
393  "(got %u keys, but need at least %d to redeem)", keys.size(), nRequired));
394  if (keys.size() > 16)
395  throw std::runtime_error("Number of addresses involved in the multisignature address creation > 16\nReduce the number");
396  std::vector<CPubKey> pubkeys;
397  pubkeys.resize(keys.size());
398  for (unsigned int i = 0; i < keys.size(); i++)
399  {
400  const std::string& ks = keys[i].get_str();
401 #ifdef ENABLE_WALLET
402  // Case 1: Dash address and we have full public key:
403  CBitcoinAddress address(ks);
404  if (pwalletMain && address.IsValid())
405  {
406  CKeyID keyID;
407  if (!address.GetKeyID(keyID))
408  throw std::runtime_error(
409  strprintf("%s does not refer to a key",ks));
410  CPubKey vchPubKey;
411  if (!pwalletMain->GetPubKey(keyID, vchPubKey))
412  throw std::runtime_error(
413  strprintf("no full public key for address %s",ks));
414  if (!vchPubKey.IsFullyValid())
415  throw std::runtime_error(" Invalid public key: "+ks);
416  pubkeys[i] = vchPubKey;
417  }
418 
419  // Case 2: hex public key
420  else
421 #endif
422  if (IsHex(ks))
423  {
424  CPubKey vchPubKey(ParseHex(ks));
425  if (!vchPubKey.IsFullyValid())
426  throw std::runtime_error(" Invalid public key: "+ks);
427  pubkeys[i] = vchPubKey;
428  }
429  else
430  {
431  throw std::runtime_error(" Invalid public key: "+ks);
432  }
433  }
434  CScript result = GetScriptForMultisig(nRequired, pubkeys);
435 
436  if (result.size() > MAX_SCRIPT_ELEMENT_SIZE)
437  throw std::runtime_error(
438  strprintf("redeemScript exceeds size limit: %d > %d", result.size(), MAX_SCRIPT_ELEMENT_SIZE));
439 
440  return result;
