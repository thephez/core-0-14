580  if (!(s.GetType() & SER_GETHASH))
581  READWRITE(nVersion);
583  READWRITE(nCreditDebit);
584  READWRITE(nTime);
585  READWRITE(LIMITED_STRING(strOtherAccount, 65536));
586 
587  if (!ser_action.ForRead())
588  {
589  WriteOrderPos(nOrderPos, mapValue);
590 
591  if (!(mapValue.empty() && _ssExtra.empty()))
592  {
593  CDataStream ss(s.GetType(), s.GetVersion());
594  ss.insert(ss.begin(), '\0');
595  ss << mapValue;
596  ss.insert(ss.end(), _ssExtra.begin(), _ssExtra.end());
597  strComment.append(ss.str());
598  }
599  }
600 
601  READWRITE(LIMITED_STRING(strComment, 65536));
602 
603  size_t nSepPos = strComment.find("\0", 0, 1);
604  if (ser_action.ForRead())
605  {
606  mapValue.clear();
607  if (std::string::npos != nSepPos)
608  {
609  CDataStream ss(std::vector<char>(strComment.begin() + nSepPos + 1, strComment.end()), s.GetType(), s.GetVersion());
610  ss >> mapValue;
611  _ssExtra = std::vector<char>(ss.begin(), ss.end());
612  }
613  ReadOrderPos(nOrderPos, mapValue);
614  }
615  if (std::string::npos != nSepPos)
616  strComment.erase(nSepPos);
617 
618  mapValue.erase("n");
619  }
