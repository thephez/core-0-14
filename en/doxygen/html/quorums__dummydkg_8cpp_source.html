28 void CDummyDKG::ProcessMessage(CNode* pfrom, const std::string& strCommand, CDataStream& vRecv, CConnman& connman)
29 {
30  if (strCommand == NetMsgType::QCONTRIB) {
31  if (!sporkManager.IsSporkActive(SPORK_17_QUORUM_DKG_ENABLED)) {
32  return;
33  }
34 
35  CDummyContribution qc;
36  try {
37  vRecv >> qc;
38  } catch (...) {
39  // When we switch to the real DKG, non-upgraded nodes will get to this point. Let them just ignore the
40  // incompatible messages
41  return;
42  }
43 
44  uint256 hash = ::SerializeHash(qc);
45  {
46  LOCK(cs_main);
47  connman.RemoveAskFor(hash);
48  }
49 
50  ProcessDummyContribution(pfrom->id, qc);
51  } else if (strCommand == NetMsgType::QDCOMMITMENT) {
52  if (!Params().GetConsensus().fLLMQAllowDummyCommitments) {
53  Misbehaving(pfrom->id, 100);
54  return;
55  }
56  if (!sporkManager.IsSporkActive(SPORK_17_QUORUM_DKG_ENABLED)) {
57  return;
58  }
59 
60  CDummyCommitment qc;
61  vRecv >> qc;
62 
63  uint256 hash = ::SerializeHash(qc);
64  {
65  LOCK(cs_main);
66  connman.RemoveAskFor(hash);
67  }
68 
69  ProcessDummyCommitment(pfrom->id, qc);
70  }

73 void CDummyDKG::ProcessDummyContribution(NodeId from, const llmq::CDummyContribution& qc)
74 {
75  if (!Params().GetConsensus().llmqs.count((Consensus::LLMQType)qc.llmqType)) {
76  LOCK(cs_main);
77  LogPrintf("CDummyDKG::%s -- invalid commitment type %d, peer=%d\n", __func__,
78  qc.llmqType, from);
79  if (from != -1) {
80  Misbehaving(from, 100);
81  }
82  return;
83  }
84 
85  auto type = (Consensus::LLMQType)qc.llmqType;
86  const auto& params = Params().GetConsensus().llmqs.at(type);
87 
88  int curQuorumHeight;
89  const CBlockIndex* quorumIndex;
90  {
91  LOCK(cs_main);
92  curQuorumHeight = chainActive.Height() - (chainActive.Height() % params.dkgInterval);
93  quorumIndex = chainActive[curQuorumHeight];
94  }
95  uint256 quorumHash = quorumIndex->GetBlockHash();
96  if (qc.quorumHash != quorumHash) {
97  LogPrintf("CDummyDKG::%s -- dummy contrinution for wrong quorum, peer=%d\n", __func__,
98  from);
99  return;
100  }
101 
102  auto members = CLLMQUtils::GetAllQuorumMembers(type, qc.quorumHash);
103  if (members.size() != params.size) {
104  LOCK(cs_main);
105  LogPrintf("CDummyDKG::%s -- invalid members count %d, peer=%d\n", __func__,
106  members.size(), from);
107  if (from != -1) {
108  Misbehaving(from, 100);
109  }
110  return;
111  }
112  if (qc.signer >= members.size()) {
113  LOCK(cs_main);
114  LogPrintf("CDummyDKG::%s -- invalid signer %d, peer=%d\n", __func__,
115  qc.signer, from);
116  if (from != -1) {
117  Misbehaving(from, 100);
118  }
119  return;
120  }
121 
122  auto signer = members[qc.signer];
123 
124  {
125  LOCK(sessionCs);
126  if (curSessions[type].dummyContributionsFromMembers.count(signer->proTxHash)) {
127  return;
128  }
129  }
130 
131  // verify member sig
132  if (!qc.sig.VerifyInsecure(signer->pdmnState->pubKeyOperator, qc.GetSignHash())) {
133  LOCK(cs_main);
134  LogPrintf("CDummyDKG::%s -- invalid memberSig, peer=%d\n", __func__,
135  from);
136  if (from != -1) {
137  Misbehaving(from, 100);
138  }
139  return;
140  }
141 
142  LogPrintf("CDummyDKG::%s -- processed dummy contribution for quorum %s:%d, signer=%d, peer=%d\n", __func__,
143  qc.quorumHash.ToString(), qc.llmqType, qc.signer, from);
144 
145  uint256 hash = ::SerializeHash(qc);
146  {
147  LOCK(sessionCs);
148  curSessions[type].dummyContributions[hash] = qc;
149  curSessions[type].dummyContributionsFromMembers[signer->proTxHash] = hash;
150  }
151 
152  CInv inv(MSG_QUORUM_DUMMY_CONTRIBUTION, hash);
153  g_connman->RelayInv(inv, DMN_PROTO_VERSION);

290 {

322 {

398  qc.quorumHash = quorumHash;
399  qc.signer = (uint16_t)myIdx;
400  qc.validMembers = validMembers;
401 
402  qc.membersSig = activeMasternodeInfo.blsKeyOperator->Sign(commitmentHash);
403 
404  CBLSSecretKey skShare;
405  if (!skShare.SecretKeyShare(svec, CBLSId::FromHash(signer->proTxHash))) {
406  return;
407  }
408  qc.quorumSig = skShare.Sign(commitmentHash);
409 
410  ProcessDummyCommitment(-1, qc);

413 void CDummyDKG::CreateFinalCommitment(Consensus::LLMQType llmqType, const CBlockIndex* pindex)
414 {
415  const auto& params = Params().GetConsensus().llmqs.at(llmqType);
416  int quorumHeight = pindex->nHeight - (pindex->nHeight % params.dkgInterval);
417  const CBlockIndex* quorumIndex;
418  {
419  LOCK(cs_main);
420  quorumIndex = chainActive[quorumHeight];
421  }
422  uint256 quorumHash = quorumIndex->GetBlockHash();
423 
424  auto members = CLLMQUtils::GetAllQuorumMembers(llmqType, quorumHash);
425  if (members.size() != params.size) {
426  return;
427  }
428 
429  auto svec = BuildDeterministicSvec(llmqType, quorumHash);
430  auto vvec = BuildVvec(svec);
431  auto vvecHash = ::SerializeHash(vvec);
432 
433  LOCK(sessionCs);
434  auto& session = curSessions[llmqType];
435 
436  for (const auto& p : session.dummyCommitmentsFromMembers) {
437  const auto& commitmentHash = p.first;
438  if (p.second.size() < params.minSize) {
439  continue;
440  }
441 
442  const auto& firstQc = session.dummyCommitments[p.second.begin()->second];
443 
444  CFinalCommitment fqc(params, quorumHash);
445  fqc.validMembers = firstQc.validMembers;
446  fqc.quorumPublicKey = vvec[0];
447  fqc.quorumVvecHash = vvecHash;
448 
449  std::vector<CBLSSignature> quorumSigs;
450  std::vector<CBLSId> quorumSigIds;
451  std::vector<CBLSSignature> memberSigs;
452  std::vector<CBLSPublicKey> memberPubKeys;
453 
454  for (const auto& p2 : p.second) {
455  const auto& proTxHash = p2.first;
456  const auto& qc = session.dummyCommitments[p2.second];
457 
458  int signerIdx = -1;
459  for (size_t i = 0; i < members.size(); i++) {
460  if (members[i]->proTxHash == proTxHash) {
461  signerIdx = (int)i;
462  break;
463  }
464  }
465  if (signerIdx == -1) {
466  break;
467  }
468  fqc.signers[signerIdx] = true;
469  if (quorumSigs.size() < params.threshold) {
470  quorumSigs.emplace_back(qc.quorumSig);
471  quorumSigIds.emplace_back(CBLSId::FromHash(proTxHash));
472  }
473  memberSigs.emplace_back(qc.membersSig);
474  memberPubKeys.emplace_back(members[signerIdx]->pdmnState->pubKeyOperator);
475  }
476 
477  if (!fqc.quorumSig.Recover(quorumSigs, quorumSigIds)) {
478  LogPrintf("CDummyDKG::%s -- recovery failed for quorum %s:%d, validMembers=%d, signers=%d\n", __func__,
479  fqc.quorumHash.ToString(), fqc.llmqType, fqc.CountValidMembers(), fqc.CountSigners());
480  continue;
481  }
482  fqc.membersSig = CBLSSignature::AggregateSecure(memberSigs, memberPubKeys, commitmentHash);
483 
484  if (!fqc.Verify(members, true)) {
485  LogPrintf("CDummyDKG::%s -- self created final commitment is invalid for quorum %s:%d, validMembers=%d, signers=%d\n", __func__,
486  fqc.quorumHash.ToString(), fqc.llmqType, fqc.CountValidMembers(), fqc.CountSigners());
487  continue;
488  }
489 
490  LogPrintf("CDummyDKG::%s -- self created final commitment for quorum %s:%d, validMembers=%d, signers=%d\n", __func__,
491  fqc.quorumHash.ToString(), fqc.llmqType, fqc.CountValidMembers(), fqc.CountSigners());
492  quorumBlockProcessor->AddMinableCommitment(fqc);
493  }
494 
495  // reset for next round
496  curSessions[llmqType] = CDummyDKGSession();
497 }

500 {

517 // The returned secret key vector is NOT SECURE AT ALL!!
518 // It is known by everyone. This is only for testnet/devnet/regtest, so this is fine. Also, we won't do any meaningful
519 // things with the commitments. This is only needed to make the final commitments validate
520 BLSSecretKeyVector CDummyDKG::BuildDeterministicSvec(Consensus::LLMQType llmqType, const uint256& quorumHash)
521 {
522  const auto& params = Params().GetConsensus().llmqs.at(llmqType);
523 
524  CHash256 seed_;
525  seed_.Write((unsigned char*)&llmqType, 1);
526  seed_.Write(quorumHash.begin(), quorumHash.size());
527 
528  uint256 seed;
529  seed_.Finalize(seed.begin());
530 
531  BLSSecretKeyVector svec;
532  svec.reserve(params.size);
533  for (size_t i = 0; i < params.threshold; i++) {
534  CBLSSecretKey sk;
535  while (true) {
536  seed = ::SerializeHash(seed);
537  sk.SetBuf(seed.begin(), seed.size());
538  if (sk.IsValid()) {
539  break;
540  }
541  }
542  svec.emplace_back(sk);
543  }
544 
545  return svec;
546 }
