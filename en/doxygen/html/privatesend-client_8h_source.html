58 
59  CPendingDsaRequest(const CService& addr_, const CPrivateSendAccept& dsa_) :
60  addr(addr_),
61  dsa(dsa_),
62  nTimeCreated(GetTime())
63  {
64  }
65 
66  CService GetAddr() { return addr; }
67  CPrivateSendAccept GetDSA() { return dsa; }

70  friend bool operator==(const CPendingDsaRequest& a, const CPendingDsaRequest& b)

72  return a.addr == b.addr && a.dsa == b.dsa;

74  friend bool operator!=(const CPendingDsaRequest& a, const CPendingDsaRequest& b)
75  {
76  return !(a == b);
77  }
78  explicit operator bool() const
79  {
