217 void CWalletDB::ListAccountCreditDebit(const std::string& strAccount, std::list<CAccountingEntry>& entries)
218 {
219  bool fAllAccounts = (strAccount == "*");
220 
221  Dbc* pcursor = GetCursor();
222  if (!pcursor)
223  throw std::runtime_error(std::string(__func__) + ": cannot create DB cursor");
224  bool setRange = true;
225  while (true)
226  {
227  // Read next record
228  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
229  if (setRange)
230  ssKey << std::make_pair(std::string("acentry"), std::make_pair((fAllAccounts ? std::string("") : strAccount), uint64_t(0)));
231  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
232  int ret = ReadAtCursor(pcursor, ssKey, ssValue, setRange);
233  setRange = false;
234  if (ret == DB_NOTFOUND)
235  break;
236  else if (ret != 0)
237  {
238  pcursor->close();
239  throw std::runtime_error(std::string(__func__) + ": error scanning DB");
240  }
241 
242  // Unserialize
243  std::string strType;
244  ssKey >> strType;
245  if (strType != "acentry")
246  break;
247  CAccountingEntry acentry;
248  ssKey >> acentry.strAccount;
249  if (!fAllAccounts && acentry.strAccount != strAccount)
250  break;
251 
252  ssValue >> acentry;
253  ssKey >> acentry.nEntryNo;
254  entries.push_back(acentry);
255  }
256 
257  pcursor->close();
258 }

279 bool
280 ReadKeyValue(CWallet* pwallet, CDataStream& ssKey, CDataStream& ssValue,
281  CWalletScanState &wss, std::string& strType, std::string& strErr)
282 {
283  try {
284  // Unserialize
285  // Taking advantage of the fact that pair serialization
286  // is just the two items serialized one after the other
287  ssKey >> strType;
288  if (strType == "name")
289  {
290  std::string strAddress;
291  ssKey >> strAddress;
292  ssValue >> pwallet->mapAddressBook[CBitcoinAddress(strAddress).Get()].name;
293  }
294  else if (strType == "purpose")
295  {
296  std::string strAddress;
297  ssKey >> strAddress;
298  ssValue >> pwallet->mapAddressBook[CBitcoinAddress(strAddress).Get()].purpose;
299  }
300  else if (strType == "tx")
301  {
302  uint256 hash;
303  ssKey >> hash;
304  CWalletTx wtx;
305  ssValue >> wtx;
306  CValidationState state;
307  if (!(CheckTransaction(wtx, state) && (wtx.GetHash() == hash) && state.IsValid()))
308  return false;
309 
310  // Undo serialize changes in 31600
311  if (31404 <= wtx.fTimeReceivedIsTxTime && wtx.fTimeReceivedIsTxTime <= 31703)
312  {
313  if (!ssValue.empty())
314  {
315  char fTmp;
316  char fUnused;
317  ssValue >> fTmp >> fUnused >> wtx.strFromAccount;
318  strErr = strprintf("LoadWallet() upgrading tx ver=%d %d '%s' %s",
319  wtx.fTimeReceivedIsTxTime, fTmp, wtx.strFromAccount, hash.ToString());
320  wtx.fTimeReceivedIsTxTime = fTmp;
321  }
322  else
323  {
324  strErr = strprintf("LoadWallet() repairing tx ver=%d %s", wtx.fTimeReceivedIsTxTime, hash.ToString());
325  wtx.fTimeReceivedIsTxTime = 0;
326  }
327  wss.vWalletUpgrade.push_back(hash);
328  }
329 
330  if (wtx.nOrderPos == -1)
331  wss.fAnyUnordered = true;
332 
333  pwallet->LoadToWallet(wtx);
334  }
335  else if (strType == "acentry")
336  {
337  std::string strAccount;
338  ssKey >> strAccount;
339  uint64_t nNumber;
340  ssKey >> nNumber;
341  if (nNumber > nAccountingEntryNumber)
342  nAccountingEntryNumber = nNumber;
343 
344  if (!wss.fAnyUnordered)
345  {
346  CAccountingEntry acentry;
347  ssValue >> acentry;
348  if (acentry.nOrderPos == -1)
349  wss.fAnyUnordered = true;
350  }
351  }
352  else if (strType == "watchs")
353  {
354  wss.nWatchKeys++;
355  CScript script;
356  ssKey >> *(CScriptBase*)(&script);
357  char fYes;
358  ssValue >> fYes;
359  if (fYes == '1')
360  pwallet->LoadWatchOnly(script);
361  }
362  else if (strType == "key" || strType == "wkey")
363  {
364  CPubKey vchPubKey;
365  ssKey >> vchPubKey;
366  if (!vchPubKey.IsValid())
367  {
368  strErr = "Error reading wallet database: CPubKey corrupt";
369  return false;
370  }
371  CKey key;
372  CPrivKey pkey;
373  uint256 hash;
374 
375  if (strType == "key")
376  {
377  wss.nKeys++;
378  ssValue >> pkey;
379  } else {
380  CWalletKey wkey;
381  ssValue >> wkey;
382  pkey = wkey.vchPrivKey;
383  }
384 
385  // Old wallets store keys as "key" [pubkey] => [privkey]
386  // ... which was slow for wallets with lots of keys, because the public key is re-derived from the private key
387  // using EC operations as a checksum.
388  // Newer wallets store keys as "key"[pubkey] => [privkey][hash(pubkey,privkey)], which is much faster while
389  // remaining backwards-compatible.
390  try
391  {
392  ssValue >> hash;
393  }
394  catch (...) {}
395 
396  bool fSkipCheck = false;
397 
398  if (!hash.IsNull())
399  {
400  // hash pubkey/privkey to accelerate wallet load
401  std::vector<unsigned char> vchKey;
402  vchKey.reserve(vchPubKey.size() + pkey.size());
403  vchKey.insert(vchKey.end(), vchPubKey.begin(), vchPubKey.end());
404  vchKey.insert(vchKey.end(), pkey.begin(), pkey.end());
405 
406  if (Hash(vchKey.begin(), vchKey.end()) != hash)
407  {
408  strErr = "Error reading wallet database: CPubKey/CPrivKey corrupt";
409  return false;
410  }
411 
412  fSkipCheck = true;
413  }
414 
415  if (!key.Load(pkey, vchPubKey, fSkipCheck))
416  {
417  strErr = "Error reading wallet database: CPrivKey corrupt";
418  return false;
419  }
420  if (!pwallet->LoadKey(key, vchPubKey))
421  {
422  strErr = "Error reading wallet database: LoadKey failed";
423  return false;
424  }
425  }
426  else if (strType == "mkey")
427  {
428  unsigned int nID;
429  ssKey >> nID;
430  CMasterKey kMasterKey;
431  ssValue >> kMasterKey;
432  if(pwallet->mapMasterKeys.count(nID) != 0)
433  {
434  strErr = strprintf("Error reading wallet database: duplicate CMasterKey id %u", nID);
435  return false;
436  }
437  pwallet->mapMasterKeys[nID] = kMasterKey;
438  if (pwallet->nMasterKeyMaxID < nID)
439  pwallet->nMasterKeyMaxID = nID;
440  }
441  else if (strType == "ckey")
442  {
443  CPubKey vchPubKey;
444  ssKey >> vchPubKey;
445  if (!vchPubKey.IsValid())
446  {
447  strErr = "Error reading wallet database: CPubKey corrupt";
448  return false;
449  }
450  std::vector<unsigned char> vchPrivKey;
451  ssValue >> vchPrivKey;
452  wss.nCKeys++;
453 
454  if (!pwallet->LoadCryptedKey(vchPubKey, vchPrivKey))
455  {
456  strErr = "Error reading wallet database: LoadCryptedKey failed";
457  return false;
458  }
459  wss.fIsEncrypted = true;
460  }
461  else if (strType == "keymeta" || strType == "watchmeta")
462  {
463  CTxDestination keyID;
464  if (strType == "keymeta")
465  {
466  CPubKey vchPubKey;
467  ssKey >> vchPubKey;
468  keyID = vchPubKey.GetID();
469  }
470  else if (strType == "watchmeta")
471  {
472  CScript script;
473  ssKey >> *(CScriptBase*)(&script);
474  keyID = CScriptID(script);
475  }
476 
477  CKeyMetadata keyMeta;
478  ssValue >> keyMeta;
479  wss.nKeyMeta++;
480 
481  pwallet->LoadKeyMetadata(keyID, keyMeta);
482  }
483  else if (strType == "defaultkey")
484  {
485  ssValue >> pwallet->vchDefaultKey;
486  }
487  else if (strType == "pool")
488  {
489  int64_t nIndex;
490  ssKey >> nIndex;
491  CKeyPool keypool;
492  ssValue >> keypool;
493  pwallet->LoadKeyPool(nIndex, keypool);
494  }
495  else if (strType == "version")
496  {
497  ssValue >> wss.nFileVersion;
498  if (wss.nFileVersion == 10300)
499  wss.nFileVersion = 300;
500  }
501  else if (strType == "cscript")
502  {
503  uint160 hash;
504  ssKey >> hash;
505  CScript script;
506  ssValue >> *(CScriptBase*)(&script);
507  if (!pwallet->LoadCScript(script))
508  {
509  strErr = "Error reading wallet database: LoadCScript failed";
510  return false;
511  }
512  }
513  else if (strType == "orderposnext")
514  {
515  ssValue >> pwallet->nOrderPosNext;
516  }
517  else if (strType == "destdata")
518  {
519  std::string strAddress, strKey, strValue;
520  ssKey >> strAddress;
521  ssKey >> strKey;
522  ssValue >> strValue;
523  if (!pwallet->LoadDestData(CBitcoinAddress(strAddress).Get(), strKey, strValue))
524  {
525  strErr = "Error reading wallet database: LoadDestData failed";
526  return false;
527  }
528  }
529  else if (strType == "hdchain")
530  {
531  CHDChain chain;
532  ssValue >> chain;
533  if (!pwallet->SetHDChain(chain, true))
534  {
535  strErr = "Error reading wallet database: SetHDChain failed";
536  return false;
537  }
538  }
539  else if (strType == "chdchain")
540  {
541  CHDChain chain;
542  ssValue >> chain;
543  if (!pwallet->SetCryptedHDChain(chain, true))
544  {
545  strErr = "Error reading wallet database: SetHDCryptedChain failed";
546  return false;
547  }
548  }
549  else if (strType == "hdpubkey")
550  {
551  CPubKey vchPubKey;
552  ssKey >> vchPubKey;
553 
554  CHDPubKey hdPubKey;
555  ssValue >> hdPubKey;
556 
557  if(vchPubKey != hdPubKey.extPubKey.pubkey)
558  {
559  strErr = "Error reading wallet database: CHDPubKey corrupt";
560  return false;
561  }
562  if (!pwallet->LoadHDPubKey(hdPubKey))
563  {
564  strErr = "Error reading wallet database: LoadHDPubKey failed";
565  return false;
566  }
567  }
568  } catch (...)
569  {
570  return false;
571  }
572  return true;
573 }

582 DBErrors CWalletDB::LoadWallet(CWallet* pwallet)
583 {
584  pwallet->vchDefaultKey = CPubKey();
585  CWalletScanState wss;
586  bool fNoncriticalErrors = false;
587  DBErrors result = DB_LOAD_OK;
588 
589  LOCK(pwallet->cs_wallet);
590  try {
591  int nMinVersion = 0;
592  if (Read((std::string)"minversion", nMinVersion))
593  {
594  if (nMinVersion > CLIENT_VERSION)
595  return DB_TOO_NEW;
596  pwallet->LoadMinVersion(nMinVersion);
597  }
598 
599  // Get cursor
600  Dbc* pcursor = GetCursor();
601  if (!pcursor)
602  {
603  LogPrintf("Error getting wallet database cursor\n");
604  return DB_CORRUPT;
605  }
606 
607  while (true)
608  {
609  // Read next record
610  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
611  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
612  int ret = ReadAtCursor(pcursor, ssKey, ssValue);
613  if (ret == DB_NOTFOUND)
614  break;
615  else if (ret != 0)
616  {
617  LogPrintf("Error reading next record from wallet database\n");
618  return DB_CORRUPT;
619  }
620 
621  // Try to be tolerant of single corrupt records:
622  std::string strType, strErr;
623  if (!ReadKeyValue(pwallet, ssKey, ssValue, wss, strType, strErr))
624  {
625  // losing keys is considered a catastrophic error, anything else
626  // we assume the user can live with:
627  if (IsKeyType(strType))
628  result = DB_CORRUPT;
629  else
630  {
631  // Leave other errors alone, if we try to fix them we might make things worse.
632  fNoncriticalErrors = true; // ... but do warn the user there is something wrong.
633  if (strType == "tx")
634  // Rescan if there is a bad transaction record:
635  SoftSetBoolArg("-rescan", true);
636  }
637  }
638  if (!strErr.empty())
639  LogPrintf("%s\n", strErr);
640  }
641  pcursor->close();
642 
643  // Store initial external keypool size since we mostly use external keys in mixing
644  pwallet->nKeysLeftSinceAutoBackup = pwallet->KeypoolCountExternalKeys();
645  LogPrintf("nKeysLeftSinceAutoBackup: %d\n", pwallet->nKeysLeftSinceAutoBackup);
646  }
647  catch (const boost::thread_interrupted&) {
648  throw;
649  }
650  catch (...) {
651  result = DB_CORRUPT;
652  }
653 
654  if (fNoncriticalErrors && result == DB_LOAD_OK)
655  result = DB_NONCRITICAL_ERROR;
656 
657  // Any wallet corruption at all: skip any rewriting or
658  // upgrading, we don't want to make it worse.
659  if (result != DB_LOAD_OK)
660  return result;
661 
662  LogPrintf("nFileVersion = %d\n", wss.nFileVersion);
663 
664  LogPrintf("Keys: %u plaintext, %u encrypted, %u w/ metadata, %u total\n",
665  wss.nKeys, wss.nCKeys, wss.nKeyMeta, wss.nKeys + wss.nCKeys);
666 
667  // nTimeFirstKey is only reliable if all keys have metadata
668  if ((wss.nKeys + wss.nCKeys + wss.nWatchKeys) != wss.nKeyMeta)
669  pwallet->UpdateTimeFirstKey(1);
670 
671  BOOST_FOREACH(uint256 hash, wss.vWalletUpgrade)
672  WriteTx(pwallet->mapWallet[hash]);
673 
674  // Rewrite encrypted wallets of versions 0.4.0 and 0.5.0rc:
675  if (wss.fIsEncrypted && (wss.nFileVersion == 40000 || wss.nFileVersion == 50000))
676  return DB_NEED_REWRITE;
677 
678  if (wss.nFileVersion < CLIENT_VERSION) // Update
679  WriteVersion(CLIENT_VERSION);
680 
681  if (wss.fAnyUnordered)
682  result = pwallet->ReorderTransactions();
683 
684  pwallet->laccentries.clear();
685  ListAccountCreditDebit("*", pwallet->laccentries);

881 bool CWalletDB::Recover(CDBEnv& dbenv, const std::string& filename, bool fOnlyKeys)
882 {
883  // Recovery procedure:
884  // move wallet file to wallet.timestamp.bak
885  // Call Salvage with fAggressive=true to
886  // get as much data as possible.
887  // Rewrite salvaged data to fresh wallet file
888  // Set -rescan so any missing transactions will be
889  // found.
890  int64_t now = GetTime();
891  std::string newFilename = strprintf("wallet.%d.bak", now);
892 
893  int result = dbenv.dbenv->dbrename(NULL, filename.c_str(), NULL,
894  newFilename.c_str(), DB_AUTO_COMMIT);
895  if (result == 0)
896  LogPrintf("Renamed %s to %s\n", filename, newFilename);
897  else
898  {
899  LogPrintf("Failed to rename %s to %s\n", filename, newFilename);
900  return false;
901  }
902 
903  std::vector<CDBEnv::KeyValPair> salvagedData;
904  bool fSuccess = dbenv.Salvage(newFilename, true, salvagedData);
905  if (salvagedData.empty())
906  {
907  LogPrintf("Salvage(aggressive) found no records in %s.\n", newFilename);
908  return false;
909  }
910  LogPrintf("Salvage(aggressive) found %u records\n", salvagedData.size());
911 
912  std::unique_ptr<Db> pdbCopy(new Db(dbenv.dbenv, 0));
913  int ret = pdbCopy->open(NULL, // Txn pointer
914  filename.c_str(), // Filename
915  "main", // Logical db name
916  DB_BTREE, // Database type
917  DB_CREATE, // Flags
918  0);
919  if (ret > 0)
920  {
921  LogPrintf("Cannot create database file %s\n", filename);
922  return false;
923  }
924  CWallet dummyWallet;
925  CWalletScanState wss;
926 
927  DbTxn* ptxn = dbenv.TxnBegin();
928  BOOST_FOREACH(CDBEnv::KeyValPair& row, salvagedData)
929  {
930  if (fOnlyKeys)
931  {
932  CDataStream ssKey(row.first, SER_DISK, CLIENT_VERSION);
933  CDataStream ssValue(row.second, SER_DISK, CLIENT_VERSION);
934  std::string strType, strErr;
935  bool fReadOK;
936  {
937  // Required in LoadKeyMetadata():
938  LOCK(dummyWallet.cs_wallet);
939  fReadOK = ReadKeyValue(&dummyWallet, ssKey, ssValue,
940  wss, strType, strErr);
941  }
942  if (!IsKeyType(strType) && strType != "hdpubkey")
943  continue;
944  if (!fReadOK)
945  {
946  LogPrintf("WARNING: CWalletDB::Recover skipping %s: %s\n", strType, strErr);
947  continue;
948  }
949  }
950  Dbt datKey(&row.first[0], row.first.size());
951  Dbt datValue(&row.second[0], row.second.size());
952  int ret2 = pdbCopy->put(ptxn, &datKey, &datValue, DB_NOOVERWRITE);
953  if (ret2 > 0)
954  fSuccess = false;
955  }
956  ptxn->commit(0);
957  pdbCopy->close(0);
958 
959  return fSuccess;
960 }
