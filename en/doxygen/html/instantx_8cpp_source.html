97 bool CInstantSend::ProcessTxLockRequest(const CTxLockRequest& txLockRequest, CConnman& connman)
98 {
99  LOCK(cs_main);
100 #ifdef ENABLE_WALLET
101  LOCK(pwalletMain ? &pwalletMain->cs_wallet : NULL);
102 #endif
103  LOCK2(mempool.cs, cs_instantsend);
104 
105  uint256 txHash = txLockRequest.GetHash();
106 
107  // Check to see if we conflict with existing completed lock
108  for (const auto& txin : txLockRequest.tx->vin) {
109  std::map<COutPoint, uint256>::iterator it = mapLockedOutpoints.find(txin.prevout);
110  if (it != mapLockedOutpoints.end() && it->second != txLockRequest.GetHash()) {
111  // Conflicting with complete lock, proceed to see if we should cancel them both
112  LogPrintf("CInstantSend::ProcessTxLockRequest -- WARNING: Found conflicting completed Transaction Lock, txid=%s, completed lock txid=%s\n",
113  txLockRequest.GetHash().ToString(), it->second.ToString());
114  }
115  }
116 
117  // Check to see if there are votes for conflicting request,
118  // if so - do not fail, just warn user
119  for (const auto& txin : txLockRequest.tx->vin) {
120  std::map<COutPoint, std::set<uint256> >::iterator it = mapVotedOutpoints.find(txin.prevout);
121  if (it != mapVotedOutpoints.end()) {
122  for (const auto& hash : it->second) {
123  if (hash != txLockRequest.GetHash()) {
124  LogPrint("instantsend", "CInstantSend::ProcessTxLockRequest -- Double spend attempt! %s\n", txin.prevout.ToStringShort());
125  // do not fail here, let it go and see which one will get the votes to be locked
126  // NOTIFY ZMQ
127  CTransaction txCurrent = *txLockRequest.tx; // currently processed tx
128  auto itPrevious = mapTxLockCandidates.find(hash);
129  if (itPrevious != mapTxLockCandidates.end() && itPrevious->second.txLockRequest) {
130  CTransaction txPrevious = *itPrevious->second.txLockRequest.tx; // previously locked one
131  GetMainSignals().NotifyInstantSendDoubleSpendAttempt(txCurrent, txPrevious);
132  }
133  }
134  }
135  }
136  }
137 
138  if (!CreateTxLockCandidate(txLockRequest)) {
139  // smth is not right
140  LogPrintf("CInstantSend::ProcessTxLockRequest -- CreateTxLockCandidate failed, txid=%s\n", txHash.ToString());
141  return false;
142  }
143  LogPrintf("CInstantSend::ProcessTxLockRequest -- accepted, txid=%s\n", txHash.ToString());
144 
145  // Masternodes will sometimes propagate votes before the transaction is known to the client.
146  // If this just happened - process orphan votes, lock inputs, resolve conflicting locks,
147  // update transaction status forcing external script/zmq notifications.
148  ProcessOrphanTxLockVotes();
149  std::map<uint256, CTxLockCandidate>::iterator itLockCandidate = mapTxLockCandidates.find(txHash);
150  TryToFinalizeLockCandidate(itLockCandidate->second);
151 
152  return true;
153 }

155 bool CInstantSend::CreateTxLockCandidate(const CTxLockRequest& txLockRequest)
156 {
157  if (!txLockRequest.IsValid()) return false;
158 
159  LOCK(cs_instantsend);
160 
161  uint256 txHash = txLockRequest.GetHash();
162 
163  std::map<uint256, CTxLockCandidate>::iterator itLockCandidate = mapTxLockCandidates.find(txHash);
164  if (itLockCandidate == mapTxLockCandidates.end()) {
165  LogPrintf("CInstantSend::CreateTxLockCandidate -- new, txid=%s\n", txHash.ToString());
166 
167  CTxLockCandidate txLockCandidate(txLockRequest);
168  // all inputs should already be checked by txLockRequest.IsValid() above, just use them now
169  for (const auto& txin : txLockRequest.tx->vin) {
170  txLockCandidate.AddOutPointLock(txin.prevout);
171  }
172  mapTxLockCandidates.insert(std::make_pair(txHash, txLockCandidate));
173  } else if (!itLockCandidate->second.txLockRequest) {
174  // i.e. empty Transaction Lock Candidate was created earlier, let's update it with actual data
175  itLockCandidate->second.txLockRequest = txLockRequest;
176  if (itLockCandidate->second.IsTimedOut()) {
177  LogPrintf("CInstantSend::CreateTxLockCandidate -- timed out, txid=%s\n", txHash.ToString());
178  return false;
179  }
180  LogPrintf("CInstantSend::CreateTxLockCandidate -- update empty, txid=%s\n", txHash.ToString());
181 
182  // all inputs should already be checked by txLockRequest.IsValid() above, just use them now
183  for (const auto& txin : txLockRequest.tx->vin) {
184  itLockCandidate->second.AddOutPointLock(txin.prevout);
185  }
186  } else {
187  LogPrint("instantsend", "CInstantSend::CreateTxLockCandidate -- seen, txid=%s\n", txHash.ToString());
188  }
189 
190  return true;

224 void CInstantSend::Vote(CTxLockCandidate& txLockCandidate, CConnman& connman)
225 {
226  if (!fMasternodeMode) return;
227  if (!sporkManager.IsSporkActive(SPORK_2_INSTANTSEND_ENABLED)) return;
228 
229  AssertLockHeld(cs_main);
230  AssertLockHeld(cs_instantsend);
231 
232  uint256 txHash = txLockCandidate.GetHash();
233  // We should never vote on a Transaction Lock Request that was not (yet) accepted by the mempool
234  if (mapLockRequestAccepted.find(txHash) == mapLockRequestAccepted.end()) return;
235  // check if we need to vote on this candidate's outpoints,
236  // it's possible that we need to vote for several of them
237  for (auto& outpointLockPair : txLockCandidate.mapOutPointLocks) {
238  int nPrevoutHeight = GetUTXOHeight(outpointLockPair.first);
239  if (nPrevoutHeight == -1) {
240  LogPrint("instantsend", "CInstantSend::Vote -- Failed to find UTXO %s\n", outpointLockPair.first.ToStringShort());
241  return;
242  }
243 
244  int nLockInputHeight = nPrevoutHeight + Params().GetConsensus().nInstantSendConfirmationsRequired - 2;
245 
246  int nRank;
247  uint256 quorumModifierHash;
248  int nMinRequiredProtocol = std::max(MIN_INSTANTSEND_PROTO_VERSION, mnpayments.GetMinMasternodePaymentsProto());
249  if (!mnodeman.GetMasternodeRank(activeMasternodeInfo.outpoint, nRank, quorumModifierHash, nLockInputHeight, nMinRequiredProtocol)) {
250  LogPrint("instantsend", "CInstantSend::Vote -- Can't calculate rank for masternode %s\n", activeMasternodeInfo.outpoint.ToStringShort());
251  continue;
252  }
253  if (!deterministicMNManager->IsDeterministicMNsSporkActive()) {
254  // not used until spork15 activation
255  quorumModifierHash = uint256();
256  }
257 
258  int nSignaturesTotal = COutPointLock::SIGNATURES_TOTAL;
259  if (nRank > nSignaturesTotal) {
260  LogPrint("instantsend", "CInstantSend::Vote -- Masternode not in the top %d (%d)\n", nSignaturesTotal, nRank);
261  continue;
262  }
263 
264  LogPrint("instantsend", "CInstantSend::Vote -- In the top %d (%d)\n", nSignaturesTotal, nRank);
265 
266  std::map<COutPoint, std::set<uint256> >::iterator itVoted = mapVotedOutpoints.find(outpointLockPair.first);
267 
268  // Check to see if we already voted for this outpoint,
269  // refuse to vote twice or to include the same outpoint in another tx
270  bool fAlreadyVoted = false;
271  if (itVoted != mapVotedOutpoints.end()) {
272  for (const auto& hash : itVoted->second) {
273  std::map<uint256, CTxLockCandidate>::iterator it2 = mapTxLockCandidates.find(hash);
274  if (it2->second.HasMasternodeVoted(outpointLockPair.first, activeMasternodeInfo.outpoint)) {
275  // we already voted for this outpoint to be included either in the same tx or in a competing one,
276  // skip it anyway
277  fAlreadyVoted = true;
278  LogPrintf("CInstantSend::Vote -- WARNING: We already voted for this outpoint, skipping: txHash=%s, outpoint=%s\n",
279  txHash.ToString(), outpointLockPair.first.ToStringShort());
280  break;
281  }
282  }
283  }
284  if (fAlreadyVoted) {
285  continue; // skip to the next outpoint
286  }
287 
288  // we haven't voted for this outpoint yet, let's try to do this now
289  // Please note that activeMasternodeInfo.proTxHash is only valid after spork15 activation
290  CTxLockVote vote(txHash, outpointLockPair.first, activeMasternodeInfo.outpoint, quorumModifierHash, activeMasternodeInfo.proTxHash);
291 
292  if (!vote.Sign()) {
293  LogPrintf("CInstantSend::Vote -- Failed to sign consensus vote\n");
294  return;
295  }
296  if (!vote.CheckSignature()) {
297  LogPrintf("CInstantSend::Vote -- Signature invalid\n");
298  return;
299  }
300 
301  // vote constructed sucessfully, let's store and relay it
302  uint256 nVoteHash = vote.GetHash();
303  mapTxLockVotes.insert(std::make_pair(nVoteHash, vote));
304  if (outpointLockPair.second.AddVote(vote)) {
305  LogPrintf("CInstantSend::Vote -- Vote created successfully, relaying: txHash=%s, outpoint=%s, vote=%s\n",
306  txHash.ToString(), outpointLockPair.first.ToStringShort(), nVoteHash.ToString());
307 
308  if (itVoted == mapVotedOutpoints.end()) {
309  std::set<uint256> setHashes;
310  setHashes.insert(txHash);
311  mapVotedOutpoints.insert(std::make_pair(outpointLockPair.first, setHashes));
312  } else {
313  mapVotedOutpoints[outpointLockPair.first].insert(txHash);
314  if (mapVotedOutpoints[outpointLockPair.first].size() > 1) {
315  // it's ok to continue, just warn user
316  LogPrintf("CInstantSend::Vote -- WARNING: Vote conflicts with some existing votes: txHash=%s, outpoint=%s, vote=%s\n",
317  txHash.ToString(), outpointLockPair.first.ToStringShort(), nVoteHash.ToString());
318  }
319  }
320 
321  vote.Relay(connman);
322  }
323  }
324 }

326 bool CInstantSend::ProcessNewTxLockVote(CNode* pfrom, const CTxLockVote& vote, CConnman& connman)
327 {
328  uint256 txHash = vote.GetTxHash();
329  uint256 nVoteHash = vote.GetHash();
330 
331  if (!vote.IsValid(pfrom, connman)) {
332  // could be because of missing MN
333  LogPrint("instantsend", "CInstantSend::%s -- Vote is invalid, txid=%s\n", __func__, txHash.ToString());
334  return false;
335  }
336 
337  // relay valid vote asap
338  vote.Relay(connman);
339 
340  LOCK(cs_main);
341 #ifdef ENABLE_WALLET
342  LOCK(pwalletMain ? &pwalletMain->cs_wallet : NULL);
343 #endif
344  LOCK2(mempool.cs, cs_instantsend);
345 
346  // Masternodes will sometimes propagate votes before the transaction is known to the client,
347  // will actually process only after the lock request itself has arrived
348 
349  std::map<uint256, CTxLockCandidate>::iterator it = mapTxLockCandidates.find(txHash);
350  if (it == mapTxLockCandidates.end() || !it->second.txLockRequest) {
351  // no or empty tx lock candidate
352  if (it == mapTxLockCandidates.end()) {
353  // start timeout countdown after the very first vote
354  CreateEmptyTxLockCandidate(txHash);
355  }
356  bool fInserted = mapTxLockVotesOrphan.emplace(nVoteHash, vote).second;
357  LogPrint("instantsend", "CInstantSend::%s -- Orphan vote: txid=%s masternode=%s %s\n",
358  __func__, txHash.ToString(), vote.GetMasternodeOutpoint().ToStringShort(), fInserted ? "new" : "seen");
359 
360  // This tracks those messages and allows only the same rate as of the rest of the network
361  // TODO: make sure this works good enough for multi-quorum
362 
363  int nMasternodeOrphanExpireTime = GetTime() + 60*10; // keep time data for 10 minutes
364  auto itMnOV = mapMasternodeOrphanVotes.find(vote.GetMasternodeOutpoint());
365  if (itMnOV == mapMasternodeOrphanVotes.end()) {
366  mapMasternodeOrphanVotes.emplace(vote.GetMasternodeOutpoint(), nMasternodeOrphanExpireTime);
367  } else {
368  if (itMnOV->second > GetTime() && itMnOV->second > GetAverageMasternodeOrphanVoteTime()) {
369  LogPrint("instantsend", "CInstantSend::%s -- masternode is spamming orphan Transaction Lock Votes: txid=%s masternode=%s\n",
370  __func__, txHash.ToString(), vote.GetMasternodeOutpoint().ToStringShort());
371  // Misbehaving(pfrom->id, 1);
372  return false;
373  }
374  // not spamming, refresh
375  itMnOV->second = nMasternodeOrphanExpireTime;
376  }
377 
378  return true;
379  }
380 
381  // We have a valid (non-empty) tx lock candidate
382  CTxLockCandidate& txLockCandidate = it->second;
383 
384  if (txLockCandidate.IsTimedOut()) {
385  LogPrint("instantsend", "CInstantSend::%s -- too late, Transaction Lock timed out, txid=%s\n", __func__, txHash.ToString());
386  return false;
387  }
388 
389  LogPrint("instantsend", "CInstantSend::%s -- Transaction Lock Vote, txid=%s\n", __func__, txHash.ToString());
390 
391  UpdateVotedOutpoints(vote, txLockCandidate);
392 
393  if (!txLockCandidate.AddVote(vote)) {
394  // this should never happen
395  return false;
396  }
397 
398  int nSignatures = txLockCandidate.CountVotes();
399  int nSignaturesMax = txLockCandidate.txLockRequest.GetMaxSignatures();
400  LogPrint("instantsend", "CInstantSend::%s -- Transaction Lock signatures count: %d/%d, vote hash=%s\n", __func__,
401  nSignatures, nSignaturesMax, nVoteHash.ToString());
402 
403  TryToFinalizeLockCandidate(txLockCandidate);
404 
405  return true;
406 }

449 void CInstantSend::UpdateVotedOutpoints(const CTxLockVote& vote, CTxLockCandidate& txLockCandidate)
450 {
451  AssertLockHeld(cs_instantsend);
452 
453  uint256 txHash = vote.GetTxHash();
454 
455  std::map<COutPoint, std::set<uint256> >::iterator it1 = mapVotedOutpoints.find(vote.GetOutpoint());
456  if (it1 != mapVotedOutpoints.end()) {
457  for (const auto& hash : it1->second) {
458  if (hash != txHash) {
459  // same outpoint was already voted to be locked by another tx lock request,
460  // let's see if it was the same masternode who voted on this outpoint
461  // for another tx lock request
462  std::map<uint256, CTxLockCandidate>::iterator it2 = mapTxLockCandidates.find(hash);
463  if (it2 !=mapTxLockCandidates.end() && it2->second.HasMasternodeVoted(vote.GetOutpoint(), vote.GetMasternodeOutpoint())) {
464  // yes, it was the same masternode
465  LogPrintf("CInstantSend::%s -- masternode sent conflicting votes! %s\n", __func__, vote.GetMasternodeOutpoint().ToStringShort());
466  // mark both Lock Candidates as attacked, none of them should complete,
467  // or at least the new (current) one shouldn't even
468  // if the second one was already completed earlier
469  txLockCandidate.MarkOutpointAsAttacked(vote.GetOutpoint());
470  it2->second.MarkOutpointAsAttacked(vote.GetOutpoint());
471  // apply maximum PoSe ban score to this masternode i.e. PoSe-ban it instantly
472  mnodeman.PoSeBan(vote.GetMasternodeOutpoint());
473  // NOTE: This vote must be relayed further to let all other nodes know about such
474  // misbehaviour of this masternode. This way they should also be able to construct
475  // conflicting lock and PoSe-ban this masternode.
476  }
477  }
478  }
479  // store all votes, regardless of them being sent by malicious masternode or not
480  it1->second.insert(txHash);
481  } else {
482  mapVotedOutpoints.emplace(vote.GetOutpoint(), std::set<uint256>({txHash}));
483  }
484 }

519 void CInstantSend::UpdateLockedTransaction(const CTxLockCandidate& txLockCandidate)
520 {
521  // cs_main, cs_wallet and cs_instantsend should be already locked
522  AssertLockHeld(cs_main);
523 #ifdef ENABLE_WALLET
524  if (pwalletMain) {
525  AssertLockHeld(pwalletMain->cs_wallet);
526  }
527 #endif
528  AssertLockHeld(cs_instantsend);
529 
530  uint256 txHash = txLockCandidate.GetHash();
531 
532  if (!IsLockedInstantSendTransaction(txHash)) return; // not a locked tx, do not update/notify
533 
534 #ifdef ENABLE_WALLET
535  if (pwalletMain && pwalletMain->UpdatedTransaction(txHash)) {
536  // bumping this to update UI
537  nCompleteTXLocks++;
538  // notify an external script once threshold is reached

577 bool CInstantSend::ResolveConflicts(const CTxLockCandidate& txLockCandidate)
578 {
579  AssertLockHeld(cs_main);
580  AssertLockHeld(cs_instantsend);
581 
582  uint256 txHash = txLockCandidate.GetHash();
583 
584  // make sure the lock is ready
585  if (!txLockCandidate.IsAllOutPointsReady()) return false;
586 
587  AssertLockHeld(mempool.cs); // protect mempool.mapNextTx
588 
589  for (const auto& txin : txLockCandidate.txLockRequest.tx->vin) {
590  uint256 hashConflicting;
591  if (GetLockedOutPointTxHash(txin.prevout, hashConflicting) && txHash != hashConflicting) {
592  // completed lock which conflicts with another completed one?
593  // this means that majority of MNs in the quorum for this specific tx input are malicious!
594  std::map<uint256, CTxLockCandidate>::iterator itLockCandidate = mapTxLockCandidates.find(txHash);
595  std::map<uint256, CTxLockCandidate>::iterator itLockCandidateConflicting = mapTxLockCandidates.find(hashConflicting);
596  if (itLockCandidate == mapTxLockCandidates.end() || itLockCandidateConflicting == mapTxLockCandidates.end()) {
597  // safety check, should never really happen
598  LogPrintf("CInstantSend::ResolveConflicts -- ERROR: Found conflicting completed Transaction Lock, but one of txLockCandidate-s is missing, txid=%s, conflicting txid=%s\n",
599  txHash.ToString(), hashConflicting.ToString());
600  return false;
601  }
602  LogPrintf("CInstantSend::ResolveConflicts -- WARNING: Found conflicting completed Transaction Lock, dropping both, txid=%s, conflicting txid=%s\n",
603  txHash.ToString(), hashConflicting.ToString());
604  CTxLockRequest txLockRequest = itLockCandidate->second.txLockRequest;
605  CTxLockRequest txLockRequestConflicting = itLockCandidateConflicting->second.txLockRequest;
606  itLockCandidate->second.SetConfirmedHeight(0); // expired
607  itLockCandidateConflicting->second.SetConfirmedHeight(0); // expired
608  CheckAndRemove(); // clean up
609  // AlreadyHave should still return "true" for both of them
610  mapLockRequestRejected.insert(std::make_pair(txHash, txLockRequest));
611  mapLockRequestRejected.insert(std::make_pair(hashConflicting, txLockRequestConflicting));
612 
613  // TODO: clean up mapLockRequestRejected later somehow
614  // (not a big issue since we already PoSe ban malicious masternodes
615  // and they won't be able to spam)
616  // TODO: ban all malicious masternodes permanently, do not accept anything from them, ever
617 
618  // TODO: notify zmq+script about this double-spend attempt
619  // and let merchant cancel/hold the order if it's not too late...
620 
621  // can't do anything else, fallback to regular txes
622  return false;
623  } else if (mempool.mapNextTx.count(txin.prevout)) {
624  // check if it's in mempool
625  hashConflicting = mempool.mapNextTx.find(txin.prevout)->second->GetHash();
626  if (txHash == hashConflicting) continue; // matches current, not a conflict, skip to next txin
627  // conflicts with tx in mempool
628  LogPrintf("CInstantSend::ResolveConflicts -- ERROR: Failed to complete Transaction Lock, conflicts with mempool, txid=%s\n", txHash.ToString());
629  return false;
630  }
631  } // FOREACH
632  // No conflicts were found so far, check to see if it was already included in block
633  CTransactionRef txTmp;
634  uint256 hashBlock;
635  if (GetTransaction(txHash, txTmp, Params().GetConsensus(), hashBlock, true) && hashBlock != uint256()) {
636  LogPrint("instantsend", "CInstantSend::ResolveConflicts -- Done, %s is included in block %s\n", txHash.ToString(), hashBlock.ToString());
637  return true;
638  }
639  // Not in block yet, make sure all its inputs are still unspent
640  for (const auto& txin : txLockCandidate.txLockRequest.tx->vin) {
641  Coin coin;
642  if (!GetUTXOCoin(txin.prevout, coin)) {
643  // Not in UTXO anymore? A conflicting tx was mined while we were waiting for votes.
644  LogPrintf("CInstantSend::ResolveConflicts -- ERROR: Failed to find UTXO %s, can't complete Transaction Lock\n", txin.prevout.ToStringShort());
645  return false;
646  }
647  }
648  LogPrint("instantsend", "CInstantSend::ResolveConflicts -- Done, txid=%s\n", txHash.ToString());
649 
650  return true;
651 }

667 void CInstantSend::CheckAndRemove()
668 {
669  if (!masternodeSync.IsMasternodeListSynced()) return;
670 
671  LOCK(cs_instantsend);
672 
673  std::map<uint256, CTxLockCandidate>::iterator itLockCandidate = mapTxLockCandidates.begin();
674 
675  // remove expired candidates
676  while (itLockCandidate != mapTxLockCandidates.end()) {
677  CTxLockCandidate &txLockCandidate = itLockCandidate->second;
678  uint256 txHash = txLockCandidate.GetHash();
679  if (txLockCandidate.IsExpired(nCachedBlockHeight)) {
680  LogPrintf("CInstantSend::CheckAndRemove -- Removing expired Transaction Lock Candidate: txid=%s\n", txHash.ToString());
681 
682  for (const auto& pair : txLockCandidate.mapOutPointLocks) {
683  mapLockedOutpoints.erase(pair.first);
684  mapVotedOutpoints.erase(pair.first);
685  }
686  mapLockRequestAccepted.erase(txHash);
687  mapLockRequestRejected.erase(txHash);
688  mapTxLockCandidates.erase(itLockCandidate++);
689  } else {
690  ++itLockCandidate;
691  }
692  }
693 
694  // remove expired votes
695  std::map<uint256, CTxLockVote>::iterator itVote = mapTxLockVotes.begin();
696  while (itVote != mapTxLockVotes.end()) {
697  if (itVote->second.IsExpired(nCachedBlockHeight)) {
698  LogPrint("instantsend", "CInstantSend::CheckAndRemove -- Removing expired vote: txid=%s masternode=%s\n",
699  itVote->second.GetTxHash().ToString(), itVote->second.GetMasternodeOutpoint().ToStringShort());
700  mapTxLockVotes.erase(itVote++);
701  } else {
702  ++itVote;
703  }
704  }
705 
706  // remove timed out orphan votes
707  std::map<uint256, CTxLockVote>::iterator itOrphanVote = mapTxLockVotesOrphan.begin();
708  while (itOrphanVote != mapTxLockVotesOrphan.end()) {
709  if (itOrphanVote->second.IsTimedOut()) {
710  LogPrint("instantsend", "CInstantSend::CheckAndRemove -- Removing timed out orphan vote: txid=%s masternode=%s\n",
711  itOrphanVote->second.GetTxHash().ToString(), itOrphanVote->second.GetMasternodeOutpoint().ToStringShort());
712  mapTxLockVotes.erase(itOrphanVote->first);
713  mapTxLockVotesOrphan.erase(itOrphanVote++);
714  } else {
715  ++itOrphanVote;
716  }
717  }
718 
719  // remove invalid votes and votes for failed lock attempts
720  itVote = mapTxLockVotes.begin();
721  while (itVote != mapTxLockVotes.end()) {
722  if (itVote->second.IsFailed()) {
723  LogPrint("instantsend", "CInstantSend::CheckAndRemove -- Removing vote for failed lock attempt: txid=%s masternode=%s\n",
724  itVote->second.GetTxHash().ToString(), itVote->second.GetMasternodeOutpoint().ToStringShort());
725  mapTxLockVotes.erase(itVote++);
726  } else {
727  ++itVote;
728  }
729  }
730 
731  // remove timed out masternode orphan votes (DOS protection)
732  std::map<COutPoint, int64_t>::iterator itMasternodeOrphan = mapMasternodeOrphanVotes.begin();
733  while (itMasternodeOrphan != mapMasternodeOrphanVotes.end()) {
734  if (itMasternodeOrphan->second < GetTime()) {
735  LogPrint("instantsend", "CInstantSend::CheckAndRemove -- Removing timed out orphan masternode vote: masternode=%s\n",
736  itMasternodeOrphan->first.ToStringShort());
737  mapMasternodeOrphanVotes.erase(itMasternodeOrphan++);
738  } else {
739  ++itMasternodeOrphan;
740  }
741  }
742  LogPrintf("CInstantSend::CheckAndRemove -- %s\n", ToString());
743 }

1030 bool CTxLockVote::IsValid(CNode* pnode, CConnman& connman) const

1033  LogPrint("instantsend", "CTxLockVote::IsValid -- Unknown masternode %s\n", outpointMasternode.ToStringShort());
1034  mnodeman.AskForMN(pnode, outpointMasternode, connman);
1035  return false;
1036  }
1037 
1038  // Verify that masternodeProTxHash belongs to the same MN referred by the collateral
1039  // Only v13 nodes will send us locks with this field set, and only after spork15 activation
1040  if (!masternodeProTxHash.IsNull()) {
1041  masternode_info_t mnInfo;
1042  if (!mnodeman.GetMasternodeInfo(masternodeProTxHash, mnInfo) || mnInfo.outpoint != outpointMasternode) {
1043  LogPrint("instantsend", "CTxLockVote::IsValid -- invalid masternodeProTxHash %s\n", masternodeProTxHash.ToString());
1044  return false;
1045  }
1046  } else if (deterministicMNManager->IsDeterministicMNsSporkActive()) {
1047  LogPrint("instantsend", "CTxLockVote::IsValid -- missing masternodeProTxHash while DIP3 is active\n");
1048  return false;
1049  }
1050 
1051  Coin coin;
1052  if (!GetUTXOCoin(outpoint, coin)) {
1053  LogPrint("instantsend", "CTxLockVote::IsValid -- Failed to find UTXO %s\n", outpoint.ToStringShort());
1054  return false;
1055  }
1056 
1057  int nLockInputHeight = coin.nHeight + Params().GetConsensus().nInstantSendConfirmationsRequired - 2;
1058 
1059  int nRank;
1060  uint256 expectedQuorumModifierHash;
1061  int nMinRequiredProtocol = std::max(MIN_INSTANTSEND_PROTO_VERSION, mnpayments.GetMinMasternodePaymentsProto());
1062  if (!mnodeman.GetMasternodeRank(outpointMasternode, nRank, expectedQuorumModifierHash, nLockInputHeight, nMinRequiredProtocol)) {
1063  //can be caused by past versions trying to vote with an invalid protocol
1064  LogPrint("instantsend", "CTxLockVote::IsValid -- Can't calculate rank for masternode %s\n", outpointMasternode.ToStringShort());
1065  return false;
1066  }
1067  if (!quorumModifierHash.IsNull()) {
1068  if (quorumModifierHash != expectedQuorumModifierHash) {
1069  LogPrint("instantsend", "CTxLockVote::IsValid -- invalid quorumModifierHash %s, expected %s\n", quorumModifierHash.ToString(), expectedQuorumModifierHash.ToString());
1070  return false;
1071  }
1072  } else if (deterministicMNManager->IsDeterministicMNsSporkActive()) {
1073  LogPrint("instantsend", "CTxLockVote::IsValid -- missing quorumModifierHash while DIP3 is active\n");
1074  return false;
1075  }
1076 
1077  LogPrint("instantsend", "CTxLockVote::IsValid -- Masternode %s, rank=%d\n", outpointMasternode.ToStringShort(), nRank);
1078 
1079  int nSignaturesTotal = COutPointLock::SIGNATURES_TOTAL;
1080  if (nRank > nSignaturesTotal) {
1081  LogPrint("instantsend", "CTxLockVote::IsValid -- Masternode %s is not in the top %d (%d), vote hash=%s\n",
1082  outpointMasternode.ToStringShort(), nSignaturesTotal, nRank, GetHash().ToString());
1083  return false;
1084  }
1085 
1086  if (!CheckSignature()) {
