104 // find 'best' local address for a particular peer
105 bool GetLocal(CService& addr, const CNetAddr *paddrPeer)
106 {
107  if (!fListen)
108  return false;
109 
110  int nBestScore = -1;
111  int nBestReachability = -1;
112  {
113  LOCK(cs_mapLocalHost);
114  for (std::map<CNetAddr, LocalServiceInfo>::iterator it = mapLocalHost.begin(); it != mapLocalHost.end(); it++)
115  {
116  int nScore = (*it).second.nScore;
117  int nReachability = (*it).first.GetReachabilityFrom(paddrPeer);
118  if (nReachability > nBestReachability || (nReachability == nBestReachability && nScore > nBestScore))
119  {
120  addr = CService((*it).first, (*it).second.nPort);
121  nBestReachability = nReachability;
122  nBestScore = nScore;
123  }
124  }
125  }
126  return nBestScore >= 0;
127 }

130 static std::vector<CAddress> convertSeed6(const std::vector<SeedSpec6> &vSeedsIn)
131 {
132  // It'll only connect to one or two seed nodes because once it connects,
133  // it'll get a pile of addresses with newer timestamps.
134  // Seed nodes are given a random 'last seen time' of between one and two
135  // weeks ago.
136  const int64_t nOneWeek = 7*24*60*60;
137  std::vector<CAddress> vSeedsOut;
138  vSeedsOut.reserve(vSeedsIn.size());
139  for (std::vector<SeedSpec6>::const_iterator i(vSeedsIn.begin()); i != vSeedsIn.end(); ++i)
140  {
141  struct in6_addr ip;
142  memcpy(&ip, i->addr, sizeof(ip));
143  CAddress addr(CService(ip, i->port), NODE_NETWORK);
144  addr.nTime = GetTime() - GetRand(nOneWeek) - nOneWeek;
145  vSeedsOut.push_back(addr);
146  }
147  return vSeedsOut;
148 }

348 CNode* CConnman::ConnectNode(CAddress addrConnect, const char *pszDest, bool fCountFailure)
349 {
350  if (pszDest == NULL) {
351  bool fAllowLocal = Params().AllowMultiplePorts() && addrConnect.GetPort() != GetListenPort();
352  if (!fAllowLocal && IsLocal(addrConnect)) {
353  return NULL;
354  }
355 
356  // Look for an existing connection
357  CNode* pnode = FindNode((CService)addrConnect);
358  if (pnode)
359  {
360  LogPrintf("Failed to open new connection, already connected\n");
361  return NULL;
362  }
363  }
364 
366  LogPrint("net", "trying connection %s lastseen=%.1fhrs\n",
367  pszDest ? pszDest : addrConnect.ToString(),
368  pszDest ? 0.0 : (double)(GetAdjustedTime() - addrConnect.nTime)/3600.0);
369 
370  // Connect
371  SOCKET hSocket;
372  bool proxyConnectionFailed = false;
373  if (pszDest ? ConnectSocketByName(addrConnect, hSocket, pszDest, Params().GetDefaultPort(), nConnectTimeout, &proxyConnectionFailed) :
374  ConnectSocket(addrConnect, hSocket, nConnectTimeout, &proxyConnectionFailed))
375  {
376  if (!IsSelectableSocket(hSocket)) {
377  LogPrintf("Cannot create connection: non-selectable socket created (fd >= FD_SETSIZE ?)\n");
378  CloseSocket(hSocket);
379  return NULL;
380  }
381 
382  if (pszDest && addrConnect.IsValid()) {
383  // It is possible that we already have a connection to the IP/port pszDest resolved to.
384  // In that case, drop the connection that was just created, and return the existing CNode instead.
385  // Also store the name we used to connect in that CNode, so that future FindNode() calls to that
386  // name catch this early.
387  LOCK(cs_vNodes);
388  CNode* pnode = FindNode((CService)addrConnect);
389  if (pnode)
390  {
391  pnode->MaybeSetAddrName(std::string(pszDest));
392  CloseSocket(hSocket);
393  LogPrintf("Failed to open new connection, already connected\n");
394  return NULL;
395  }
396  }
397 
398  addrman.Attempt(addrConnect, fCountFailure);
399 
400  // Add node
401  NodeId id = GetNewNodeId();
402  uint64_t nonce = GetDeterministicRandomizer(RANDOMIZER_ID_LOCALHOSTNONCE).Write(id).Finalize();
403  CNode* pnode = new CNode(id, nLocalServices, GetBestHeight(), hSocket, addrConnect, CalculateKeyedNetGroup(addrConnect), nonce, pszDest ? pszDest : "", false);
404 
405  pnode->nServicesExpected = ServiceFlags(addrConnect.nServices & nRelevantServices);
406  pnode->AddRef();
407 
408 
409  return pnode;
410  } else if (!proxyConnectionFailed) {
411  // If connecting to the node failed, and failure is not caused by a problem connecting to
412  // the proxy, mark this as an attempt.
413  addrman.Attempt(addrConnect, fCountFailure);
414  }
415 
416  return NULL;
417 }

944 bool CConnman::AttemptToEvictConnection()
945 {
946  std::vector<NodeEvictionCandidate> vEvictionCandidates;
947  {
948  LOCK(cs_vNodes);
949 
950  BOOST_FOREACH(CNode *node, vNodes) {
951  if (node->fWhitelisted)
952  continue;
953  if (!node->fInbound)
954  continue;
955  if (node->fDisconnect)
956  continue;
957  NodeEvictionCandidate candidate = {node->id, node->nTimeConnected, node->nMinPingUsecTime,
958  node->nLastBlockTime, node->nLastTXTime,
959  (node->nServices & nRelevantServices) == nRelevantServices,
960  node->fRelayTxes, node->pfilter != NULL, node->nKeyedNetGroup};
961  vEvictionCandidates.push_back(candidate);
962  }
963  }
964 
965  if (vEvictionCandidates.empty()) return false;
966 
967  // Protect connections with certain characteristics
968 
969  // Deterministically select 4 peers to protect by netgroup.
970  // An attacker cannot predict which netgroups will be protected
971  std::sort(vEvictionCandidates.begin(), vEvictionCandidates.end(), CompareNetGroupKeyed);
972  vEvictionCandidates.erase(vEvictionCandidates.end() - std::min(4, static_cast<int>(vEvictionCandidates.size())), vEvictionCandidates.end());
973 
974  if (vEvictionCandidates.empty()) return false;
975 
976  // Protect the 8 nodes with the lowest minimum ping time.
977  // An attacker cannot manipulate this metric without physically moving nodes closer to the target.
978  std::sort(vEvictionCandidates.begin(), vEvictionCandidates.end(), ReverseCompareNodeMinPingTime);
979  vEvictionCandidates.erase(vEvictionCandidates.end() - std::min(8, static_cast<int>(vEvictionCandidates.size())), vEvictionCandidates.end());
980 
981  if (vEvictionCandidates.empty()) return false;
982 
983  // Protect 4 nodes that most recently sent us transactions.
984  // An attacker cannot manipulate this metric without performing useful work.
985  std::sort(vEvictionCandidates.begin(), vEvictionCandidates.end(), CompareNodeTXTime);
986  vEvictionCandidates.erase(vEvictionCandidates.end() - std::min(4, static_cast<int>(vEvictionCandidates.size())), vEvictionCandidates.end());
987 
988  if (vEvictionCandidates.empty()) return false;
989 
990  // Protect 4 nodes that most recently sent us blocks.
991  // An attacker cannot manipulate this metric without performing useful work.
992  std::sort(vEvictionCandidates.begin(), vEvictionCandidates.end(), CompareNodeBlockTime);
993  vEvictionCandidates.erase(vEvictionCandidates.end() - std::min(4, static_cast<int>(vEvictionCandidates.size())), vEvictionCandidates.end());
994 
995  if (vEvictionCandidates.empty()) return false;
996 
997  // Protect the half of the remaining nodes which have been connected the longest.
998  // This replicates the non-eviction implicit behavior, and precludes attacks that start later.
999  std::sort(vEvictionCandidates.begin(), vEvictionCandidates.end(), ReverseCompareNodeTimeConnected);
1000  vEvictionCandidates.erase(vEvictionCandidates.end() - static_cast<int>(vEvictionCandidates.size() / 2), vEvictionCandidates.end());
1001 
1002  if (vEvictionCandidates.empty()) return false;
1003 
1004  // Identify the network group with the most connections and youngest member.
1005  // (vEvictionCandidates is already sorted by reverse connect time)
1006  uint64_t naMostConnections;
1007  unsigned int nMostConnections = 0;
1008  int64_t nMostConnectionsTime = 0;
1009  std::map<uint64_t, std::vector<NodeEvictionCandidate> > mapNetGroupNodes;
1010  BOOST_FOREACH(const NodeEvictionCandidate &node, vEvictionCandidates) {
1011  mapNetGroupNodes[node.nKeyedNetGroup].push_back(node);
1012  int64_t grouptime = mapNetGroupNodes[node.nKeyedNetGroup][0].nTimeConnected;
1013  size_t groupsize = mapNetGroupNodes[node.nKeyedNetGroup].size();
1014 
1015  if (groupsize > nMostConnections || (groupsize == nMostConnections && grouptime > nMostConnectionsTime)) {
1016  nMostConnections = groupsize;
1017  nMostConnectionsTime = grouptime;
1018  naMostConnections = node.nKeyedNetGroup;
1019  }
1020  }
1021 
1022  // Reduce to the network group with the most connections
1023  vEvictionCandidates = std::move(mapNetGroupNodes[naMostConnections]);
1024 
1025  // Disconnect from the network group with the most connections
1026  NodeId evicted = vEvictionCandidates.front().id;
1027  LOCK(cs_vNodes);
1028  for(std::vector<CNode*>::const_iterator it(vNodes.begin()); it != vNodes.end(); ++it) {
1029  if ((*it)->GetId() == evicted) {
1030  (*it)->fDisconnect = true;
1031  return true;
1032  }
1033  }
1034  return false;
1035 }

1336  if (!it->complete())
1337  break;
1338  nSizeAdded += it->vRecv.size() + CMessageHeader::HEADER_SIZE;
1339  }

1342  pnode->vProcessMsg.splice(pnode->vProcessMsg.end(), pnode->vRecvMsg, pnode->vRecvMsg.begin(), it);

1367  }

1373  //

1379  RecordBytesSent(nBytes);

1381  }

1384  // Inactivity checking
1385  //
1386  int64_t nTime = GetSystemTimeInSeconds();
1387  if (nTime - pnode->nTimeConnected > 60)
1388  {
1389  if (pnode->nLastRecv == 0 || pnode->nLastSend == 0)
1390  {

1395  {
1396  LogPrintf("socket sending timeout: %is\n", nTime - pnode->nLastSend);
1397  pnode->fDisconnect = true;
1398  }

1400  {

1680 {
1681  // Connect to specific addresses
1682  if (mapMultiArgs.count("-connect") && mapMultiArgs.at("-connect").size() > 0)
1683  {
1684  for (int64_t nLoop = 0;; nLoop++)
1685  {
1686  ProcessOneShot();

1688  {
1689  CAddress addr(CService(), NODE_NONE);
1690  OpenNetworkConnection(addr, false, NULL, strAddr.c_str());

1692  {
1693  if (!interruptNet.sleep_for(std::chrono::milliseconds(500)))
1694  return;
1695  }
1696  }
1697  if (!interruptNet.sleep_for(std::chrono::milliseconds(500)))
1698  return;
1699  }
1700  }
1701 
1702  // Initiate network connections
1703  int64_t nStart = GetTime();
1704 
1705  // Minimum time before next feeler connection (in microseconds).
1706  int64_t nNextFeeler = PoissonNextSend(nStart*1000*1000, FEELER_INTERVAL);
1707  while (!interruptNet)
1708  {
1709  ProcessOneShot();
1710 
1711  if (!interruptNet.sleep_for(std::chrono::milliseconds(500)))
1712  return;
1713 
1714  CSemaphoreGrant grant(*semOutbound);
1715  if (interruptNet)
1716  return;
1717 
1718  // Add seed nodes if DNS seeds are all down (an infrastructure attack?).
1719  if (addrman.size() == 0 && (GetTime() - nStart > 60)) {
1720  static bool done = false;
1721  if (!done) {
1722  LogPrintf("Adding fixed seed nodes as DNS doesn't seem to be available.\n");
1723  CNetAddr local;
1724  LookupHost("127.0.0.1", local, false);
1725  addrman.Add(convertSeed6(Params().FixedSeeds()), local);
1726  done = true;
1727  }
1728  }
1729 
1730  //
1731  // Choose an address to connect to based on most recently seen
1732  //
1733  CAddress addrConnect;
1734 
1735  // Only connect out to one peer per network group (/16 for IPv4).
1736  // Do this here so we don't have to critsect vNodes inside mapAddresses critsect.
1737  // This is only done for mainnet and testnet
1738  int nOutbound = 0;
1739  int nOutboundRelevant = 0;
1740  std::set<std::vector<unsigned char> > setConnected;
1741  if (!Params().AllowMultipleAddressesFromGroup()) {
1742  LOCK(cs_vNodes);
1743  BOOST_FOREACH(CNode* pnode, vNodes) {
1744  if (!pnode->fInbound && !pnode->fAddnode && !pnode->fMasternode) {
1745 
1746  // Count the peers that have all relevant services
1747  if (pnode->fSuccessfullyConnected && !pnode->fFeeler && ((pnode->nServices & nRelevantServices) == nRelevantServices)) {
1748  nOutboundRelevant++;
1749  }
1750  // Netgroups for inbound and addnode peers are not excluded because our goal here
1751  // is to not use multiple of our limited outbound slots on a single netgroup
1752  // but inbound and addnode peers do not use our outbound slots. Inbound peers
1753  // also have the added issue that they're attacker controlled and could be used
1754  // to prevent us from connecting to particular hosts if we used them here.
1755  setConnected.insert(pnode->addr.GetGroup());

1757  }
1758  }
1759  }

1762  //
1763  // Design goals:
1764  // * Increase the number of connectable addresses in the tried table.
1765  //
1766  // Method:
1767  // * Choose a random address from new and attempt to connect to it if we can connect
1768  // successfully it is added to tried.
1769  // * Start attempting feeler connections only after node finishes making outbound
1770  // connections.
1771  // * Only make a feeler connection once every few minutes.
1772  //
1773  bool fFeeler = false;
1774  if (nOutbound >= nMaxOutbound) {
1775  int64_t nTime = GetTimeMicros(); // The current time right now (in microseconds).
1776  if (nTime > nNextFeeler) {
1777  nNextFeeler = PoissonNextSend(nTime, FEELER_INTERVAL);
1778  fFeeler = true;
1779  } else {
1780  continue;
1781  }
1782  }
1783 
1784  int64_t nANow = GetAdjustedTime();
1785  int nTries = 0;
1786  while (!interruptNet)
1787  {
1788  CAddrInfo addr = addrman.Select(fFeeler);
1789 
1790  // if we selected an invalid address, restart
1791  if (!addr.IsValid() || setConnected.count(addr.GetGroup()))
1792  break;
1793 
1794  // if we selected a local address, restart (local addresses are allowed in regtest and devnet)
1795  bool fAllowLocal = Params().AllowMultiplePorts() && addrConnect.GetPort() != GetListenPort();
1796  if (!fAllowLocal && IsLocal(addrConnect))
1797  break;
1798 
1799  // If we didn't find an appropriate destination after trying 100 addresses fetched from addrman,
1800  // stop this loop, and let the outer loop run again (which sleeps, adds seed nodes, recalculates
1801  // already-connected network ranges, ...) before trying new addrman addresses.
1802  nTries++;
1803  if (nTries > 100)
1804  break;
1805 
1806  if (IsLimited(addr))
1807  continue;
1808 
1809  // only connect to full nodes
1810  if ((addr.nServices & REQUIRED_SERVICES) != REQUIRED_SERVICES)
1811  continue;
1812 
1813  // only consider very recently tried nodes after 30 failed attempts
1814  if (nANow - addr.nLastTry < 600 && nTries < 30)
1815  continue;
1816 
1817  // only consider nodes missing relevant services after 40 failed attempts and only if less than half the outbound are up.
1818  ServiceFlags nRequiredServices = nRelevantServices;
1819  if (nTries >= 40 && nOutbound < (nMaxOutbound >> 1)) {
1820  nRequiredServices = REQUIRED_SERVICES;
1821  }
1822 
1823  if ((addr.nServices & nRequiredServices) != nRequiredServices) {
1824  continue;
1825  }
1826 
1827  // do not allow non-default ports, unless after 50 invalid addresses selected already
1828  if (addr.GetPort() != Params().GetDefaultPort() && addr.GetPort() != GetListenPort() && nTries < 50)
1829  continue;
1830 
1831  addrConnect = addr;
1832 
1833  // regardless of the services assumed to be available, only require the minimum if half or more outbound have relevant services
1834  if (nOutboundRelevant >= (nMaxOutbound >> 1)) {
1835  addrConnect.nServices = REQUIRED_SERVICES;
1836  } else {
1837  addrConnect.nServices = nRequiredServices;
1838  }
1839  break;
1840  }
1841 
1842  if (addrConnect.IsValid()) {
1843 
1844  if (fFeeler) {
1845  // Add small amount of random noise before connection to avoid synchronization.
1846  int randsleep = GetRandInt(FEELER_SLEEP_WINDOW * 1000);
1847  if (!interruptNet.sleep_for(std::chrono::milliseconds(randsleep)))
1848  return;
1849  LogPrint("net", "Making feeler connection to %s\n", addrConnect.ToString());
1850  }
1851 
1852  OpenNetworkConnection(addrConnect, (int)setConnected.size() >= std::min(nMaxConnections - 1, 2), &grant, NULL, false, fFeeler);
1853  }
1854  }
