87  void SetBuf(const void* buf, size_t size)
88  {
89  if (size != SerSize) {
90  Reset();
91  return;
92  }
93 
94  if (std::all_of((const char*)buf, (const char*)buf + SerSize, [](char c) { return c == 0; })) {
95  Reset();
96  } else {
97  fValid = InternalSetBuf(buf);
98  if (!fValid) {
99  Reset();
100  }
101  }
102  UpdateHash();
103  }
104 
105  void Reset()
106  {
107  *((C*)this) = C();
108  }
109 
110  void GetBuf(void* buf, size_t size) const
111  {
112  assert(size == SerSize);
113 
114  if (!fValid) {
115  memset(buf, 0, SerSize);
116  } else {
117  bool ok = InternalGetBuf(buf);
118  assert(ok);
119  }
120  }

177  inline void CheckMalleable(void* buf, size_t size) const
178  {
179  char buf2[SerSize];
180  C tmp;
181  tmp.SetBuf(buf, SerSize);
182  tmp.GetBuf(buf2, SerSize);
183  if (memcmp(buf, buf2, SerSize)) {
184  // TODO not sure if this is actually possible with the BLS libs. I'm assuming here that somewhere deep inside
185  // these libs masking might happen, so that 2 different binary representations could result in the same object
186  // representation
187  throw std::ios_base::failure("malleable BLS object");
188  }
189  }
