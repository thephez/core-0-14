101  try {
102  return tinyformat::format(fmt, args...);

255 template <typename Callable> void TraceThread(const char* name, Callable func)
256 {
257  std::string s = strprintf("dash-%s", name);
258  RenameThread(s.c_str());
259  try
260  {
261  LogPrintf("%s thread start\n", name);
262  func();
263  LogPrintf("%s thread exit\n", name);
264  }
265  catch (const boost::thread_interrupted&)
266  {
267  LogPrintf("%s thread interrupt\n", name);
268  throw;
269  }
270  catch (const std::exception& e) {
271  PrintExceptionContinue(&e, name);
272  throw;
273  }
274  catch (...) {
275  PrintExceptionContinue(NULL, name);
276  throw;
277  }
278 }
