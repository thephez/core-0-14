21 void CPrivateSendServer::ProcessMessage(CNode* pfrom, const std::string& strCommand, CDataStream& vRecv, CConnman& connman)
22 {
23  if (!fMasternodeMode) return;
24  if (fLiteMode) return; // ignore all Dash related functionality
25  if (!masternodeSync.IsBlockchainSynced()) return;
26 
27  if (strCommand == NetMsgType::DSACCEPT) {
28  if (pfrom->nVersion < MIN_PRIVATESEND_PEER_PROTO_VERSION) {
29  LogPrint("privatesend", "DSACCEPT -- peer=%d using obsolete version %i\n", pfrom->id, pfrom->nVersion);
30  connman.PushMessage(pfrom, CNetMsgMaker(pfrom->GetSendVersion()).Make(NetMsgType::REJECT, strCommand, REJECT_OBSOLETE, strprintf("Version must be %d or greater", MIN_PRIVATESEND_PEER_PROTO_VERSION)));
31  PushStatus(pfrom, STATUS_REJECTED, ERR_VERSION, connman);
32  return;
33  }
34 
35  if (IsSessionReady()) {
36  // too many users in this session already, reject new ones
37  LogPrintf("DSACCEPT -- queue is already full!\n");
38  PushStatus(pfrom, STATUS_ACCEPTED, ERR_QUEUE_FULL, connman);
39  return;
40  }
41 
42  CPrivateSendAccept dsa;
43  vRecv >> dsa;
44 
45  LogPrint("privatesend", "DSACCEPT -- nDenom %d (%s) txCollateral %s", dsa.nDenom, CPrivateSend::GetDenominationsToString(dsa.nDenom), dsa.txCollateral.ToString());
46 
47  masternode_info_t mnInfo;
48  if (!mnodeman.GetMasternodeInfo(activeMasternodeInfo.outpoint, mnInfo)) {
49  PushStatus(pfrom, STATUS_REJECTED, ERR_MN_LIST, connman);
50  return;
51  }
52 
53  if (vecSessionCollaterals.size() == 0) {
54  {
55  TRY_LOCK(cs_vecqueue, lockRecv);
56  if (!lockRecv) return;
57 
58  for (const auto& q : vecPrivateSendQueue) {
59  if (q.masternodeOutpoint == activeMasternodeInfo.outpoint) {
60  // refuse to create another queue this often
61  LogPrint("privatesend", "DSACCEPT -- last dsq is still in queue, refuse to mix\n");
62  PushStatus(pfrom, STATUS_REJECTED, ERR_RECENT, connman);
63  return;
64  }
65  }
66  }
67 
68  if (mnInfo.nLastDsq != 0 && mnInfo.nLastDsq + mnodeman.CountMasternodes() / 5 > mnodeman.nDsqCount) {
69  LogPrintf("DSACCEPT -- last dsq too recent, must wait: addr=%s\n", pfrom->addr.ToString());
70  PushStatus(pfrom, STATUS_REJECTED, ERR_RECENT, connman);
71  return;
72  }
73  }
74 
75  PoolMessage nMessageID = MSG_NOERR;
76 
77  bool fResult = nSessionID == 0 ? CreateNewSession(dsa, nMessageID, connman)
78  : AddUserToExistingSession(dsa, nMessageID);
79  if (fResult) {
80  LogPrintf("DSACCEPT -- is compatible, please submit!\n");
81  PushStatus(pfrom, STATUS_ACCEPTED, nMessageID, connman);
82  return;
83  } else {
84  LogPrintf("DSACCEPT -- not compatible with existing transactions!\n");
85  PushStatus(pfrom, STATUS_REJECTED, nMessageID, connman);
86  return;
87  }
88 
89  } else if (strCommand == NetMsgType::DSQUEUE) {
90  TRY_LOCK(cs_vecqueue, lockRecv);
91  if (!lockRecv) return;
92 
93  if (pfrom->nVersion < MIN_PRIVATESEND_PEER_PROTO_VERSION) {
94  LogPrint("privatesend", "DSQUEUE -- peer=%d using obsolete version %i\n", pfrom->id, pfrom->nVersion);
95  connman.PushMessage(pfrom, CNetMsgMaker(pfrom->GetSendVersion()).Make(NetMsgType::REJECT, strCommand, REJECT_OBSOLETE, strprintf("Version must be %d or greater", MIN_PRIVATESEND_PEER_PROTO_VERSION)));
96  return;
97  }
98 
99  CPrivateSendQueue dsq;
100  vRecv >> dsq;
101 
102  // process every dsq only once
103  for (const auto& q : vecPrivateSendQueue) {
104  if (q == dsq) {
105  // LogPrint("privatesend", "DSQUEUE -- %s seen\n", dsq.ToString());
106  return;
107  }
108  }
109 
110  LogPrint("privatesend", "DSQUEUE -- %s new\n", dsq.ToString());
111 
112  if (dsq.IsExpired()) return;
113 
114  masternode_info_t mnInfo;
115  if (!mnodeman.GetMasternodeInfo(dsq.masternodeOutpoint, mnInfo)) return;
116 
117  if (!dsq.CheckSignature(mnInfo.legacyKeyIDOperator, mnInfo.blsPubKeyOperator)) {
118  // we probably have outdated info
119  mnodeman.AskForMN(pfrom, dsq.masternodeOutpoint, connman);
120  return;
121  }
122 
123  if (!dsq.fReady) {
124  for (const auto& q : vecPrivateSendQueue) {
125  if (q.masternodeOutpoint == dsq.masternodeOutpoint) {
126  // no way same mn can send another "not yet ready" dsq this soon
127  LogPrint("privatesend", "DSQUEUE -- Masternode %s is sending WAY too many dsq messages\n", mnInfo.addr.ToString());
128  return;
129  }
130  }
131 
132  int nThreshold = mnInfo.nLastDsq + mnodeman.CountMasternodes() / 5;
133  LogPrint("privatesend", "DSQUEUE -- nLastDsq: %d threshold: %d nDsqCount: %d\n", mnInfo.nLastDsq, nThreshold, mnodeman.nDsqCount);
134  //don't allow a few nodes to dominate the queuing process
135  if (mnInfo.nLastDsq != 0 && nThreshold > mnodeman.nDsqCount) {
136  LogPrint("privatesend", "DSQUEUE -- Masternode %s is sending too many dsq messages\n", mnInfo.addr.ToString());
137  return;
138  }
139  mnodeman.AllowMixing(dsq.masternodeOutpoint);
140 
141  LogPrint("privatesend", "DSQUEUE -- new PrivateSend queue (%s) from masternode %s\n", dsq.ToString(), mnInfo.addr.ToString());
142  vecPrivateSendQueue.push_back(dsq);
143  dsq.Relay(connman);
144  }
145 
146  } else if (strCommand == NetMsgType::DSVIN) {
147  if (pfrom->nVersion < MIN_PRIVATESEND_PEER_PROTO_VERSION) {
148  LogPrint("privatesend", "DSVIN -- peer=%d using obsolete version %i\n", pfrom->id, pfrom->nVersion);
149  connman.PushMessage(pfrom, CNetMsgMaker(pfrom->GetSendVersion()).Make(NetMsgType::REJECT, strCommand, REJECT_OBSOLETE, strprintf("Version must be %d or greater", MIN_PRIVATESEND_PEER_PROTO_VERSION)));
150  PushStatus(pfrom, STATUS_REJECTED, ERR_VERSION, connman);
151  return;
152  }
153 
154  //do we have enough users in the current session?
155  if (!IsSessionReady()) {
156  LogPrintf("DSVIN -- session not complete!\n");
157  PushStatus(pfrom, STATUS_REJECTED, ERR_SESSION, connman);
158  return;
159  }
160 
161  CPrivateSendEntry entry;
162  vRecv >> entry;
163 
164  LogPrint("privatesend", "DSVIN -- txCollateral %s", entry.txCollateral->ToString());
165 
166  if (entry.vecTxDSIn.size() > PRIVATESEND_ENTRY_MAX_SIZE) {
167  LogPrintf("DSVIN -- ERROR: too many inputs! %d/%d\n", entry.vecTxDSIn.size(), PRIVATESEND_ENTRY_MAX_SIZE);
168  PushStatus(pfrom, STATUS_REJECTED, ERR_MAXIMUM, connman);
169  return;
170  }
171 
172  if (entry.vecTxOut.size() > PRIVATESEND_ENTRY_MAX_SIZE) {
173  LogPrintf("DSVIN -- ERROR: too many outputs! %d/%d\n", entry.vecTxOut.size(), PRIVATESEND_ENTRY_MAX_SIZE);
174  PushStatus(pfrom, STATUS_REJECTED, ERR_MAXIMUM, connman);
175  return;
176  }
177 
178  //do we have the same denominations as the current session?
179  if (!IsOutputsCompatibleWithSessionDenom(entry.vecTxOut)) {
180  LogPrintf("DSVIN -- not compatible with existing transactions!\n");
181  PushStatus(pfrom, STATUS_REJECTED, ERR_EXISTING_TX, connman);
182  return;
183  }
184 
185  //check it like a transaction
186  {
187  CAmount nValueIn = 0;
188  CAmount nValueOut = 0;
189 
190  CMutableTransaction tx;
191 
192  for (const auto& txout : entry.vecTxOut) {
193  nValueOut += txout.nValue;
194  tx.vout.push_back(txout);
195 
196  if (txout.scriptPubKey.size() != 25) {
197  LogPrintf("DSVIN -- non-standard pubkey detected! scriptPubKey=%s\n", ScriptToAsmStr(txout.scriptPubKey));
198  PushStatus(pfrom, STATUS_REJECTED, ERR_NON_STANDARD_PUBKEY, connman);
199  return;
200  }
201  if (!txout.scriptPubKey.IsPayToPublicKeyHash()) {
202  LogPrintf("DSVIN -- invalid script! scriptPubKey=%s\n", ScriptToAsmStr(txout.scriptPubKey));
203  PushStatus(pfrom, STATUS_REJECTED, ERR_INVALID_SCRIPT, connman);
204  return;
205  }
206  }
207 
208  for (const auto& txin : entry.vecTxDSIn) {
209  tx.vin.push_back(txin);
210 
211  LogPrint("privatesend", "DSVIN -- txin=%s\n", txin.ToString());
212 
213  Coin coin;
214  if (GetUTXOCoin(txin.prevout, coin)) {
215  nValueIn += coin.out.nValue;
216  } else {
217  LogPrintf("DSVIN -- missing input! txin=%s\n", txin.ToString());
218  PushStatus(pfrom, STATUS_REJECTED, ERR_MISSING_TX, connman);
219  return;
220  }
221  }
222 
223  // There should be no fee in mixing tx
224  CAmount nFee = nValueIn - nValueOut;
225  if (nFee != 0) {
226  LogPrintf("DSVIN -- there should be no fee in mixing tx! fees: %lld, tx=%s", nFee, tx.ToString());
227  PushStatus(pfrom, STATUS_REJECTED, ERR_FEES, connman);
228  return;
229  }
230  }
231 
232  PoolMessage nMessageID = MSG_NOERR;
233 
234  entry.addr = pfrom->addr;
235  if (AddEntry(entry, nMessageID)) {
236  PushStatus(pfrom, STATUS_ACCEPTED, nMessageID, connman);
237  CheckPool(connman);
238  RelayStatus(STATUS_ACCEPTED, connman);
239  } else {
240  PushStatus(pfrom, STATUS_REJECTED, nMessageID, connman);
241  SetNull();
242  }
243 
244  } else if (strCommand == NetMsgType::DSSIGNFINALTX) {
245  if (pfrom->nVersion < MIN_PRIVATESEND_PEER_PROTO_VERSION) {
246  LogPrint("privatesend", "DSSIGNFINALTX -- peer=%d using obsolete version %i\n", pfrom->id, pfrom->nVersion);
247  connman.PushMessage(pfrom, CNetMsgMaker(pfrom->GetSendVersion()).Make(NetMsgType::REJECT, strCommand, REJECT_OBSOLETE, strprintf("Version must be %d or greater", MIN_PRIVATESEND_PEER_PROTO_VERSION)));
248  return;
249  }
250 
251  std::vector<CTxIn> vecTxIn;
252  vRecv >> vecTxIn;
253 
254  LogPrint("privatesend", "DSSIGNFINALTX -- vecTxIn.size() %s\n", vecTxIn.size());
255 
256  int nTxInIndex = 0;
257  int nTxInsCount = (int)vecTxIn.size();
258 
259  for (const auto& txin : vecTxIn) {
260  nTxInIndex++;
261  if (!AddScriptSig(txin)) {
262  LogPrint("privatesend", "DSSIGNFINALTX -- AddScriptSig() failed at %d/%d, session: %d\n", nTxInIndex, nTxInsCount, nSessionID);
263  RelayStatus(STATUS_REJECTED, connman);
264  return;
265  }
266  LogPrint("privatesend", "DSSIGNFINALTX -- AddScriptSig() %d/%d success\n", nTxInIndex, nTxInsCount);
267  }
268  // all is good
269  CheckPool(connman);
270  }

286 {

302  return;

306 void CPrivateSendServer::CreateFinalTransaction(CConnman& connman)
307 {
308  LogPrint("privatesend", "CPrivateSendServer::CreateFinalTransaction -- FINALIZE TRANSACTIONS\n");
309 
310  CMutableTransaction txNew;
311 
312  // make our new transaction
313  for (int i = 0; i < GetEntriesCount(); i++) {
314  for (const auto& txout : vecEntries[i].vecTxOut)
315  txNew.vout.push_back(txout);
316 
317  for (const auto& txdsin : vecEntries[i].vecTxDSIn)
318  txNew.vin.push_back(txdsin);
319  }
320 
321  sort(txNew.vin.begin(), txNew.vin.end(), CompareInputBIP69());
322  sort(txNew.vout.begin(), txNew.vout.end(), CompareOutputBIP69());
323 
324  finalMutableTransaction = txNew;
325  LogPrint("privatesend", "CPrivateSendServer::CreateFinalTransaction -- finalMutableTransaction=%s", txNew.ToString());
326 
327  // request signatures from clients
328  RelayFinalTransaction(finalMutableTransaction, connman);
329  SetState(POOL_STATE_SIGNING);
330 }

332 void CPrivateSendServer::CommitFinalTransaction(CConnman& connman)
333 {
334  if (!fMasternodeMode) return; // check and relay final tx only on masternode
335 
336  CTransactionRef finalTransaction = MakeTransactionRef(finalMutableTransaction);
337  uint256 hashTx = finalTransaction->GetHash();
338 
339  LogPrint("privatesend", "CPrivateSendServer::CommitFinalTransaction -- finalTransaction=%s", finalTransaction->ToString());
340 
341  {
342  // See if the transaction is valid
343  TRY_LOCK(cs_main, lockMain);
344  CValidationState validationState;
345  mempool.PrioritiseTransaction(hashTx, hashTx.ToString(), 1000, 0.1 * COIN);
346  if (!lockMain || !AcceptToMemoryPool(mempool, validationState, finalTransaction, false, NULL, false, maxTxFee, true)) {
347  LogPrintf("CPrivateSendServer::CommitFinalTransaction -- AcceptToMemoryPool() error: Transaction not valid\n");
348  SetNull();
349  // not much we can do in this case, just notify clients
350  RelayCompletedTransaction(ERR_INVALID_TX, connman);
351  return;
352  }
353  }
354 
355  LogPrintf("CPrivateSendServer::CommitFinalTransaction -- CREATING DSTX\n");
356 
357  // create and sign masternode dstx transaction
358  if (!CPrivateSend::GetDSTX(hashTx)) {
359  CPrivateSendBroadcastTx dstxNew(finalTransaction, activeMasternodeInfo.outpoint, GetAdjustedTime());
360  dstxNew.Sign();
361  CPrivateSend::AddDSTX(dstxNew);
362  }
363 
364  LogPrintf("CPrivateSendServer::CommitFinalTransaction -- TRANSMITTING DSTX\n");
365 
366  CInv inv(MSG_DSTX, hashTx);
367  connman.RelayInv(inv);
368 
369  // Tell the clients it was successful
370  RelayCompletedTransaction(MSG_SUCCESS, connman);
371 
372  // Randomly charge clients
373  ChargeRandomFees(connman);
374 
375  // Reset
376  LogPrint("privatesend", "CPrivateSendServer::CommitFinalTransaction -- COMPLETED -- RESETTING\n");
377  SetNull();
378 }

380 //
381 // Charge clients a fee if they're abusive
382 //
383 // Why bother? PrivateSend uses collateral to ensure abuse to the process is kept to a minimum.
384 // The submission and signing stages are completely separate. In the cases where
385 // a client submits a transaction then refused to sign, there must be a cost. Otherwise they
386 // would be able to do this over and over again and bring the mixing to a halt.
387 //
388 // How does this work? Messages to Masternodes come in via NetMsgType::DSVIN, these require a valid collateral
389 // transaction for the client to be able to enter the pool. This transaction is kept by the Masternode
390 // until the transaction is either complete or fails.
391 //
392 void CPrivateSendServer::ChargeFees(CConnman& connman)
393 {
394  if (!fMasternodeMode) return;
395 
396  //we don't need to charge collateral for every offence.
397  if (GetRandInt(100) > 33) return;
398 
399  std::vector<CTransactionRef> vecOffendersCollaterals;
400 
401  if (nState == POOL_STATE_ACCEPTING_ENTRIES) {
402  for (const auto& txCollateral : vecSessionCollaterals) {
403  bool fFound = false;
404  for (const auto& entry : vecEntries)
405  if (*entry.txCollateral == *txCollateral)
406  fFound = true;
407 
408  // This queue entry didn't send us the promised transaction
409  if (!fFound) {
410  LogPrintf("CPrivateSendServer::ChargeFees -- found uncooperative node (didn't send transaction), found offence\n");
411  vecOffendersCollaterals.push_back(txCollateral);
412  }
413  }
414  }
415 
416  if (nState == POOL_STATE_SIGNING) {
417  // who didn't sign?
418  for (const auto& entry : vecEntries) {
419  for (const auto& txdsin : entry.vecTxDSIn) {
420  if (!txdsin.fHasSig) {
421  LogPrintf("CPrivateSendServer::ChargeFees -- found uncooperative node (didn't sign), found offence\n");
422  vecOffendersCollaterals.push_back(entry.txCollateral);
423  }
424  }
425  }
426  }
427 
428  // no offences found
429  if (vecOffendersCollaterals.empty()) return;
430 
431  //mostly offending? Charge sometimes
432  if ((int)vecOffendersCollaterals.size() >= Params().PoolMaxTransactions() - 1 && GetRandInt(100) > 33) return;
433 
434  //everyone is an offender? That's not right
435  if ((int)vecOffendersCollaterals.size() >= Params().PoolMaxTransactions()) return;
436 
437  //charge one of the offenders randomly
438  std::random_shuffle(vecOffendersCollaterals.begin(), vecOffendersCollaterals.end());
439 
440  if (nState == POOL_STATE_ACCEPTING_ENTRIES || nState == POOL_STATE_SIGNING) {
441  LogPrintf("CPrivateSendServer::ChargeFees -- found uncooperative node (didn't %s transaction), charging fees: %s",
442  (nState == POOL_STATE_SIGNING) ? "sign" : "send", vecOffendersCollaterals[0]->ToString());
443 
444  LOCK(cs_main);
445 
446  CValidationState state;
447  if (!AcceptToMemoryPool(mempool, state, vecOffendersCollaterals[0], false, NULL, false, maxTxFee)) {
448  // should never really happen
449  LogPrintf("CPrivateSendServer::ChargeFees -- ERROR: AcceptToMemoryPool failed!\n");
450  } else {
451  connman.RelayTransaction(*vecOffendersCollaterals[0]);
452  }
453  }
454 }
455 
456 /*
457  Charge the collateral randomly.
458  Mixing is completely free, to pay miners we randomly pay the collateral of users.
459 
460  Collateral Fee Charges:
461 
462  Being that mixing has "no fees" we need to have some kind of cost associated
463  with using it to stop abuse. Otherwise it could serve as an attack vector and
464  allow endless transaction that would bloat Dash and make it unusable. To
465  stop these kinds of attacks 1 in 10 successful transactions are charged. This
466  adds up to a cost of 0.001DRK per transaction on average.
467 */
468 void CPrivateSendServer::ChargeRandomFees(CConnman& connman)
469 {
470  if (!fMasternodeMode) return;
471 
472  LOCK(cs_main);
473 
474  for (const auto& txCollateral : vecSessionCollaterals) {
475  if (GetRandInt(100) > 10) return;
476  LogPrintf("CPrivateSendServer::ChargeRandomFees -- charging random fees, txCollateral=%s", txCollateral->ToString());
477 
478  CValidationState state;
479  if (!AcceptToMemoryPool(mempool, state, txCollateral, false, NULL, false, maxTxFee)) {
480  // should never really happen
481  LogPrintf("CPrivateSendServer::ChargeRandomFees -- ERROR: AcceptToMemoryPool failed!\n");
482  } else {
483  connman.RelayTransaction(*txCollateral);
484  }
485  }
486 }

527 // Check to make sure a given input matches an input in the pool and its scriptSig is valid
528 bool CPrivateSendServer::IsInputScriptSigValid(const CTxIn& txin)
529 {
530  CMutableTransaction txNew;
531  txNew.vin.clear();
532  txNew.vout.clear();
533 
534  int i = 0;
535  int nTxInIndex = -1;
536  CScript sigPubKey = CScript();
537 
538  for (const auto& entry : vecEntries) {
539  for (const auto& txout : entry.vecTxOut)
540  txNew.vout.push_back(txout);
541 
542  for (const auto& txdsin : entry.vecTxDSIn) {
543  txNew.vin.push_back(txdsin);
544 
545  if (txdsin.prevout == txin.prevout) {
546  nTxInIndex = i;
547  sigPubKey = txdsin.prevPubKey;
548  }
549  i++;
550  }
551  }
552 
553  if (nTxInIndex >= 0) { //might have to do this one input at a time?
554  txNew.vin[nTxInIndex].scriptSig = txin.scriptSig;
555  LogPrint("privatesend", "CPrivateSendServer::IsInputScriptSigValid -- verifying scriptSig %s\n", ScriptToAsmStr(txin.scriptSig).substr(0, 24));
556  if (!VerifyScript(txNew.vin[nTxInIndex].scriptSig, sigPubKey, SCRIPT_VERIFY_P2SH | SCRIPT_VERIFY_STRICTENC, MutableTransactionSignatureChecker(&txNew, nTxInIndex))) {
557  LogPrint("privatesend", "CPrivateSendServer::IsInputScriptSigValid -- VerifyScript() failed on input %d\n", nTxInIndex);
558  return false;
559  }
560  } else {
561  LogPrint("privatesend", "CPrivateSendServer::IsInputScriptSigValid -- Failed to find matching input in pool, %s\n", txin.ToString());
562  return false;
563  }

800 void CPrivateSendServer::RelayStatus(PoolStatusUpdate nStatusUpdate, CConnman& connman, PoolMessage nMessageID)
801 {
802  unsigned int nDisconnected{};
803  // status updates should be relayed to mixing participants only
804  for (const auto entry : vecEntries) {
805  // make sure everyone is still connected
806  bool fOk = connman.ForNode(entry.addr, [&nStatusUpdate, &nMessageID, &connman, this](CNode* pnode) {
807  PushStatus(pnode, nStatusUpdate, nMessageID, connman);
808  return true;
809  });
810  if (!fOk) {
811  // no such node? maybe this client disconnected or our own connection went down
812  ++nDisconnected;
813  }
814  }
815  if (nDisconnected == 0) return; // all is clear
816 
817  // smth went wrong
818  LogPrintf("CPrivateSendServer::%s -- can't continue, %llu client(s) disconnected, nSessionID: %d nSessionDenom: %d (%s)\n",
819  __func__, nDisconnected, nSessionID, nSessionDenom, CPrivateSend::GetDenominationsToString(nSessionDenom));
820 
821  // notify everyone else that this session should be terminated
822  for (const auto& entry : vecEntries) {
823  connman.ForNode(entry.addr, [&connman, this](CNode* pnode) {
824  PushStatus(pnode, STATUS_REJECTED, MSG_NOERR, connman);
825  return true;
826  });
827  }
828 
829  if (nDisconnected == vecEntries.size()) {
830  // all clients disconnected, there is probably some issues with our own connection
831  // do not charge any fees, just reset the pool
832  SetNull();
833  }
834 }
