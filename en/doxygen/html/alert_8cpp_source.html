126 bool CAlert::RelayTo(CNode* pnode, CConnman& connman) const
127 {
128  if (!IsInEffect())
129  return false;
130  // don't relay to nodes which haven't sent their version message
131  if (pnode->nVersion == 0)
132  return false;
133  // returns true if wasn't already contained in the set
134  if (pnode->setKnown.insert(GetHash()).second)
135  {
136  if (AppliesTo(pnode->nVersion, pnode->strSubVer) ||
137  AppliesToMe() ||
138  GetAdjustedTime() < nRelayUntil)
139  {
140  connman.PushMessage(pnode, CNetMsgMaker(pnode->GetSendVersion()).Make(NetMsgType::ALERT, *this));
141  return true;
142  }
143  }
144  return false;
145 }

192 bool CAlert::ProcessAlert(const std::vector<unsigned char>& alertKey, bool fThread) const
193 {
194  if (!CheckSignature(alertKey))
195  return false;
196  if (!IsInEffect())
197  return false;
198 
199  // alert.nID=max is reserved for if the alert key is
200  // compromised. It must have a pre-defined message,
201  // must never expire, must apply to all versions,
202  // and must cancel all previous
203  // alerts or it will be ignored (so an attacker can't
204  // send an "everything is OK, don't panic" version that
205  // cannot be overridden):
206  int maxInt = std::numeric_limits<int>::max();
207  if (nID == maxInt)
208  {
209  if (!(
210  nExpiration == maxInt &&
211  nCancel == (maxInt-1) &&
212  nMinVer == 0 &&
213  nMaxVer == maxInt &&
214  setSubVer.empty() &&
215  nPriority == maxInt &&
216  strStatusBar == "URGENT: Alert key compromised, upgrade required"
217  ))
218  return false;
219  }
220 
221  {
222  LOCK(cs_mapAlerts);
223  // Cancel previous alerts
224  for (std::map<uint256, CAlert>::iterator mi = mapAlerts.begin(); mi != mapAlerts.end();)
225  {
226  const CAlert& alert = (*mi).second;
227  if (Cancels(alert))
228  {
229  LogPrint("alert", "cancelling alert %d\n", alert.nID);
230  uiInterface.NotifyAlertChanged((*mi).first, CT_DELETED);
231  mapAlerts.erase(mi++);
232  }
233  else if (!alert.IsInEffect())
234  {
235  LogPrint("alert", "expiring alert %d\n", alert.nID);
236  uiInterface.NotifyAlertChanged((*mi).first, CT_DELETED);
237  mapAlerts.erase(mi++);
238  }
239  else
240  mi++;
241  }
242 
243  // Check if this alert has been cancelled
244  for (const auto& item : mapAlerts)
245  {
246  const CAlert& alert = item.second;
247  if (alert.Cancels(*this))
248  {
249  LogPrint("alert", "alert already cancelled by %d\n", alert.nID);
250  return false;
251  }
252  }
253 
254  // Add to mapAlerts
255  mapAlerts.insert(std::make_pair(GetHash(), *this));
256  // Notify UI and -alertnotify if it applies to me
257  if(AppliesToMe())
258  {
259  uiInterface.NotifyAlertChanged(GetHash(), CT_NEW);
260  Notify(strStatusBar, fThread);
261  }
262  }
263 
264  LogPrint("alert", "accepted alert %d, AppliesToMe()=%d\n", nID, AppliesToMe());
265  return true;
266 }

268 void
269 CAlert::Notify(const std::string& strMessage, bool fThread)
270 {
271  std::string strCmd = GetArg("-alertnotify", "");
272  if (strCmd.empty()) return;
273 
274  // Alert text should be plain ascii coming from a trusted source, but to
275  // be safe we first strip anything not in safeChars, then add single quotes around
276  // the whole string before passing it to the shell:
277  std::string singleQuote("'");
278  std::string safeStatus = SanitizeString(strMessage);
279  safeStatus = singleQuote+safeStatus+singleQuote;
280  boost::replace_all(strCmd, "%s", safeStatus);
281 
282  if (fThread)
283  boost::thread t(runCommand, strCmd); // thread runs free
284  else
285  runCommand(strCmd);
286 }
