117 bool BuildSimplifiedMNListDiff(const uint256& baseBlockHash, const uint256& blockHash, CSimplifiedMNListDiff& mnListDiffRet, std::string& errorRet)
118 {
119  AssertLockHeld(cs_main);
120  mnListDiffRet = CSimplifiedMNListDiff();
121 
122  const CBlockIndex* baseBlockIndex = chainActive.Genesis();
123  if (!baseBlockHash.IsNull()) {
124  auto it = mapBlockIndex.find(baseBlockHash);
125  if (it == mapBlockIndex.end()) {
126  errorRet = strprintf("block %s not found", baseBlockHash.ToString());
127  return false;
128  }
129  baseBlockIndex = it->second;
130  }
131  auto blockIt = mapBlockIndex.find(blockHash);
132  if (blockIt == mapBlockIndex.end()) {
133  errorRet = strprintf("block %s not found", blockHash.ToString());
134  return false;
135  }
136  const CBlockIndex* blockIndex = blockIt->second;
137 
138  if (!chainActive.Contains(baseBlockIndex) || !chainActive.Contains(blockIndex)) {
139  errorRet = strprintf("block %s and %s are not in the same chain", baseBlockHash.ToString(), blockHash.ToString());
140  return false;
141  }
142  if (baseBlockIndex->nHeight > blockIndex->nHeight) {
143  errorRet = strprintf("base block %s is higher then block %s", baseBlockHash.ToString(), blockHash.ToString());
144  return false;
145  }
146 
147  LOCK(deterministicMNManager->cs);
148 
149  auto baseDmnList = deterministicMNManager->GetListForBlock(baseBlockHash);
150  auto dmnList = deterministicMNManager->GetListForBlock(blockHash);
151  mnListDiffRet = baseDmnList.BuildSimplifiedDiff(dmnList);
152 
153  // TODO store coinbase TX in CBlockIndex
154  CBlock block;
155  if (!ReadBlockFromDisk(block, blockIndex, Params().GetConsensus())) {
156  errorRet = strprintf("failed to read block %s from disk", blockHash.ToString());
157  return false;
158  }
159 
160  mnListDiffRet.cbTx = block.vtx[0];
161 
162  std::vector<uint256> vHashes;
163  std::vector<bool> vMatch(block.vtx.size(), false);
164  for (const auto& tx : block.vtx) {
165  vHashes.emplace_back(tx->GetHash());
166  }
167  vMatch[0] = true; // only coinbase matches
168  mnListDiffRet.cbTxMerkleTree = CPartialMerkleTree(vHashes, vMatch);
169 
170  return true;
171 }
