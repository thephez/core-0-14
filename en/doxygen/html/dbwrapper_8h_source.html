353  struct KeyHolder {
354  virtual ~KeyHolder() = default;
355  virtual bool Less(const KeyHolder &b) const = 0;
356  virtual void Erase(CommitTarget &commitTarget) = 0;

358  typedef std::unique_ptr<KeyHolder> KeyHolderPtr;

360  template <typename K>
361  struct KeyHolderImpl : KeyHolder {
362  KeyHolderImpl(const K &_key)
363  : key(_key) {
364  }
365  virtual bool Less(const KeyHolder &b) const {
366  auto *b2 = dynamic_cast<const KeyHolderImpl<K>*>(&b);
367  return key < b2->key;
368  }
369  virtual void Erase(CommitTarget &commitTarget) {
370  commitTarget.Erase(key);
371  }
372  K key;
373  };

375  struct KeyValueHolder {
376  virtual ~KeyValueHolder() = default;
377  virtual void Write(CommitTarget &parent) = 0;
378  };
379  typedef std::unique_ptr<KeyValueHolder> KeyValueHolderPtr;
380 
381  template <typename K, typename V>
382  struct KeyValueHolderImpl : KeyValueHolder {
383  KeyValueHolderImpl(const KeyHolderImpl<K> &_key, const V &_value)
384  : key(_key),
385  value(_value) { }
386  KeyValueHolderImpl(const KeyHolderImpl<K> &_key, V &&_value)
387  : key(_key),
388  value(std::forward<V>(_value)) { }
389  virtual void Write(CommitTarget &commitTarget) {
390  // we're moving the value instead of copying it. This means that Write() can only be called once per
391  // KeyValueHolderImpl instance. Commit() clears the write maps, so this ok.
392  commitTarget.Write(key.key, std::move(value));
393  }
394  const KeyHolderImpl<K> &key;
395  V value;
396  };

432  template <typename K, typename KV>
433  void writeImpl(KeyHolderImpl<K>* k, KV&& kv) {
434  auto k2 = KeyHolderPtr(k);
435 
436  KeyValueMap *ds = getDeletesMap<K>(false);
437  if (ds)

439

462  template <typename K, typename V>
463  bool Read(const K& key, V& value) {
464  KeyHolderPtr k(new KeyHolderImpl<K>(key));
465 
466  KeyValueMap *ds = getDeletesMap<K>(false);
467  if (ds && ds->count(k))
468  return false;
469 
470  KeyValueMap *ws = getWritesMap<K>(false);
471  if (ws) {
472  auto it = ws->find(k);
473  if (it != ws->end()) {
474  auto *impl = dynamic_cast<KeyValueHolderImpl<K, V> *>(it->second.get());
475  if (!impl)
476  return false;
477  value = impl->value;
478  return true;
479  }
480  }
481 
482  return parent.Read(key, value);
483  }

516  void Commit() {
517  for (auto &p : deletes) {
